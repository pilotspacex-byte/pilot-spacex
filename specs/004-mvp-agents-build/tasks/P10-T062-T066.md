# backend/src/pilot_space/api/v1/schemas/workspace.py

from pydantic import BaseModel
from datetime import datetime


class ProviderStatus(BaseModel):
    """Status of a configured AI provider."""
    provider: str  # anthropic, openai, google
    is_configured: bool
    is_valid: bool | None  # None if never validated
    last_validated_at: datetime | None


class AIFeatureToggles(BaseModel):
    """AI feature toggles for workspace."""
    ghost_text_enabled: bool = True
    ai_context_enabled: bool = True
    pr_review_enabled: bool = True
    issue_extraction_enabled: bool = True
    margin_annotations_enabled: bool = True
    auto_approve_non_destructive: bool = True


class WorkspaceAISettingsResponse(BaseModel):
    """Response for workspace AI settings."""
    workspace_id: str
    providers: list[ProviderStatus]
    features: AIFeatureToggles
    default_provider: str = "anthropic"
    cost_limit_usd: float | None = None  # Monthly limit


# backend/src/pilot_space/api/v1/routers/workspaces.py

@router.get("/{workspace_id}/ai/settings", response_model=WorkspaceAISettingsResponse)
async def get_ai_settings(
    workspace_id: UUID,
    current_user = Depends(get_current_user),
    key_storage: SecureKeyStorage = Depends(get_key_storage),
):
    """Get workspace AI settings.

    Returns configured providers (not keys) and feature toggles.
    Requires workspace admin permission.
    """
    # Verify admin permission
    await verify_workspace_admin(current_user, workspace_id)

    # Get provider statuses
    providers = []
    for provider in ["anthropic", "openai", "google"]:
        key_info = await key_storage.get_key_info(workspace_id, provider)
        providers.append(ProviderStatus(
            provider=provider,
            is_configured=key_info is not None,
            is_valid=key_info.is_valid if key_info else None,
            last_validated_at=key_info.last_validated_at if key_info else None,
        ))

    # Get feature toggles from workspace settings
    features = await get_workspace_ai_features(workspace_id)

    return WorkspaceAISettingsResponse(
        workspace_id=str(workspace_id),
        providers=providers,
        features=features,
    )
```

# backend/src/pilot_space/api/v1/schemas/workspace.py

class APIKeyUpdate(BaseModel):
    """API key update for a provider."""
    provider: str
    api_key: str | None = None  # None to remove key


class WorkspaceAISettingsUpdate(BaseModel):
    """Request to update workspace AI settings."""
    api_keys: list[APIKeyUpdate] | None = None
    features: AIFeatureToggles | None = None
    cost_limit_usd: float | None = None


class KeyValidationResult(BaseModel):
    """Result of API key validation."""
    provider: str
    is_valid: bool
    error_message: str | None = None


class WorkspaceAISettingsUpdateResponse(BaseModel):
    """Response after updating AI settings."""
    success: bool
    validation_results: list[KeyValidationResult]
    updated_providers: list[str]
    updated_features: bool


# backend/src/pilot_space/api/v1/routers/workspaces.py

@router.put("/{workspace_id}/ai/settings", response_model=WorkspaceAISettingsUpdateResponse)
async def update_ai_settings(
    workspace_id: UUID,
    body: WorkspaceAISettingsUpdate,
    current_user = Depends(get_current_user),
    key_storage: SecureKeyStorage = Depends(get_key_storage),
):
    """Update workspace AI settings.

    Validates API keys before saving. Keys are encrypted with Supabase Vault.
    Requires workspace admin permission.
    """
    await verify_workspace_admin(current_user, workspace_id)

    validation_results = []
    updated_providers = []

    # Process API key updates
    if body.api_keys:
        for key_update in body.api_keys:
            if key_update.api_key:
                # Validate key
                is_valid, error = await validate_api_key(
                    key_update.provider,
                    key_update.api_key,
                )
                validation_results.append(KeyValidationResult(
                    provider=key_update.provider,
                    is_valid=is_valid,
                    error_message=error,
                ))

                if is_valid:
                    # Store encrypted key
                    await key_storage.store_key(
                        workspace_id=workspace_id,
                        provider=key_update.provider,
                        api_key=key_update.api_key,
                    )
                    updated_providers.append(key_update.provider)
            else:
                # Remove key
                await key_storage.delete_key(workspace_id, key_update.provider)
                updated_providers.append(key_update.provider)

    # Update feature toggles
    updated_features = False
    if body.features:
        await update_workspace_ai_features(workspace_id, body.features)
        updated_features = True

    return WorkspaceAISettingsUpdateResponse(
        success=all(r.is_valid for r in validation_results),
        validation_results=validation_results,
        updated_providers=updated_providers,
        updated_features=updated_features,
    )


async def validate_api_key(provider: str, api_key: str) -> tuple[bool, str | None]:
    """Validate API key by making a test API call."""
    try:
        if provider == "anthropic":
            from anthropic import Anthropic
            client = Anthropic(api_key=api_key)
            # Minimal API call to validate
            await client.messages.create(
                model="claude-3-5-haiku-20241022",
                max_tokens=1,
                messages=[{"role": "user", "content": "test"}],
            )
        elif provider == "openai":
            from openai import OpenAI
            client = OpenAI(api_key=api_key)
            await client.models.list()
        elif provider == "google":
            import google.generativeai as genai
            genai.configure(api_key=api_key)
            # List models to validate
            list(genai.list_models())
        return True, None
    except Exception as e:
        return False, str(e)
```

import pytest
from unittest.mock import AsyncMock, MagicMock, patch
from uuid import uuid4

from pilot_space.ai.infrastructure.key_storage import SecureKeyStorage


@pytest.fixture
def mock_supabase():
    return MagicMock()


@pytest.fixture
def key_storage(mock_supabase):
    return SecureKeyStorage(supabase_client=mock_supabase)


class TestSecureKeyStorage:
    @pytest.mark.asyncio
    async def test_store_key_encrypts_with_vault(self, key_storage, mock_supabase):
        workspace_id = uuid4()
        mock_supabase.rpc.return_value.execute = AsyncMock(return_value=MagicMock(data={"id": "encrypted-ref"}))

        await key_storage.store_key(
            workspace_id=workspace_id,
            provider="anthropic",
            api_key="sk-ant-test-key",
        )

        # Verify Vault encryption was called
        mock_supabase.rpc.assert_called_with(
            "vault.create_secret",
            {"secret": "sk-ant-test-key", "name": f"api_key_{workspace_id}_anthropic"}
        )

    @pytest.mark.asyncio
    async def test_get_key_decrypts_from_vault(self, key_storage, mock_supabase):
        workspace_id = uuid4()
        mock_supabase.rpc.return_value.execute = AsyncMock(
            return_value=MagicMock(data={"decrypted_secret": "sk-ant-test-key"})
        )

        key = await key_storage.get_key(workspace_id, "anthropic")

        assert key == "sk-ant-test-key"

    @pytest.mark.asyncio
    async def test_get_key_returns_none_if_not_found(self, key_storage, mock_supabase):
        workspace_id = uuid4()
        mock_supabase.rpc.return_value.execute = AsyncMock(return_value=MagicMock(data=None))

        key = await key_storage.get_key(workspace_id, "anthropic")

        assert key is None

    @pytest.mark.asyncio
    async def test_delete_key_removes_from_vault(self, key_storage, mock_supabase):
        workspace_id = uuid4()

        await key_storage.delete_key(workspace_id, "anthropic")

        mock_supabase.rpc.assert_called()
