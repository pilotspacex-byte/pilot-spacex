```typescript
/**
 * AI Context Panel - Main container for AI context display.
 */
import { observer } from 'mobx-react-lite';
import { useStore } from '@/stores/use-store';
import { AIContextStreaming } from './ai-context-streaming';
import { ClaudeCodePromptCard } from './claude-code-prompt-card';
import { RelatedItemsList } from './related-items-list';
import { Button } from '@/components/ui/button';
import { Loader2, RefreshCw, Sparkles } from 'lucide-react';

interface AIContextPanelProps {
  issueId: string;
}

export const AIContextPanel = observer(function AIContextPanel({
  issueId,
}: AIContextPanelProps) {
  const { ai } = useStore();
  const { aiContext } = ai;
  const { isLoading, error, phases, result } = aiContext;

  const handleGenerate = () => {
    aiContext.generateContext(issueId);
  };

  const handleRefresh = () => {
    aiContext.clearCache(issueId);
    aiContext.generateContext(issueId);
  };

  if (!ai.settings.aiContextEnabled) {
    return (
      <div className="p-4 text-center text-muted-foreground">
        <p>AI Context is disabled for this workspace.</p>
        <p className="text-sm mt-2">
          Enable it in{' '}
          <a href="/settings/ai" className="text-primary underline">
            AI Settings
          </a>
        </p>
      </div>
    );
  }

  return (
    <div className="ai-context-panel space-y-4">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <Sparkles className="h-5 w-5 text-primary" />
          <h3 className="font-semibold">AI Context</h3>
        </div>

        {result && (
          <Button variant="ghost" size="sm" onClick={handleRefresh}>
            <RefreshCw className="h-4 w-4 mr-1" />
            Refresh
          </Button>
        )}
      </div>

      {/* Initial state */}
      {!isLoading && !result && !error && (
        <div className="text-center py-8">
          <p className="text-muted-foreground mb-4">
            Generate AI-powered context for this issue including related docs,
            code patterns, and implementation guidance.
          </p>
          <Button onClick={handleGenerate}>
            <Sparkles className="h-4 w-4 mr-2" />
            Generate AI Context
          </Button>
        </div>
      )}

      {/* Loading state with streaming phases */}
      {isLoading && <AIContextStreaming phases={phases} />}

      {/* Error state */}
      {error && (
        <div className="p-4 bg-destructive/10 text-destructive rounded-lg">
          <p>{error}</p>
          <Button variant="outline" size="sm" className="mt-2" onClick={handleGenerate}>
            Try Again
          </Button>
        </div>
      )}

      {/* Result display */}
      {result && (
        <div className="space-y-4">
          {/* Claude Code Prompt */}
          <ClaudeCodePromptCard prompt={result.claudeCodePrompt} />

          {/* Related Documents */}
          {result.relatedDocs.length > 0 && (
            <RelatedItemsList
              title="Related Documents"
              items={result.relatedDocs}
              icon="file-text"
            />
          )}

          {/* Related Code */}
          {result.relatedCode.length > 0 && (
            <RelatedItemsList
              title="Related Code"
              items={result.relatedCode}
              icon="code"
            />
          )}

          {/* Similar Issues */}
          {result.similarIssues.length > 0 && (
            <RelatedItemsList
              title="Similar Issues"
              items={result.similarIssues}
              icon="git-pull-request"
            />
          )}
        </div>
      )}
    </div>
  );
});
```

```typescript
/**
 * AI Context Streaming - Shows progress through 5 analysis phases.
 */
import { observer } from 'mobx-react-lite';
import { Check, Loader2 } from 'lucide-react';
import { cn } from '@/lib/utils';
import type { AIContextPhase } from '@/stores/ai-context-store';

interface AIContextStreamingProps {
  phases: AIContextPhase[];
}

export const AIContextStreaming = observer(function AIContextStreaming({
  phases,
}: AIContextStreamingProps) {
  return (
    <div className="space-y-3 py-4">
      <p className="text-sm text-muted-foreground">
        Analyzing issue and gathering context...
      </p>

      <div className="space-y-2">
        {phases.map((phase, index) => (
          <PhaseItem key={index} phase={phase} />
        ))}
      </div>
    </div>
  );
});

interface PhaseItemProps {
  phase: AIContextPhase;
}

function PhaseItem({ phase }: PhaseItemProps) {
  const { name, status, content } = phase;

  return (
    <div
      className={cn(
        'flex items-start gap-3 p-3 rounded-lg border',
        status === 'complete' && 'bg-primary/5 border-primary/20',
        status === 'in_progress' && 'bg-muted border-primary animate-pulse',
        status === 'pending' && 'bg-muted/50 border-muted'
      )}
    >
      {/* Status icon */}
      <div className="mt-0.5">
        {status === 'complete' && (
          <Check className="h-4 w-4 text-primary" />
        )}
        {status === 'in_progress' && (
          <Loader2 className="h-4 w-4 text-primary animate-spin" />
        )}
        {status === 'pending' && (
          <div className="h-4 w-4 rounded-full border-2 border-muted-foreground/30" />
        )}
      </div>

      {/* Phase content */}
      <div className="flex-1 min-w-0">
        <p
          className={cn(
            'text-sm font-medium',
            status === 'pending' && 'text-muted-foreground'
          )}
        >
          {name}
        </p>

        {/* Phase result preview */}
        {content && status === 'complete' && (
          <p className="text-xs text-muted-foreground mt-1 truncate">
            {content}
          </p>
        )}
      </div>
    </div>
  );
}
```

```typescript
/**
 * Claude Code Prompt Card - Displays prompt with copy functionality.
 */
import { useState } from 'react';
import { observer } from 'mobx-react-lite';
import { Button } from '@/components/ui/button';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Check, Copy, Terminal } from 'lucide-react';
import { cn } from '@/lib/utils';

interface ClaudeCodePromptCardProps {
  prompt: string;
}

export const ClaudeCodePromptCard = observer(function ClaudeCodePromptCard({
  prompt,
}: ClaudeCodePromptCardProps) {
  const [copied, setCopied] = useState(false);
  const [expanded, setExpanded] = useState(false);

  const handleCopy = async () => {
    await navigator.clipboard.writeText(prompt);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const previewLength = 300;
  const needsTruncation = prompt.length > previewLength;
  const displayText = expanded ? prompt : prompt.slice(0, previewLength);

  return (
    <Card className="border-primary/20 bg-primary/5">
      <CardHeader className="pb-2">
        <CardTitle className="flex items-center justify-between text-base">
          <div className="flex items-center gap-2">
            <Terminal className="h-4 w-4" />
            Claude Code Prompt
          </div>
          <Button
            variant="outline"
            size="sm"
            onClick={handleCopy}
            className="h-8"
          >
            {copied ? (
              <>
                <Check className="h-3 w-3 mr-1" />
                Copied
              </>
            ) : (
              <>
                <Copy className="h-3 w-3 mr-1" />
                Copy
              </>
            )}
          </Button>
        </CardTitle>
      </CardHeader>
      <CardContent>
        <pre
          className={cn(
            'text-xs bg-muted p-3 rounded-md overflow-x-auto',
            'font-mono whitespace-pre-wrap'
          )}
        >
          {displayText}
          {needsTruncation && !expanded && '...'}
        </pre>

        {needsTruncation && (
          <Button
            variant="link"
            size="sm"
            className="mt-2 h-auto p-0"
            onClick={() => setExpanded(!expanded)}
          >
            {expanded ? 'Show less' : 'Show full prompt'}
          </Button>
        )}

        <p className="text-xs text-muted-foreground mt-3">
          Paste this prompt into Claude Code to get started with implementation.
        </p>
      </CardContent>
    </Card>
  );
});
```

```typescript
    store.generateContext('issue-1');

    expect(store.isLoading).toBe(true);
    expect(store.phases).toHaveLength(5);

  it('should cache results per issue', async () => {
    const mockResult = {
      phases: [],
      claudeCodePrompt: 'test prompt',
      relatedDocs: [],
      relatedCode: [],
      similarIssues: [],
    };

    store['cache'].set('issue-1', mockResult);

    await store.generateContext('issue-1');

    expect(store.result).toEqual(mockResult);
    expect(store.isLoading).toBe(false);

  it('should clear specific issue from cache', () => {
    store['cache'].set('issue-1', {} as any);

    store.clearCache('issue-1');

    expect(store['cache'].has('issue-2')).toBe(true);
  });
```
