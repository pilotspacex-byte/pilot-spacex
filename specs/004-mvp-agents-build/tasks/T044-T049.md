## Architecture Decision: Claude Agent SDK with Haiku

Per DD-011 routing update, GhostTextAgent now uses Claude Agent SDK with `claude-3-5-haiku-20241022`:

| Attribute | Value | Rationale |
|-----------|-------|-----------|
| **SDK Method** | `query()` | One-shot, no multi-turn needed |
| **Model** | `claude-3-5-haiku-20241022` | Fastest Claude model (<1s latency) |
| **Max Tokens** | 50 | Short suggestions only |
| **Timeout** | 2000ms | Hard limit for UX |
| **Fallback** | Empty suggestion | Graceful degradation |

**Why SDK over direct Gemini API**:
- Unified architecture (all agents use SDK)
- Built-in retry and rate limiting
- Consistent cost tracking via SDK telemetry
- Haiku provides comparable latency to Gemini Flash for short outputs

### Implementation Guidelines

#### Structure

```python
from claude_agent_sdk import query, ClaudeAgentOptions
from claude_agent_sdk.types import AssistantMessage, TextBlock
from pilot_space.ai.agents.sdk_base import SDKBaseAgent
import asyncio

class GhostTextAgent(SDKBaseAgent):
    """Real-time ghost text suggestions using Claude Haiku.

    Optimized for <2s latency with graceful degradation.
    """

    AGENT_NAME = "ghost_text"
    DEFAULT_MODEL = "claude-3-5-haiku-20241022"
    MAX_TOKENS = 50
    TIMEOUT_SECONDS = 2.0
    MAX_BUDGET_USD = 0.01  # Very small budget per request

    async def execute(
        self,
        input_data: GhostTextInput,
        user_id: str,
        workspace_id: str,
    ) -> AsyncIterator[str]:
        """Generate ghost text with streaming output."""
        try:
            async with asyncio.timeout(self.TIMEOUT_SECONDS):
                async for chunk in self._generate(input_data):
                    yield chunk
        except asyncio.TimeoutError:
            logger.warning("ghost_text_timeout", input_length=len(input_data.current_text))
            yield ""  # Empty suggestion
        except Exception as e:
            logger.error("ghost_text_error", error=str(e))
            yield ""

    async def _generate(self, input_data: GhostTextInput) -> AsyncIterator[str]:
        """Internal generation with SDK."""
        prompt = self._build_prompt(input_data)
        system_prompt = (
            self.CODE_SYSTEM_PROMPT if input_data.is_code
            else self.PROSE_SYSTEM_PROMPT
        )

        options = self._build_options(
            system_prompt=system_prompt,
            model=self.DEFAULT_MODEL,
            max_budget_usd=self.MAX_BUDGET_USD,
        )

        accumulated = ""
        async for message in query(prompt=prompt, options=options):
            if isinstance(message, AssistantMessage):
                for block in message.content:
                    if isinstance(block, TextBlock):
                        text = block.text
                        accumulated += text
                        yield text

                        # Stop at word boundary (DD-067)
                        if self._should_stop(accumulated):
                            return

    def _should_stop(self, text: str) -> bool:
        """Check if we should stop at word boundary."""
        stop_chars = [".", "!", "?", "\n", ":", ";"]
        return (
            len(text) > 20 and
            any(char in text for char in stop_chars)
        )

    def _build_prompt(self, input_data: GhostTextInput) -> str:
        """Build completion prompt."""
        context = input_data.current_text[-2000:]  # Truncate to last 2000 chars
        return f"Continue this text naturally:\n\n{context}"
```

Optimized for <50 token completions with <2s latency.

GHOST_TEXT_PROSE_PROMPT = """Continue this text naturally. Output ONLY the continuation (1-2 sentences max). Stop at a period or question mark.

Text to continue:
{context}"""

GHOST_TEXT_CODE_PROMPT = """Complete this code. Output ONLY the code continuation (1-3 lines max). Match the existing style.

Code to complete:
```{language}
{context}
```"""

def build_ghost_text_prompt(
    context: str,
    is_code: bool = False,
    language: str | None = None,
) -> str:
    """Build prompt for ghost text generation."""
    if is_code:
        return GHOST_TEXT_CODE_PROMPT.format(
            context=context[-2000:],
            language=language or "python",
        )
    return GHOST_TEXT_PROSE_PROMPT.format(context=context[-2000:])

```python
from fastapi import APIRouter, Depends
from fastapi.responses import StreamingResponse
from pydantic import BaseModel, Field

router = APIRouter(prefix="/ai", tags=["AI"])

class GhostTextRequest(BaseModel):
    """Request body for ghost text generation."""
    context: str = Field(..., max_length=5000, description="Text before cursor")
    cursor_position: int = Field(..., ge=0)
    is_code: bool = Field(default=False)
    language: str | None = Field(default=None)
    block_id: str | None = Field(default=None)

@router.post("/notes/{note_id}/ghost-text")
async def stream_ghost_text(
    note_id: str,
    request: GhostTextRequest,
    current_user: User = Depends(get_current_user),
    orchestrator: SDKOrchestrator = Depends(get_sdk_orchestrator),
) -> StreamingResponse:
    """Stream ghost text suggestions via SSE.

    Returns streaming text completions for the current cursor position.
    Optimized for <2s latency with graceful timeout handling.
    """
    return StreamingResponse(
        create_sse_stream(
            orchestrator.agents["ghost_text"].execute(
                input_data=GhostTextInput(
                    current_text=request.context,
                    cursor_position=request.cursor_position,
                    is_code=request.is_code,
                    language=request.language,
                    note_id=note_id,
                    block_id=request.block_id,
                ),
                user_id=str(current_user.id),
                workspace_id=str(current_user.workspace_id),
            ),
            event_type="ghost_text",
        ),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",
        },
    )
```

import pytest
import time
import statistics
from httpx import AsyncClient

@pytest.mark.performance
@pytest.mark.asyncio
async def test_ghost_text_latency_p95(
    async_client: AsyncClient,
    auth_headers: dict,
):
    """Test ghost text latency meets <2s p95 target."""
    latencies = []
    first_token_latencies = []

    test_contexts = [
        "The quick brown fox " * 5,   # ~100 chars
        "The quick brown fox " * 25,  # ~500 chars
        "The quick brown fox " * 50,  # ~1000 chars
        "The quick brown fox " * 100, # ~2000 chars
    ]

    for context in test_contexts:
        for _ in range(10):  # 10 iterations per size
            start = time.perf_counter()
            first_token_time = None

            async with async_client.stream(
                "POST",
                "/api/v1/ai/notes/test-note/ghost-text",
                json={
                    "context": context,
                    "cursor_position": len(context),
                    "is_code": False,
                },
                headers=auth_headers,
            ) as response:
                async for line in response.aiter_lines():
                    if first_token_time is None and line.startswith("data:"):
                        first_token_time = time.perf_counter() - start
                        first_token_latencies.append(first_token_time * 1000)

            total_latency = (time.perf_counter() - start) * 1000
            latencies.append(total_latency)

    # Calculate p95
    p95_latency = statistics.quantiles(latencies, n=20)[18]  # 95th percentile
    p95_first_token = statistics.quantiles(first_token_latencies, n=20)[18]

    print(f"Total latency p95: {p95_latency:.0f}ms")
    print(f"First token p95: {p95_first_token:.0f}ms")
    print(f"Median latency: {statistics.median(latencies):.0f}ms")

    assert p95_latency < 2000, f"p95 latency {p95_latency}ms exceeds 2000ms target"
    assert p95_first_token < 500, f"First token p95 {p95_first_token}ms exceeds 500ms target"
```
