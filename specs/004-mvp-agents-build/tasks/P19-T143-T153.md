/**
 * PR Review Panel - Main container for AI PR review display.
 */
import { observer } from 'mobx-react-lite';
import { useStore } from '@/stores/use-store';
import { PRReviewStreaming } from './pr-review-streaming';
import { ReviewAspectCard } from './review-aspect-card';
import { PRReviewCostBadge } from './pr-review-cost-badge';
import { Button } from '@/components/ui/button';
import { Loader2, RefreshCw, Sparkles, AlertCircle } from 'lucide-react';

interface PRReviewPanelProps {
  repoId: string;
  prNumber: number;
}

export const PRReviewPanel = observer(function PRReviewPanel({
  repoId,
  prNumber,
}: PRReviewPanelProps) {
  const { ai } = useStore();
  const { prReview } = ai;
  const { isLoading, error, aspects, result, tokenUsage } = prReview;

  const handleRequestReview = () => {
    prReview.requestReview(repoId, prNumber);
  };

  const handleRefresh = () => {
    prReview.clearCache(`${repoId}:${prNumber}`);
    prReview.requestReview(repoId, prNumber);
  };

  return (
    <div className="pr-review-panel space-y-4">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <Sparkles className="h-5 w-5 text-primary" />
          <h3 className="font-semibold">AI Code Review</h3>
          {tokenUsage && <PRReviewCostBadge usage={tokenUsage} />}
        </div>

        {result && (
          <Button variant="ghost" size="sm" onClick={handleRefresh}>
            <RefreshCw className="h-4 w-4 mr-1" />
            Re-review
          </Button>
        )}
      </div>

      {/* Initial state */}
      {!isLoading && !result && !error && (
        <div className="text-center py-8 border rounded-lg bg-muted/30">
          <Sparkles className="h-12 w-12 mx-auto mb-4 text-muted-foreground" />
          <p className="text-muted-foreground mb-4">
            Get comprehensive AI-powered code review covering architecture,
            security, code quality, performance, and documentation.
          </p>
          <Button onClick={handleRequestReview}>
            <Sparkles className="h-4 w-4 mr-2" />
            Request AI Review
          </Button>
        </div>
      )}

      {/* Loading state with streaming */}
      {isLoading && <PRReviewStreaming aspects={aspects} />}

      {/* Error state */}
      {error && (
        <div className="p-4 bg-destructive/10 text-destructive rounded-lg flex items-start gap-3">
          <AlertCircle className="h-5 w-5 mt-0.5" />
          <div>
            <p>{error}</p>

        <div className="space-y-4">

            <p className="text-sm text-muted-foreground mt-1">{result.summary}</p>
          </div>

          {/* 5 Aspect Cards */}
          <div className="grid gap-4">
            <ReviewAspectCard
              aspect="architecture"
              title="Architecture"
              icon="layout"
              findings={result.architecture}
            />
            <ReviewAspectCard
              aspect="security"
              title="Security"
              icon="shield"
              findings={result.security}

import { observer } from 'mobx-react-lite';
import { Check, Loader2, Layout, Shield, Code, Zap, FileText } from 'lucide-react';
import { cn } from '@/lib/utils';

interface ReviewAspect {
  name: string;
  status: 'pending' | 'in_progress' | 'complete';
}

interface PRReviewStreamingProps {
  aspects: ReviewAspect[];
}

const aspectIcons = {
  architecture: Layout,
  security: Shield,
  quality: Code,
  performance: Zap,
  documentation: FileText,
};

export const PRReviewStreaming = observer(function PRReviewStreaming({
  aspects,
}: PRReviewStreamingProps) {
  return (
    <div className="space-y-3 py-4">
      <p className="text-sm text-muted-foreground">
        Analyzing pull request...
      </p>

      <div className="space-y-2">
        {aspects.map((aspect) => (
          <AspectStreamItem key={aspect.name} aspect={aspect} />
        ))}
      </div>
    </div>
  );
});

interface AspectStreamItemProps {
  aspect: ReviewAspect;
}

function AspectStreamItem({ aspect }: AspectStreamItemProps) {
  const { name, status } = aspect;
  const Icon = aspectIcons[name.toLowerCase() as keyof typeof aspectIcons] || Code;

  return (
    <div
      className={cn(
        'flex items-center gap-3 p-3 rounded-lg border',
        status === 'complete' && 'bg-primary/5 border-primary/20',
        status === 'in_progress' && 'bg-muted border-primary animate-pulse',
        status === 'pending' && 'bg-muted/50 border-muted'
      )}
    >
      {/* Aspect icon */}
      <Icon
        className={cn(
          'h-4 w-4',
          status === 'complete' && 'text-primary',
          status === 'in_progress' && 'text-primary',
          status === 'pending' && 'text-muted-foreground/50'
        )}
      />

      {/* Aspect name */}
      <span
        className={cn(
          'flex-1 text-sm font-medium',
          status === 'pending' && 'text-muted-foreground'
        )}
      >
        {name}
      </span>

      {/* Status indicator */}
      <div>
        {status === 'complete' && (
          <Check className="h-4 w-4 text-primary" />
        )}
        {status === 'in_progress' && (
          <Loader2 className="h-4 w-4 text-primary animate-spin" />
        )}
      </div>
    </div>
  );
}

/**
 * Review Aspect Card - Displays findings for one review aspect.
 */
import { useState } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { ChevronDown, ChevronUp, Layout, Shield, Code, Zap, FileText } from 'lucide-react';
import { cn } from '@/lib/utils';

const icons = {
  layout: Layout,
  shield: Shield,
  code: Code,
  zap: Zap,
  'file-text': FileText,
};

interface Finding {
  severity: 'critical' | 'warning' | 'info' | 'success';
  message: string;
  file?: string;
  line?: number;
  suggestion?: string;
}

interface ReviewAspectCardProps {
  aspect: string;
  title: string;
  icon: keyof typeof icons;
  findings: Finding[];
}

const severityColors = {
  critical: 'bg-destructive text-destructive-foreground',
  warning: 'bg-yellow-500/10 text-yellow-600',
  info: 'bg-blue-500/10 text-blue-600',
  success: 'bg-green-500/10 text-green-600',
};

export function ReviewAspectCard({
  aspect,
  title,
  icon,
  findings,
}: ReviewAspectCardProps) {
  const [expanded, setExpanded] = useState(true);
  const Icon = icons[icon];

  const criticalCount = findings.filter(f => f.severity === 'critical').length;
  const warningCount = findings.filter(f => f.severity === 'warning').length;

  return (
    <Card>
      <CardHeader className="pb-2">
        <CardTitle
          className="flex items-center justify-between text-base cursor-pointer"
          onClick={() => setExpanded(!expanded)}
        >
          <div className="flex items-center gap-2">
            <Icon className="h-4 w-4" />
            {title}
            <div className="flex gap-1 ml-2">
              {criticalCount > 0 && (
                <Badge variant="destructive" className="h-5 px-1.5">
                  {criticalCount}
                </Badge>
              )}
              {warningCount > 0 && (
                <Badge variant="secondary" className="h-5 px-1.5 bg-yellow-500/10 text-yellow-600">
                  {warningCount}
                </Badge>
              )}
            </div>
          </div>
          <Button variant="ghost" size="sm" className="h-6 w-6 p-0">
            {expanded ? (
              <ChevronUp className="h-4 w-4" />
            ) : (
              <ChevronDown className="h-4 w-4" />
            )}
          </Button>
        </CardTitle>
      </CardHeader>

      {expanded && (
        <CardContent className="space-y-2">
          {findings.length === 0 ? (
            <p className="text-sm text-muted-foreground">No issues found</p>
          ) : (
            findings.map((finding, index) => (
              <FindingItem key={index} finding={finding} />
            ))
          )}
        </CardContent>
      )}
    </Card>
  );
}

function FindingItem({ finding }: { finding: Finding }) {
  return (
    <div
      className={cn(
        'p-3 rounded-md text-sm',
        severityColors[finding.severity]
      )}
    >
      <p className="font-medium">{finding.message}</p>

      {finding.file && (
        <p className="text-xs mt-1 opacity-70">
          {finding.file}
          {finding.line && `:${finding.line}`}
        </p>
      )}

      {finding.suggestion && (
        <p className="mt-2 text-xs">
          <span className="font-medium">Suggestion:</span> {finding.suggestion}
        </p>
      )}
    </div>
  );
}

/**
 * PR Review Cost Badge - Shows token usage and estimated cost.
 */
import { Badge } from '@/components/ui/badge';
import { Tooltip, TooltipContent, TooltipTrigger } from '@/components/ui/tooltip';
import { Coins } from 'lucide-react';

interface TokenUsage {
  inputTokens: number;
  outputTokens: number;
  estimatedCostUsd: number;
}

interface PRReviewCostBadgeProps {
  usage: TokenUsage;
}

export function PRReviewCostBadge({ usage }: PRReviewCostBadgeProps) {
  const formatCost = (cost: number) => {
    if (cost < 0.01) return '<$0.01';
    return `$${cost.toFixed(2)}`;
  };

  const formatTokens = (tokens: number) => {
    if (tokens >= 1000) return `${(tokens / 1000).toFixed(1)}k`;
    return tokens.toString();
  };

  return (
    <Tooltip>
      <TooltipTrigger>
        <Badge variant="outline" className="text-xs gap-1">
          <Coins className="h-3 w-3" />
          {formatCost(usage.estimatedCostUsd)}
        </Badge>
      </TooltipTrigger>
      <TooltipContent>
        <div className="text-xs space-y-1">
          <p>Input: {formatTokens(usage.inputTokens)} tokens</p>
          <p>Output: {formatTokens(usage.outputTokens)} tokens</p>
          <p>Estimated cost: {formatCost(usage.estimatedCostUsd)}</p>
        </div>
      </TooltipContent>
    </Tooltip>
  );
}

// frontend/src/stores/pr-review-store.ts
/**
 * PR Review Store - State management for AI PR reviews.
 */
import { makeAutoObservable, runInAction } from 'mobx';
import { SSEClient, SSEEvent } from '@/lib/sse-client';
import { aiApi } from '@/services/ai-api';
import type { AIStore } from './ai-store';

export interface ReviewAspect {
  name: string;
  status: 'pending' | 'in_progress' | 'complete';
}

export interface ReviewFinding {
  severity: 'critical' | 'warning' | 'info' | 'success';
  message: string;
  file?: string;
  line?: number;
  suggestion?: string;
}

export interface PRReviewResult {
  summary: string;
  architecture: ReviewFinding[];
  security: ReviewFinding[];
  quality: ReviewFinding[];
  performance: ReviewFinding[];
  documentation: ReviewFinding[];
}

export interface TokenUsage {
  inputTokens: number;
  outputTokens: number;
  estimatedCostUsd: number;
}

export class PRReviewStore {
  isLoading = false;
  error: string | null = null;
  aspects: ReviewAspect[] = [];
  result: PRReviewResult | null = null;
  tokenUsage: TokenUsage | null = null;

  private client: SSEClient | null = null;
  private cache = new Map<string, PRReviewResult>();

  constructor(private rootStore: AIStore) {
    makeAutoObservable(this);
  }

  async requestReview(repoId: string, prNumber: number): Promise<void> {
    const cacheKey = `${repoId}:${prNumber}`;

    // Check cache
    const cached = this.cache.get(cacheKey);
    if (cached) {
      this.result = cached;
      return;
    }

    this.abort();

    runInAction(() => {
      this.isLoading = true;
      this.error = null;
      this.result = null;
      this.tokenUsage = null;
      this.aspects = [
        { name: 'Architecture', status: 'pending' },
        { name: 'Security', status: 'pending' },
        { name: 'Quality', status: 'pending' },
        { name: 'Performance', status: 'pending' },
        { name: 'Documentation', status: 'pending' },
      ];
    });

    this.client = new SSEClient({
      url: aiApi.getPRReviewUrl(repoId, prNumber),
      onMessage: (event: SSEEvent) => this.handleEvent(event),
      onComplete: () => {
        runInAction(() => {
          this.isLoading = false;
          if (this.result) {
            this.cache.set(cacheKey, this.result);
          }
        });
      },
      onError: (err) => {
        runInAction(() => {
          this.isLoading = false;
          this.error = err.message;
        });
      },
    });

    await this.client.connect();
  }

  private handleEvent(event: SSEEvent): void {
    runInAction(() => {
      const data = event.data as Record<string, unknown>;

      switch (event.type) {
        case 'aspect':
          const aspectName = data.name as string;
          const aspectIndex = this.aspects.findIndex(
            a => a.name.toLowerCase() === aspectName.toLowerCase()
          );
          if (aspectIndex >= 0) {
            this.aspects[aspectIndex].status = data.status as 'pending' | 'in_progress' | 'complete';
          }
          break;

        case 'complete':
          this.result = data.result as PRReviewResult;
          this.tokenUsage = data.usage as TokenUsage;
          break;
      }
    });
  }

  abort(): void {
    this.client?.abort();
    this.client = null;
    this.isLoading = false;
  }

  clearCache(key?: string): void {
    if (key) {
      this.cache.delete(key);
    } else {
      this.cache.clear();
    }
  }
}
