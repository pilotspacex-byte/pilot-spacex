```typescript
/**
 * TipTap Extension for Ghost Text Suggestions.
 *
 * Provides:
 * - Debounced suggestion requests on typing pause
 * - Tab to accept, Esc to dismiss
 * - Gray ghost text decoration
 */
import { Extension } from '@tiptap/core';
import { Plugin, PluginKey } from '@tiptap/pm/state';
import { Decoration, DecorationSet } from '@tiptap/pm/view';
import { getAIStore } from '@/stores/ai-store';

export interface GhostTextOptions {
  debounceMs: number;
  contextChars: number;
  enabled: boolean;
}

export const ghostTextPluginKey = new PluginKey('ghostText');

export const GhostTextExtension = Extension.create<GhostTextOptions>({
  name: 'ghostText',

  addOptions() {
    return {
      debounceMs: 500,
      contextChars: 500,
      enabled: true,
    };
  },

  addKeyboardShortcuts() {
    return {
      Tab: () => {
        const store = getAIStore().ghostText;
        if (store.suggestion) {
          // Insert the suggestion
          this.editor.commands.insertContent(store.suggestion);
          store.clearSuggestion();
          return true;
        }
        return false;
      },
      Escape: () => {
        const store = getAIStore().ghostText;
        if (store.suggestion) {
          store.clearSuggestion();
          return true;
        }
        return false;
      },
    };
  },

  addProseMirrorPlugins() {
    const extension = this;

    return [
      new Plugin({
        key: ghostTextPluginKey,

        state: {
          init() {
            return DecorationSet.empty;
          },

          apply(tr, oldState) {
            const store = getAIStore().ghostText;

            if (!store.suggestion || !extension.options.enabled) {
              return DecorationSet.empty;
            }

            // Create ghost text decoration at cursor position
            const { selection } = tr;
            const pos = selection.$head.pos;

            const decoration = Decoration.widget(
              pos,
              () => {
                const span = document.createElement('span');
                span.className = 'ghost-text-suggestion';
                span.textContent = store.suggestion;
                span.style.cssText = `
                  color: #9CA3AF;
                  opacity: 0.7;
                  pointer-events: none;
                `;
                return span;
              },
              { side: 1 }
            );

            return DecorationSet.create(tr.doc, [decoration]);
          },
        },

        props: {
          decorations(state) {
            return this.getState(state);
          },

          handleTextInput(view, from, to, text) {
            if (!extension.options.enabled) return false;

            // Request suggestion after text input
            const store = getAIStore().ghostText;
            const noteId = view.dom.dataset.noteId;

            if (noteId) {
              const context = view.state.doc.textBetween(
                Math.max(0, from - extension.options.contextChars),
                from
              ) + text;

              store.requestSuggestion(noteId, context, from);
            }

            return false;
          },
        },
      }),
    ];
  },
});
```

```typescript
/**
 * Ghost text decoration styles and utilities.
 */
import { DecorationSet, Decoration } from '@tiptap/pm/view';

export interface GhostTextDecorationOptions {
  className?: string;
  style?: Partial<CSSStyleDeclaration>;
}

const defaultOptions: GhostTextDecorationOptions = {
  className: 'ghost-text-suggestion',
  style: {
    color: '#9CA3AF',
    opacity: '0.7',
    fontStyle: 'italic',
  },
};

export function createGhostTextDecoration(
  pos: number,
  text: string,
  options: GhostTextDecorationOptions = defaultOptions
): Decoration {
  return Decoration.widget(
    pos,
    () => {
      const element = document.createElement('span');
      element.className = options.className ?? defaultOptions.className!;
      element.textContent = text;

      const style = { ...defaultOptions.style, ...options.style };
      Object.assign(element.style, style);

      // Accessibility
      element.setAttribute('aria-hidden', 'true');
      element.setAttribute('data-ghost-text', 'true');

      return element;
    },
    { side: 1, key: 'ghost-text' }
  );
}

export function createGhostTextDecorationSet(
  doc: any,
  pos: number,
  text: string
): DecorationSet {
  if (!text) {
    return DecorationSet.empty;
  }

  const decoration = createGhostTextDecoration(pos, text);
  return DecorationSet.create(doc, [decoration]);
}

// CSS to add to global styles
export const ghostTextStyles = `
  .ghost-text-suggestion {
    color: #9CA3AF;
    opacity: 0.7;
    font-style: italic;
    pointer-events: none;
    user-select: none;
  }

  .ghost-text-suggestion::before {
    content: '';
  }

  /* Dark mode */
  .dark .ghost-text-suggestion {
    color: #6B7280;
  }
`;
```

```typescript
/**
 * NoteEditor with GhostText integration.
 */
import { useEditor, EditorContent } from '@tiptap/react';
import StarterKit from '@tiptap/starter-kit';
import { GhostTextExtension } from '@/components/editor/extensions/ghost-text-extension';
import { observer } from 'mobx-react-lite';
import { useStore } from '@/stores/use-store';

interface NoteEditorProps {
  noteId: string;
  initialContent?: string;
  onChange?: (content: string) => void;
}

export const NoteEditor = observer(function NoteEditor({
  noteId,
  initialContent,
  onChange,
}: NoteEditorProps) {
  const { ai } = useStore();
  const ghostTextEnabled = ai.settings.ghostTextEnabled && ai.isGloballyEnabled;

  const editor = useEditor({
    extensions: [
      StarterKit,
      GhostTextExtension.configure({
        enabled: ghostTextEnabled,
        debounceMs: 500,
        contextChars: 500,
      }),
    ],
    content: initialContent,
    onUpdate: ({ editor }) => {
      onChange?.(editor.getHTML());
    },
  });

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      ai.ghostText.abort();
    };
  }, [ai.ghostText]);

  return (
    <div
      className="note-editor"
      data-note-id={noteId}
    >
      <EditorContent editor={editor} />
    </div>
  );
});
```

```typescript
describe('GhostTextStore', () => {
  let store: GhostTextStore;
  let rootStore: AIStore;

  beforeEach(() => {
    rootStore = new AIStore();
    store = rootStore.ghostText;
  });

  afterEach(() => {
    store.abort();
  });

  it('should debounce suggestion requests', async () => {
    const fetchSpy = jest.spyOn(global, 'fetch').mockResolvedValue(new Response());

    store.requestSuggestion('note-1', 'context', 10);
    store.requestSuggestion('note-1', 'context2', 11);
    store.requestSuggestion('note-1', 'context3', 12);

    // Only one request should be made after debounce
    await new Promise(r => setTimeout(r, 600));
    expect(fetchSpy).toHaveBeenCalledTimes(1);
  });

  it('should cache suggestions', () => {
    store['cache'].set('note-1:context', 'cached suggestion');

    store.requestSuggestion('note-1', 'context', 10);

    expect(store.suggestion).toBe('cached suggestion');
  });

  it('should clear suggestion on disable', () => {
    store['suggestion'] = 'test suggestion';

    store.setEnabled(false);

    expect(store.suggestion).toBe('');
    expect(store.isEnabled).toBe(false);
```
