```python
"""Ghost text latency benchmark."""
import pytest
import time
import statistics

@pytest.mark.performance
async def test_ghost_text_p95_under_2s(perf_client, auth_headers, test_note):
    """Verify ghost text p95 latency < 2s."""
    latencies = []

    for _ in range(50):
        start = time.time()
        async with perf_client.stream(
            "POST",
            f"/ai/notes/{test_note.id}/ghost-text",
            headers=auth_headers,
            json={"context": "The authentication system should", "cursor_position": 35},
        ) as response:
            first_token = False
            async for line in response.aiter_lines():
                if "token" in line and not first_token:
                    latencies.append(time.time() - start)
                    first_token = True
                    break

    latencies.sort()
    p95 = latencies[int(len(latencies) * 0.95)]

    assert p95 < 2.0, f"Ghost text p95 latency {p95}s exceeds 2s target"
    print(f"Ghost text p95: {p95:.2f}s, median: {statistics.median(latencies):.2f}s")
```

```python
async def analyze_agent_token_usage(db_session, days: int = 30):
    """Generate token usage report by agent."""
    query = """
    SELECT
        agent_name,
        COUNT(*) as request_count,
        AVG(input_tokens) as avg_input,
        AVG(output_tokens) as avg_output,
        PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY input_tokens) as p95_input,
        PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY output_tokens) as p95_output,
        SUM(cost_usd) as total_cost
    FROM ai_cost_records
    WHERE created_at > NOW() - INTERVAL '%s days'
    GROUP BY agent_name
    ORDER BY total_cost DESC
    """

    result = await db_session.execute(query, [days])
    return result.fetchall()
```

### T319: Implement Response Caching

```python
# backend/src/pilot_space/ai/infrastructure/cache.py
"""Response caching for repeated queries."""
from typing import Any
from hashlib import sha256
import json

class AIResponseCache:
    """Redis-based cache for AI responses."""

    def __init__(self, redis_client, ttl_seconds: int = 3600):
        self._redis = redis_client
        self._ttl = ttl_seconds

    def _make_key(self, agent_name: str, input_hash: str) -> str:
        return f"ai:cache:{agent_name}:{input_hash}"

    def _hash_input(self, input_data: Any) -> str:
        serialized = json.dumps(input_data, sort_keys=True)
        return sha256(serialized.encode()).hexdigest()[:16]

    async def get(self, agent_name: str, input_data: Any) -> Any | None:
        key = self._make_key(agent_name, self._hash_input(input_data))
        cached = await self._redis.get(key)
        if cached:
            return json.loads(cached)
        return None

    async def set(self, agent_name: str, input_data: Any, response: Any) -> None:
        key = self._make_key(agent_name, self._hash_input(input_data))
        await self._redis.setex(key, self._ttl, json.dumps(response))
```

```python
import pytest
import asyncio
from pilot_space.ai.infrastructure.rate_limiter import RateLimiter

class TestRateLimiter:
    @pytest.mark.asyncio
    async def test_handles_concurrent_requests(self, redis_client):
        limiter = RateLimiter(redis_client, requests_per_minute=10)

        # Fire 20 concurrent requests
        results = await asyncio.gather(
            *[limiter.acquire("test-key") for _ in range(20)],
            return_exceptions=True
        )

        allowed = sum(1 for r in results if r is True)
        denied = sum(1 for r in results if r is False)

        assert allowed == 10
        assert denied == 10

    @pytest.mark.asyncio
    async def test_sliding_window(self, redis_client):
        limiter = RateLimiter(redis_client, requests_per_minute=60)

        # Use 30 requests
        for _ in range(30):
            assert await limiter.acquire("test-key")

        # After 30 seconds, should have 30 more available
        await asyncio.sleep(30)

        for _ in range(30):
            assert await limiter.acquire("test-key")
```

```python
# Add to SessionManager
async def get_session_metrics(self) -> dict:
    """Get session metrics for monitoring."""
    total = await self._redis.dbsize()

    # Count by agent
    by_agent = {}
    cursor = 0
    while True:
        cursor, keys = await self._redis.scan(cursor, match="ai:session:*")
        for key in keys:
            session = await self._redis.get(key)
            if session:
                data = json.loads(session)
                agent = data.get("agent_name", "unknown")
                by_agent[agent] = by_agent.get(agent, 0) + 1
        if cursor == 0:
            break

    return {
        "total_sessions": total,
        "by_agent": by_agent,
    }
```
