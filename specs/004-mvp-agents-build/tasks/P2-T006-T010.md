"""Add workspace_api_keys table for BYOK (DD-002).

Revision ID: <auto-generated>
"""
from alembic import op
import sqlalchemy as sa
from sqlalchemy.dialects.postgresql import UUID

def upgrade() -> None:
    op.create_table(
        "workspace_api_keys",
        sa.Column("id", UUID(as_uuid=True), primary_key=True, server_default=sa.text("gen_random_uuid()")),
        sa.Column("workspace_id", UUID(as_uuid=True), sa.ForeignKey("workspaces.id", ondelete="CASCADE"), nullable=False),
        sa.Column("provider", sa.String(50), nullable=False),  # anthropic, openai, google
        sa.Column("encrypted_key", sa.Text(), nullable=False),  # Supabase Vault encrypted
        sa.Column("is_valid", sa.Boolean(), server_default="true"),
        sa.Column("last_validated_at", sa.DateTime(timezone=True)),
        sa.Column("created_at", sa.DateTime(timezone=True), server_default=sa.text("now()")),
        sa.Column("updated_at", sa.DateTime(timezone=True), server_default=sa.text("now()")),
    )
    op.create_unique_constraint("uq_workspace_api_keys_workspace_provider", "workspace_api_keys", ["workspace_id", "provider"])
    op.create_index("ix_workspace_api_keys_workspace_id", "workspace_api_keys", ["workspace_id"])

def downgrade() -> None:
    op.drop_index("ix_workspace_api_keys_workspace_id")
    op.drop_constraint("uq_workspace_api_keys_workspace_provider", "workspace_api_keys")
    op.drop_table("workspace_api_keys")
```

"""Add ai_approval_requests table for human-in-the-loop (DD-003).

Revision ID: <auto-generated>
"""
from alembic import op
import sqlalchemy as sa
from sqlalchemy.dialects.postgresql import UUID, JSONB

def upgrade() -> None:
    # Create status enum
    approval_status = sa.Enum("pending", "approved", "rejected", "expired", name="approval_status")
    approval_status.create(op.get_bind(), checkfirst=True)

    op.create_table(
        "ai_approval_requests",
        sa.Column("id", UUID(as_uuid=True), primary_key=True, server_default=sa.text("gen_random_uuid()")),
        sa.Column("workspace_id", UUID(as_uuid=True), sa.ForeignKey("workspaces.id", ondelete="CASCADE"), nullable=False),
        sa.Column("user_id", UUID(as_uuid=True), sa.ForeignKey("users.id", ondelete="SET NULL")),
        sa.Column("agent_name", sa.String(100), nullable=False),  # e.g., issue_extractor, pr_review
        sa.Column("action_type", sa.String(100), nullable=False),  # e.g., create_issue, merge_pr
        sa.Column("payload", JSONB(), nullable=False),  # Action-specific data
        sa.Column("context", JSONB()),  # Optional context for reviewer
        sa.Column("status", approval_status, server_default="pending", nullable=False),
        sa.Column("expires_at", sa.DateTime(timezone=True), nullable=False),
        sa.Column("created_at", sa.DateTime(timezone=True), server_default=sa.text("now()")),
        sa.Column("resolved_at", sa.DateTime(timezone=True)),
        sa.Column("resolved_by", UUID(as_uuid=True), sa.ForeignKey("users.id", ondelete="SET NULL")),
        sa.Column("resolution_note", sa.Text()),
    )
    op.create_index("ix_ai_approval_requests_workspace_status", "ai_approval_requests", ["workspace_id", "status"])
    op.create_index("ix_ai_approval_requests_expires_at", "ai_approval_requests", ["expires_at"], postgresql_where=sa.text("status = 'pending'"))

def downgrade() -> None:
    op.drop_index("ix_ai_approval_requests_expires_at")
    op.drop_index("ix_ai_approval_requests_workspace_status")
    op.drop_table("ai_approval_requests")
    sa.Enum(name="approval_status").drop(op.get_bind(), checkfirst=True)
```

"""Add ai_sessions table for multi-turn conversations.

Revision ID: <auto-generated>
"""
from alembic import op
import sqlalchemy as sa
from sqlalchemy.dialects.postgresql import UUID, JSONB

def upgrade() -> None:
    op.create_table(
        "ai_sessions",
        sa.Column("id", UUID(as_uuid=True), primary_key=True, server_default=sa.text("gen_random_uuid()")),
        sa.Column("workspace_id", UUID(as_uuid=True), sa.ForeignKey("workspaces.id", ondelete="CASCADE"), nullable=False),
        sa.Column("user_id", UUID(as_uuid=True), sa.ForeignKey("users.id", ondelete="CASCADE"), nullable=False),
        sa.Column("agent_name", sa.String(100), nullable=False),  # e.g., ai_context, conversation
        sa.Column("context_id", UUID(as_uuid=True)),  # Optional: issue_id, note_id, etc.
        sa.Column("session_data", JSONB(), nullable=False),  # Conversation history, turn count, etc.
        sa.Column("total_cost_usd", sa.Numeric(10, 6), server_default="0"),
        sa.Column("turn_count", sa.Integer(), server_default="0"),
        sa.Column("created_at", sa.DateTime(timezone=True), server_default=sa.text("now()")),
        sa.Column("updated_at", sa.DateTime(timezone=True), server_default=sa.text("now()")),
        sa.Column("expires_at", sa.DateTime(timezone=True), nullable=False),
    )
    # Index for cleanup job
    op.create_index("ix_ai_sessions_expires_at", "ai_sessions", ["expires_at"])
    # Index for user session lookup
    op.create_index("ix_ai_sessions_user_agent", "ai_sessions", ["user_id", "agent_name"])
    # Unique active session per user/agent/context
    op.create_unique_constraint(
        "uq_ai_sessions_user_agent_context",
        "ai_sessions",
        ["user_id", "agent_name", "context_id"]
    )

def downgrade() -> None:
    op.drop_constraint("uq_ai_sessions_user_agent_context", "ai_sessions")
    op.drop_index("ix_ai_sessions_user_agent")
    op.drop_index("ix_ai_sessions_expires_at")
    op.drop_table("ai_sessions")
