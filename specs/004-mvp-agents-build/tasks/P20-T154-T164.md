```typescript
// frontend/src/features/notes/components/issue-extraction-panel.tsx
import { observer } from 'mobx-react-lite';
import { useStore } from '@/stores/use-store';
import { ExtractedIssueCard } from './extracted-issue-card';
import { IssueExtractionApprovalModal } from './issue-extraction-approval-modal';
import { Button } from '@/components/ui/button';
import { Loader2, Sparkles, AlertCircle } from 'lucide-react';
import { useState } from 'react';

interface IssueExtractionPanelProps {
  noteId: string;
}

export const IssueExtractionPanel = observer(function IssueExtractionPanel({
  noteId,
}: IssueExtractionPanelProps) {
  const { ai } = useStore();
  const { issueExtraction } = ai;
  const [showApprovalModal, setShowApprovalModal] = useState(false);
  const [selectedIssues, setSelectedIssues] = useState<number[]>([]);

  const handleExtract = () => {
    issueExtraction.extractIssues(noteId);
  };

  const handleToggleIssue = (index: number) => {
    setSelectedIssues(prev =>
      prev.includes(index)
        ? prev.filter(i => i !== index)
        : [...prev, index]
    );
  };

  const handleCreateIssues = () => {
    if (selectedIssues.length > 0) {
      setShowApprovalModal(true);
    }
  };

  return (
    <div className="issue-extraction-panel space-y-4">
      <div className="flex items-center justify-between">
        <h3 className="font-semibold flex items-center gap-2">
          <Sparkles className="h-4 w-4" />
          Extract Issues
        </h3>
      </div>

      {!issueExtraction.isLoading && !issueExtraction.extractedIssues.length && (
        <div className="text-center py-6">
          <p className="text-sm text-muted-foreground mb-4">
            Extract actionable issues from this note using AI analysis.
          </p>
          <Button onClick={handleExtract}>
            <Sparkles className="h-4 w-4 mr-2" />
            Extract Issues
          </Button>
        </div>
      )}

      {issueExtraction.isLoading && (
        <div className="flex items-center gap-2 text-muted-foreground">
          <Loader2 className="h-4 w-4 animate-spin" />
          Analyzing note content...
        </div>
      )}

      {issueExtraction.error && (
        <div className="p-3 bg-destructive/10 text-destructive rounded flex items-center gap-2">
          <AlertCircle className="h-4 w-4" />
          {issueExtraction.error}
        </div>
      )}

      {issueExtraction.extractedIssues.length > 0 && (
        <>
          <div className="space-y-3">
            {issueExtraction.extractedIssues.map((issue, index) => (
              <ExtractedIssueCard
                key={index}
                issue={issue}
                selected={selectedIssues.includes(index)}
                onToggle={() => handleToggleIssue(index)}
              />
            ))}
          </div>

          <div className="flex justify-between items-center pt-4 border-t">
            <span className="text-sm text-muted-foreground">
              {selectedIssues.length} of {issueExtraction.extractedIssues.length} selected
            </span>
            <Button
              onClick={handleCreateIssues}
              disabled={selectedIssues.length === 0}
            >
              Create Selected Issues
            </Button>
          </div>
        </>
      )}

      <IssueExtractionApprovalModal
        open={showApprovalModal}
        onOpenChange={setShowApprovalModal}
        selectedIndices={selectedIssues}
        issues={issueExtraction.extractedIssues}
      />
    </div>
  );
});
```

### T156: ConfidenceTagBadge

```typescript
// frontend/src/components/ui/confidence-tag-badge.tsx
import { Badge } from '@/components/ui/badge';
import { Tooltip, TooltipContent, TooltipTrigger } from '@/components/ui/tooltip';
import { Star, Check, Minus, HelpCircle } from 'lucide-react';
import { cn } from '@/lib/utils';

type ConfidenceTag = 'recommended' | 'default' | 'current' | 'alternative';

interface ConfidenceTagBadgeProps {
  tag: ConfidenceTag;
  score?: number;
  showScore?: boolean;
}

const tagConfig = {
  recommended: {
    label: 'Recommended',
    icon: Star,
    className: 'bg-green-500/10 text-green-600 border-green-200',
    description: 'High confidence - strongly suggested',
  },
  default: {
    label: 'Default',
    icon: Check,
    className: 'bg-blue-500/10 text-blue-600 border-blue-200',
    description: 'Good confidence - reasonable option',
  },
  current: {
    label: 'Current',
    icon: Minus,
    className: 'bg-gray-500/10 text-gray-600 border-gray-200',
    description: 'Moderate confidence - matches existing pattern',
  },
  alternative: {
    label: 'Alternative',
    icon: HelpCircle,
    className: 'bg-yellow-500/10 text-yellow-600 border-yellow-200',
    description: 'Lower confidence - consider carefully',
  },
};

export function ConfidenceTagBadge({
  tag,
  score,
  showScore = false,
}: ConfidenceTagBadgeProps) {
  const config = tagConfig[tag];
  const Icon = config.icon;

  return (
    <Tooltip>
      <TooltipTrigger>
        <Badge
          variant="outline"
          className={cn('text-xs gap-1 font-normal', config.className)}
        >
          <Icon className="h-3 w-3" />
          {config.label}
          {showScore && score !== undefined && (
            <span className="opacity-70">({Math.round(score * 100)}%)</span>
          )}
        </Badge>
      </TooltipTrigger>
      <TooltipContent>
        <p>{config.description}</p>
        {score !== undefined && (
          <p className="text-xs opacity-70">
            Confidence: {Math.round(score * 100)}%
          </p>
        )}
      </TooltipContent>
    </Tooltip>
  );
}
```

### T158: Create IssueExtractionStore

```typescript
// frontend/src/stores/issue-extraction-store.ts
import { makeAutoObservable, runInAction } from 'mobx';
import { SSEClient, SSEEvent } from '@/lib/sse-client';
import { aiApi } from '@/services/ai-api';
import type { AIStore } from './ai-store';

export interface ExtractedIssue {
  title: string;
  description: string;
  confidence_tag: 'recommended' | 'default' | 'current' | 'alternative';
  confidence_score: number;
  labels?: string[];
  priority?: string;
}

export class IssueExtractionStore {
  isLoading = false;
  isCreating = false;
  error: string | null = null;
  extractedIssues: ExtractedIssue[] = [];
  currentNoteId: string | null = null;
  approvalId: string | null = null;

  private client: SSEClient | null = null;

  constructor(private rootStore: AIStore) {
    makeAutoObservable(this);
  }

  async extractIssues(noteId: string, maxIssues = 10): Promise<void> {
    this.abort();

    runInAction(() => {
      this.isLoading = true;
      this.error = null;
      this.extractedIssues = [];
      this.currentNoteId = noteId;
      this.approvalId = null;
    });

    this.client = new SSEClient({
      url: aiApi.getIssueExtractionUrl(noteId),
      onMessage: (event: SSEEvent) => this.handleEvent(event),
      onComplete: () => {
        runInAction(() => {
          this.isLoading = false;
        });
      },
      onError: (err) => {
        runInAction(() => {
          this.isLoading = false;
          this.error = err.message;
        });
      },
    });

    await this.client.connect();
  }

  private handleEvent(event: SSEEvent): void {
    runInAction(() => {
      const data = event.data as Record<string, unknown>;

      switch (event.type) {
        case 'issue':
          this.extractedIssues.push(data as ExtractedIssue);
          break;
        case 'complete':
          this.approvalId = data.approval_id as string;
          break;
      }
    });
  }

  async createApprovedIssues(selectedIndices: number[]): Promise<void> {
    if (!this.approvalId) {
      throw new Error('No approval ID available');
    }

    runInAction(() => {
      this.isCreating = true;
      this.error = null;
    });

    try {
      await aiApi.resolveApproval(this.approvalId, {
        approved: true,
        selected_issues: selectedIndices,
      });

      runInAction(() => {
        this.isCreating = false;
        this.extractedIssues = [];
        this.approvalId = null;
      });
    } catch (err) {
      runInAction(() => {
        this.isCreating = false;
        this.error = err instanceof Error ? err.message : 'Failed to create issues';
      });
      throw err;
    }
  }

  abort(): void {
    this.client?.abort();
    this.client = null;
    this.isLoading = false;
  }

  clear(): void {
    this.extractedIssues = [];
    this.approvalId = null;
    this.error = null;
  }
}
```
