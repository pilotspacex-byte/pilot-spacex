    ):

        response = await client.post(
            "/issues",

                "title": "Bypassed issue",

            },
        )

        # Should require approval or reject

    @pytest.mark.asyncio

        self, client, auth_headers, expired_approval

        assert "expired" in response.json()["detail"].lower()
