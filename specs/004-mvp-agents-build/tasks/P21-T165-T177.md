// frontend/src/components/editor/extensions/margin-annotation-extension.ts
import { Extension } from '@tiptap/core';
import { Plugin, PluginKey } from 'prosemirror-state';
import { Decoration, DecorationSet } from 'prosemirror-view';

export interface MarginAnnotationOptions {
  enabled: boolean;
  annotationStore: MarginAnnotationStore;
}

export const marginAnnotationPluginKey = new PluginKey('marginAnnotation');

export const MarginAnnotationExtension = Extension.create<MarginAnnotationOptions>({
  name: 'marginAnnotation',

  addOptions() {
    return {
      enabled: true,
      annotationStore: null!,
    };
  },

  addProseMirrorPlugins() {
    const { enabled, annotationStore } = this.options;

    if (!enabled) return [];

    return [
      new Plugin({
        key: marginAnnotationPluginKey,
        state: {
          init() {
            return DecorationSet.empty;
          },
          apply(tr, old, oldState, newState) {
            // Update decorations based on annotation store state
            const annotations = annotationStore.getAnnotationsForDoc();
            return buildDecorations(newState.doc, annotations);
          },
        },
        props: {
          decorations(state) {
            return this.getState(state);
          },
        },
      }),
    ];
  },
});

function buildDecorations(doc: Node, annotations: Annotation[]): DecorationSet {
  const decorations: Decoration[] = [];

  annotations.forEach((annotation) => {
    const pos = findBlockPosition(doc, annotation.blockId);
    if (pos !== null) {
      decorations.push(
        Decoration.widget(pos, () => {
          const indicator = document.createElement('div');
          indicator.className = 'margin-annotation-indicator';
          indicator.dataset.annotationId = annotation.id;
          indicator.dataset.annotationType = annotation.type;
          return indicator;
        }, { side: 1 })
      );
    }
  });

  return DecorationSet.create(doc, decorations);
}

### T166: Annotation Positioning Plugin

```typescript
// frontend/src/components/editor/plugins/annotation-positioning.ts
import { Plugin, PluginKey } from 'prosemirror-state';

export const annotationPositioningKey = new PluginKey('annotationPositioning');

export interface BlockPosition {
  blockId: string;
  top: number;
  height: number;
}

export function createAnnotationPositioningPlugin(
  onPositionsUpdate: (positions: BlockPosition[]) => void
): Plugin {
  return new Plugin({
    key: annotationPositioningKey,
    view(editorView) {
      let rafId: number | null = null;

      const updatePositions = () => {
        const positions: BlockPosition[] = [];
        const { doc } = editorView.state;

        doc.descendants((node, pos) => {
          if (node.isBlock && node.attrs.blockId) {
            const dom = editorView.nodeDOM(pos);
            if (dom instanceof HTMLElement) {
              const rect = dom.getBoundingClientRect();
              const editorRect = editorView.dom.getBoundingClientRect();
              positions.push({
                blockId: node.attrs.blockId,
                top: rect.top - editorRect.top,
                height: rect.height,
              });
            }
          }
          return true;
        });

        onPositionsUpdate(positions);
      };

      const scheduleUpdate = () => {
        if (rafId) cancelAnimationFrame(rafId);
        rafId = requestAnimationFrame(updatePositions);
      };

      // Initial update
      scheduleUpdate();

      return {
        update: scheduleUpdate,
        destroy() {
          if (rafId) cancelAnimationFrame(rafId);
        },
      };
    },
  });
}

// frontend/src/stores/margin-annotation-store.ts
import { makeAutoObservable, runInAction } from 'mobx';
import { aiApi } from '@/services/ai-api';
import { SSEClient } from '@/lib/sse-client';

export interface NoteAnnotation {
  id: string;
  noteId: string;
  blockId: string;
  type: 'suggestion' | 'warning' | 'question' | 'insight' | 'reference';
  title: string;
  summary: string;
  content: string;
  suggestedText?: string;
  references?: { title: string; url: string }[];
  confidence: number;
  createdAt: string;
}

export class MarginAnnotationStore {
  annotations: Map<string, NoteAnnotation[]> = new Map(); // noteId -> annotations
  selectedAnnotationId: string | null = null;
  isLoading: boolean = false;
  error: string | null = null;

  private sseClient: SSEClient | null = null;

  constructor() {
    makeAutoObservable(this, {
      sseClient: false,
    });
  }

  // T172: fetchAnnotations
  async fetchAnnotations(noteId: string): Promise<void> {
    this.isLoading = true;
    this.error = null;

    try {
      const response = await aiApi.getAnnotations(noteId);
      runInAction(() => {
        this.annotations.set(noteId, response.annotations);
        this.isLoading = false;
      });
    } catch (err) {
      runInAction(() => {
        this.error = err instanceof Error ? err.message : 'Failed to load annotations';
        this.isLoading = false;
      });
    }
  }

  // Generate new annotations via SSE
  async generateAnnotations(noteId: string, content: string): Promise<void> {
    this.abort();
    this.isLoading = true;
    this.error = null;

    // Clear existing annotations for regeneration
    runInAction(() => {
      this.annotations.set(noteId, []);
    });

    this.sseClient = new SSEClient({
      url: `/api/v1/ai/notes/${noteId}/annotations`,
      method: 'POST',
      body: { content },
      onMessage: (event) => {
        if (event.type === 'annotation') {
          runInAction(() => {
            const current = this.annotations.get(noteId) || [];
            this.annotations.set(noteId, [...current, event.data as NoteAnnotation]);
          });
        }
      },
      onComplete: () => {
        runInAction(() => {
          this.isLoading = false;
        });
      },
      onError: (err) => {
        runInAction(() => {
          this.error = err.message;
          this.isLoading = false;
        });
      },
    });

    await this.sseClient.connect();
  }

  // T173: block→annotation mapping
  getAnnotationsByBlock(noteId: string): Map<string, NoteAnnotation[]> {
    const annotations = this.annotations.get(noteId) || [];
    const byBlock = new Map<string, NoteAnnotation[]>();

    annotations.forEach((annotation) => {
      const existing = byBlock.get(annotation.blockId) || [];
      byBlock.set(annotation.blockId, [...existing, annotation]);
    });

    return byBlock;
  }

  getAnnotationsForDoc(): NoteAnnotation[] {
    // Return all annotations for the currently active note
    const allAnnotations: NoteAnnotation[] = [];
    this.annotations.forEach((noteAnnotations) => {
      allAnnotations.push(...noteAnnotations);
    });
    return allAnnotations;
  }

  selectAnnotation(annotationId: string): void {
    this.selectedAnnotationId = annotationId;
  }

  clearSelection(): void {
    this.selectedAnnotationId = null;
  }

  dismissAnnotation(annotationId: string): void {
    this.annotations.forEach((noteAnnotations, noteId) => {
      const filtered = noteAnnotations.filter((a) => a.id !== annotationId);
      if (filtered.length !== noteAnnotations.length) {
        this.annotations.set(noteId, filtered);
      }
    });

    if (this.selectedAnnotationId === annotationId) {
      this.selectedAnnotationId = null;
    }
  }

  abort(): void {
    this.sseClient?.abort();
    this.sseClient = null;
    this.isLoading = false;
  }

  clearAnnotations(noteId: string): void {
    this.annotations.delete(noteId);
    this.selectedAnnotationId = null;
  }
}
```
