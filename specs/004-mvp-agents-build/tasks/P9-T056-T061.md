```python
"""Issue Extractor Agent using Claude Agent SDK.

Extracts actionable issues from note content with confidence tags
per DD-048 (Recommended/Default/Current/Alternative).

Reference: docs/architect/claude-agent-sdk-architecture.md
"""
from dataclasses import dataclass
from enum import Enum
from typing import Any
from uuid import UUID

from claude_agent_sdk import query

from pilot_space.ai.agents.sdk_base import SDKBaseAgent, AgentContext


class ConfidenceTag(str, Enum):
    """Confidence tags per DD-048."""
    RECOMMENDED = "recommended"  # AI strongly suggests
    DEFAULT = "default"  # Standard choice
    CURRENT = "current"  # Matches existing pattern
    ALTERNATIVE = "alternative"  # Valid but less preferred


@dataclass
class ExtractedIssue:
    """An issue extracted from note content."""
    title: str
    description: str
    labels: list[str]
    priority: int  # 0-4 (urgent to no priority)
    confidence_tag: ConfidenceTag
    confidence_score: float  # 0.0-1.0
    source_block_ids: list[str]  # Note blocks this came from
    rationale: str  # Why this issue was extracted


@dataclass
class IssueExtractorInput:
    """Input for issue extraction."""
    note_id: UUID
    project_id: UUID
    max_issues: int = 10
    min_confidence: float = 0.5


@dataclass
class IssueExtractorOutput:
    """Output from issue extraction."""
    issues: list[ExtractedIssue]
    source_note_id: UUID
    extraction_summary: str


class IssueExtractorAgent(SDKBaseAgent[IssueExtractorInput, IssueExtractorOutput]):
    """Extracts actionable issues from note content.

    Uses Claude Sonnet for balanced extraction quality.
    Returns issues with confidence tags per DD-048.

    MCP Tools Used:
    - get_note_content: Retrieve note blocks
    - get_project_context: Get labels, states for categorization
    """

    AGENT_NAME = "issue_extractor"
    DEFAULT_MODEL = "claude-sonnet-4-20250514"
    MAX_TOKENS = 4096

    async def execute(
        self,
        input_data: IssueExtractorInput,
        context: AgentContext,
    ) -> IssueExtractorOutput:
        """Extract issues from note content.

        Args:
            input_data: Note and project context
            context: Execution context

        Returns:
            Extracted issues with confidence tags
        """
        options = self._build_options(
            context,
            max_tokens=self.MAX_TOKENS,
            temperature=0.3,  # Lower for structured extraction
            tools=["get_note_content", "get_project_context"],
        )

        # Build extraction prompt
        prompt = self._build_prompt(input_data)

        # Execute SDK query
        response = await query(
            prompt=prompt,
            system=self._get_system_prompt(),
            options=options,
        )

        # Parse structured output
        return self._parse_response(response, input_data.note_id)

    def _get_system_prompt(self) -> str:
        """System prompt for issue extraction."""
        return """You are an expert at analyzing technical notes and extracting actionable issues.

Your task is to identify potential issues, tasks, bugs, and improvements from the note content.

For each extracted issue, provide:
1. A clear, actionable title (imperative form)
2. A detailed description with context
3. Suggested labels based on the project's label schema
4. Priority (0=urgent, 1=high, 2=medium, 3=low, 4=no priority)
5. Confidence tag per the following schema:
   - RECOMMENDED: You strongly believe this should be an issue
   - DEFAULT: Standard extraction, reasonable confidence
   - CURRENT: Matches existing issues or patterns
   - ALTERNATIVE: Valid but may need human review
6. Confidence score (0.0-1.0)
7. Source block IDs from the note
8. Rationale for the extraction

Output as JSON array with this structure:
{
  "issues": [
    {
      "title": "string",
      "description": "string",
      "labels": ["string"],
      "priority": number,
      "confidence_tag": "recommended|default|current|alternative",
      "confidence_score": number,
      "source_block_ids": ["string"],
      "rationale": "string"
    }
  ],
  "extraction_summary": "Brief summary of what was found"
}

Use the MCP tools to:
1. get_note_content: Read the note blocks
2. get_project_context: Understand available labels and conventions"""

    def _build_prompt(self, input_data: IssueExtractorInput) -> str:
        """Build the extraction prompt."""
        return f"""Extract actionable issues from the note.

Note ID: {input_data.note_id}
Project ID: {input_data.project_id}
Max issues to extract: {input_data.max_issues}
Minimum confidence threshold: {input_data.min_confidence}

Steps:
1. Use get_note_content to read the note blocks
2. Use get_project_context to understand the project's labels and conventions
3. Analyze the content for actionable items
4. Return structured JSON with extracted issues

Only include issues with confidence_score >= {input_data.min_confidence}."""

    def _parse_response(
        self,
        response: Any,
        note_id: UUID,
    ) -> IssueExtractorOutput:
        """Parse SDK response to output format."""
        import json

        # Extract JSON from response
        content = response.content if hasattr(response, 'content') else str(response)

        # Find JSON in response
        try:
            # Try to parse entire response as JSON
            data = json.loads(content)
        except json.JSONDecodeError:
            # Extract JSON from markdown code block
            import re
            json_match = re.search(r'```json\s*(.*?)\s*```', content, re.DOTALL)
            if json_match:
                data = json.loads(json_match.group(1))
            else:
                data = {"issues": [], "extraction_summary": "Failed to parse response"}

        issues = [
            ExtractedIssue(
                title=i["title"],
                description=i["description"],
                labels=i.get("labels", []),
                priority=i.get("priority", 2),
                confidence_tag=ConfidenceTag(i.get("confidence_tag", "default")),
                confidence_score=i.get("confidence_score", 0.5),
                source_block_ids=i.get("source_block_ids", []),
                rationale=i.get("rationale", ""),
            )
            for i in data.get("issues", [])
        ]

        return IssueExtractorOutput(
            issues=issues,
            source_note_id=note_id,
            extraction_summary=data.get("extraction_summary", f"Extracted {len(issues)} issues"),
        )
```

```python
from uuid import UUID
from fastapi import APIRouter, Depends, Request
from pydantic import BaseModel, Field

from pilot_space.ai.agents.issue_extractor_agent import (
    IssueExtractorInput,
    IssueExtractorOutput,
    ExtractedIssue,
    ConfidenceTag,
)
from pilot_space.ai.agents.sdk_base import AgentContext
from pilot_space.api.deps import get_current_user, get_sdk_orchestrator


class ExtractIssuesRequest(BaseModel):
    """Request body for issue extraction."""
    max_issues: int = Field(default=10, ge=1, le=50)
    min_confidence: float = Field(default=0.5, ge=0.0, le=1.0)


class ExtractedIssueResponse(BaseModel):
    """Response model for extracted issue."""
    title: str
    description: str
    labels: list[str]
    priority: int
    confidence_tag: str
    confidence_score: float
    source_block_ids: list[str]
    rationale: str


class ExtractIssuesResponse(BaseModel):
    """Response for issue extraction."""
    issues: list[ExtractedIssueResponse]
    extraction_summary: str
    requires_approval: bool
    approval_id: str | None = None


@router.post("/notes/{note_id}/extract-issues")
async def extract_issues_stream(
    note_id: UUID,
    request: Request,
    body: ExtractIssuesRequest,
    current_user = Depends(get_current_user),
    orchestrator = Depends(get_sdk_orchestrator),
):
    """Extract issues from note content with confidence tags.

    Returns SSE stream with:
    - progress: Extraction progress updates
    - issue: Each extracted issue as found
    - complete: Final summary with all issues
    - error: If extraction fails

    Issues will require approval before creation (DD-003).
    """
    async def generate_events():
        try:
            # Emit start event
            yield create_sse_event("progress", {
                "status": "analyzing",
                "message": "Reading note content..."
            })

            # Build input
            input_data = IssueExtractorInput(
                note_id=note_id,
                project_id=current_user.current_project_id,
                max_issues=body.max_issues,
                min_confidence=body.min_confidence,
            )

            context = AgentContext(
                workspace_id=current_user.workspace_id,
                user_id=current_user.id,
            )

            yield create_sse_event("progress", {
                "status": "extracting",
                "message": "Extracting issues..."
            })

            # Execute extraction
            result = await orchestrator.execute(
                "issue_extractor",
                input_data,
                context,
            )

            # Emit each issue
            for i, issue in enumerate(result.output.issues):
                yield create_sse_event("issue", {
                    "index": i,
                    "title": issue.title,
                    "description": issue.description,
                    "labels": issue.labels,
                    "priority": issue.priority,
                    "confidence_tag": issue.confidence_tag.value,
                    "confidence_score": issue.confidence_score,
                    "rationale": issue.rationale,
                })

            # Emit completion
            yield create_sse_event("complete", {
                "status": "complete",
                "total_issues": len(result.output.issues),
                "summary": result.output.extraction_summary,
                "requires_approval": True,
                "cost_usd": result.cost_usd,
            })

        except Exception as e:
            yield create_sse_event("error", {
                "message": str(e),
                "type": type(e).__name__,
            })

    return StreamingResponse(
        generate_events(),
        media_type="text/event-stream",
    )
```

```python
# Add approval endpoint for extracted issues

@router.post("/notes/{note_id}/extract-issues/approve")
async def approve_extracted_issues(
    note_id: UUID,
    approval_id: UUID,
    selected_issues: list[int],  # Indices of issues to create
    current_user = Depends(get_current_user),
    orchestrator = Depends(get_sdk_orchestrator),
    approval_service = Depends(get_approval_service),
):
    """Approve and create selected extracted issues.

    Args:
        note_id: Source note ID
        approval_id: Approval request ID
        selected_issues: Indices of issues to create (from extraction)

    Returns:
        Created issue IDs
    """
    # Verify approval belongs to user's workspace
    approval = await approval_service.get_request(approval_id)
    if not approval or approval.workspace_id != current_user.workspace_id:
        raise HTTPException(404, "Approval request not found")

    if approval.status != "pending":
        raise HTTPException(400, f"Approval already {approval.status}")

    # Resolve approval
    await approval_service.resolve(
        approval_id=approval_id,
        resolved_by=current_user.id,
        approved=True,
        note=f"Creating {len(selected_issues)} issues",
    )

    # Create selected issues
    created_ids = []
    for idx in selected_issues:
        if idx < len(approval.payload["issues"]):
            issue_data = approval.payload["issues"][idx]
            # Create issue via service
            issue = await issue_service.create(
                project_id=current_user.current_project_id,
                title=issue_data["title"],
                description=issue_data["description"],
                labels=issue_data.get("labels", []),
                priority=issue_data.get("priority", 2),
                created_by=current_user.id,
                source_note_id=note_id,
            )
            created_ids.append(str(issue.id))

    return {"created_issues": created_ids}
```
