### Implementation

```python
"""Conversation Agent using Claude Agent SDK.

Multi-turn conversation agent for general AI assistance.
Uses ClaudeSDKClient for session management.
"""
from dataclasses import dataclass
from uuid import UUID

from claude_agent_sdk import ClaudeSDKClient

from pilot_space.ai.agents.sdk_base import SDKBaseAgent, AgentContext
from pilot_space.ai.session.session_manager import SessionManager


@dataclass
class ConversationInput:
    """Input for conversation turn."""
    message: str
    session_id: UUID | None = None  # None for new conversation


@dataclass
class ConversationOutput:
    """Output from conversation turn."""
    response: str
    session_id: UUID
    turn_number: int


class ConversationAgent(SDKBaseAgent[ConversationInput, ConversationOutput]):
    """Multi-turn conversation agent.

    Uses ClaudeSDKClient for maintaining conversation history.
    Sessions are stored in Redis via SessionManager.
    """

    AGENT_NAME = "conversation"
    DEFAULT_MODEL = "claude-opus-4-5-20251101"
    MAX_TOKENS = 4096
    SESSION_TTL = 1800  # 30 minutes

    def __init__(self, session_manager: SessionManager, **kwargs):
        super().__init__(**kwargs)
        self._session_manager = session_manager

    async def execute(
        self,
        input_data: ConversationInput,
        context: AgentContext,
    ) -> ConversationOutput:
        """Execute a conversation turn."""
        # Get or create session
        if input_data.session_id:
            session = await self._session_manager.get_session(input_data.session_id)
            if not session:
                session = await self._create_session(context)
        else:
            session = await self._create_session(context)

        # Create SDK client with session history
        options = self._build_options(
            context,
            max_tokens=self.MAX_TOKENS,
            temperature=0.7,
            tools=["get_issue_context", "get_note_content", "semantic_search"],
        )

        client = ClaudeSDKClient(
            model=options.model,
            system=self._get_system_prompt(),
            options=options,
        )

        # Add conversation history
        for msg in session.get("history", []):
            client.add_message(msg["role"], msg["content"])

        # Send user message
        response = await client.send(input_data.message)

        # Update session
        session["history"].append({"role": "user", "content": input_data.message})
        session["history"].append({"role": "assistant", "content": response.content})
        session["turn_count"] = session.get("turn_count", 0) + 1

        await self._session_manager.update_session(session["id"], session)

        return ConversationOutput(
            response=response.content,
            session_id=session["id"],
            turn_number=session["turn_count"],
        )

    async def _create_session(self, context: AgentContext) -> dict:
        """Create a new conversation session."""
        return await self._session_manager.create_session(
            workspace_id=context.workspace_id,
            user_id=context.user_id,
            agent_name=self.AGENT_NAME,
            ttl_seconds=self.SESSION_TTL,
        )

    def _get_system_prompt(self) -> str:
        return """You are a helpful AI assistant for software development.

You have access to tools to search and retrieve context from the workspace:
- get_issue_context: Get details about issues
- get_note_content: Read notes and documentation
- semantic_search: Search across workspace content

Be helpful, concise, and provide actionable advice.
Reference specific issues, notes, or code when relevant."""
```

### T080: SSE Endpoint

@router.post("/conversation")
async def conversation_stream(
    request: Request,
    body: ConversationRequest,
    current_user = Depends(get_current_user),
    orchestrator = Depends(get_sdk_orchestrator),
):
    """Multi-turn conversation with AI assistant."""
    async def generate():
        input_data = ConversationInput(
            message=body.message,
            session_id=body.session_id,
        )
        context = AgentContext(
            workspace_id=current_user.workspace_id,
            user_id=current_user.id,
        )

        # Stream response
        async for chunk in orchestrator.stream("conversation", input_data, context):
            yield create_sse_event("token", {"content": chunk})

        yield create_sse_event("complete", {"status": "done"})

    return StreamingResponse(generate(), media_type="text/event-stream")

"""Documentation Generator Agent."""
from dataclasses import dataclass
from enum import Enum

from claude_agent_sdk import query

from pilot_space.ai.agents.sdk_base import SDKBaseAgent, AgentContext


class DocType(str, Enum):
    README = "readme"
    API_DOCS = "api_docs"
    ARCHITECTURE = "architecture"
    USER_GUIDE = "user_guide"


@dataclass
class DocGeneratorInput:
    """Input for doc generation."""
    doc_type: DocType
    source_id: str  # Issue, note, or project ID
    template: str | None = None


@dataclass
class DocGeneratorOutput:
    """Generated documentation."""
    content: str
    format: str  # markdown, html
    sections: list[str]


class DocGeneratorAgent(SDKBaseAgent[DocGeneratorInput, DocGeneratorOutput]):
    """Generates documentation from project context."""

    AGENT_NAME = "doc_generator"
    DEFAULT_MODEL = "claude-sonnet-4-20250514"
    MAX_TOKENS = 8192

    async def execute(
        self,
        input_data: DocGeneratorInput,
        context: AgentContext,
    ) -> DocGeneratorOutput:
        options = self._build_options(
            context,
            max_tokens=self.MAX_TOKENS,
            temperature=0.5,
            tools=["get_issue_context", "get_project_context", "get_page_content"],
        )

        prompt = f"""Generate {input_data.doc_type.value} documentation.
Source: {input_data.source_id}

Use the MCP tools to gather context, then generate comprehensive documentation."""

        response = await query(
            prompt=prompt,
            system=self._get_system_prompt(input_data.doc_type),
            options=options,
        )

        return self._parse_response(response)

"""Diagram Generator Agent."""
from dataclasses import dataclass
from enum import Enum

from claude_agent_sdk import query

from pilot_space.ai.agents.sdk_base import SDKBaseAgent, AgentContext


class DiagramType(str, Enum):
    FLOWCHART = "flowchart"
    SEQUENCE = "sequence"
    CLASS = "class"
    ERD = "erd"
    STATE = "state"


@dataclass
class DiagramGeneratorInput:
    """Input for diagram generation."""
    diagram_type: DiagramType
    description: str
    source_id: str | None = None  # Optional context


@dataclass
class DiagramGeneratorOutput:
    """Generated diagram."""
    mermaid_code: str
    diagram_type: DiagramType
    title: str


class DiagramGeneratorAgent(SDKBaseAgent[DiagramGeneratorInput, DiagramGeneratorOutput]):
    """Generates Mermaid diagrams from descriptions."""

    AGENT_NAME = "diagram_generator"
    DEFAULT_MODEL = "claude-sonnet-4-20250514"
    MAX_TOKENS = 2048

    async def execute(
        self,
        input_data: DiagramGeneratorInput,
        context: AgentContext,
    ) -> DiagramGeneratorOutput:
        options = self._build_options(
            context,
            max_tokens=self.MAX_TOKENS,
            temperature=0.3,
            tools=["get_issue_context", "get_note_content"] if input_data.source_id else None,
        )

        prompt = f"""Generate a {input_data.diagram_type.value} diagram in Mermaid syntax.

Description: {input_data.description}

Return valid Mermaid code that renders correctly."""

        response = await query(prompt=prompt, system=self._get_system_prompt(), options=options)
        return self._parse_response(response, input_data.diagram_type)
