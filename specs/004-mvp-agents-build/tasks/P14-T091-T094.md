```python
from datetime import datetime, date
from pydantic import BaseModel


class CostByAgent(BaseModel):
    """Cost breakdown by agent."""
    agent_name: str
    total_cost_usd: float
    request_count: int
    input_tokens: int
    output_tokens: int


class CostByUser(BaseModel):
    """Cost breakdown by user."""
    user_id: str
    user_name: str
    total_cost_usd: float
    request_count: int


class CostByDay(BaseModel):
    """Daily cost summary."""
    date: date
    total_cost_usd: float
    request_count: int


class CostSummaryResponse(BaseModel):
    """Cost summary response."""
    workspace_id: str
    period_start: date
    period_end: date
    total_cost_usd: float
    total_requests: int
    total_input_tokens: int
    total_output_tokens: int
    by_agent: list[CostByAgent]
    by_user: list[CostByUser]
    by_day: list[CostByDay]


@router.get("/costs/summary", response_model=CostSummaryResponse)
async def get_cost_summary(
    start_date: date = Query(default=None),
    end_date: date = Query(default=None),
    current_user = Depends(get_current_user),
    cost_tracker: CostTracker = Depends(get_cost_tracker),
):
    """Get AI cost summary for workspace.

    Default period: last 30 days.
    Requires workspace admin permission.
    """
    await verify_workspace_admin(current_user, current_user.workspace_id)

    # Default to last 30 days
    if not end_date:
        end_date = date.today()
    if not start_date:
        start_date = end_date - timedelta(days=30)

    summary = await cost_tracker.get_summary(
        workspace_id=current_user.workspace_id,
        start_date=start_date,
        end_date=end_date,
    )

    return CostSummaryResponse(
        workspace_id=str(current_user.workspace_id),
        period_start=start_date,
        period_end=end_date,
        total_cost_usd=summary["total_cost"],
        total_requests=summary["total_requests"],
        total_input_tokens=summary["total_input_tokens"],
        total_output_tokens=summary["total_output_tokens"],
        by_agent=summary["by_agent"],
        by_user=summary["by_user"],
        by_day=summary["by_day"],
    )
```

```python
"""Unit tests for CostTracker."""
import pytest
from datetime import date, timedelta
from decimal import Decimal
from uuid import uuid4

from pilot_space.ai.infrastructure.cost_tracker import CostTracker


class TestCostTracker:
    @pytest.mark.asyncio
    async def test_records_cost_with_tokens(self, cost_tracker, db_session):
        workspace_id = uuid4()
        user_id = uuid4()

        await cost_tracker.record(
            workspace_id=workspace_id,
            user_id=user_id,
            agent_name="ghost_text",
            provider="anthropic",
            model="claude-3-5-haiku-20241022",
            input_tokens=100,
            output_tokens=50,
        )

        # Verify recorded
        records = await cost_tracker.get_records(workspace_id, limit=1)
        assert len(records) == 1
        assert records[0].input_tokens == 100
        assert records[0].output_tokens == 50

    @pytest.mark.asyncio
    async def test_calculates_cost_correctly(self, cost_tracker):
        # Haiku pricing: $0.25/1M input, $1.25/1M output
        cost = cost_tracker.calculate_cost(
            provider="anthropic",
            model="claude-3-5-haiku-20241022",
            input_tokens=1000,
            output_tokens=500,
        )

        expected = (1000 * 0.25 / 1_000_000) + (500 * 1.25 / 1_000_000)
        assert abs(cost - expected) < 0.0001

    @pytest.mark.asyncio
    async def test_aggregates_by_agent(self, cost_tracker, db_session, test_workspace):
        # Record multiple costs
        for _ in range(3):
            await cost_tracker.record(
                workspace_id=test_workspace.id,
                user_id=uuid4(),
                agent_name="pr_review",
                provider="anthropic",
                model="claude-opus-4-5-20251101",
                input_tokens=5000,
                output_tokens=2000,
            )

        summary = await cost_tracker.get_summary(
            workspace_id=test_workspace.id,
            start_date=date.today() - timedelta(days=1),
            end_date=date.today(),
        )

        pr_review_costs = [a for a in summary["by_agent"] if a["agent_name"] == "pr_review"]
        assert len(pr_review_costs) == 1
        assert pr_review_costs[0]["request_count"] == 3

    @pytest.mark.asyncio
    async def test_aggregates_by_day(self, cost_tracker, db_session, test_workspace):
        summary = await cost_tracker.get_summary(
            workspace_id=test_workspace.id,
            start_date=date.today() - timedelta(days=7),
            end_date=date.today(),
        )

        assert len(summary["by_day"]) <= 8  # Up to 8 days
```

```python
"""Integration tests for cost tracking endpoint."""
import pytest
from datetime import date, timedelta


class TestCostTrackingEndpoint:
    @pytest.mark.asyncio
    async def test_returns_cost_summary(self, client, auth_headers, test_workspace):
        response = await client.get(
            "/ai/costs/summary",
            headers=auth_headers,
        )

        assert response.status_code == 200
        data = response.json()
        assert "total_cost_usd" in data
        assert "by_agent" in data
        assert "by_user" in data
        assert "by_day" in data

    @pytest.mark.asyncio
    async def test_filters_by_date_range(self, client, auth_headers, test_workspace):
        start = (date.today() - timedelta(days=7)).isoformat()
        end = date.today().isoformat()

        response = await client.get(
            f"/ai/costs/summary?start_date={start}&end_date={end}",
            headers=auth_headers,
        )

        assert response.status_code == 200
        data = response.json()
        assert data["period_start"] == start
        assert data["period_end"] == end

    @pytest.mark.asyncio
    async def test_requires_admin_permission(self, client, member_auth_headers):
        response = await client.get(
            "/ai/costs/summary",
            headers=member_auth_headers,
        )

        assert response.status_code == 403
```
