```python
from enum import Enum
from pydantic import BaseModel
from datetime import datetime


class ApprovalStatus(str, Enum):
    PENDING = "pending"
    APPROVED = "approved"
    REJECTED = "rejected"
    EXPIRED = "expired"


class ApprovalRequestResponse(BaseModel):
    """Approval request in list response."""
    id: str
    agent_name: str
    action_type: str
    status: ApprovalStatus
    created_at: datetime
    expires_at: datetime
    requested_by: str  # User name
    context_preview: str  # Brief context


class ApprovalListResponse(BaseModel):
    """Response for approval list."""
    requests: list[ApprovalRequestResponse]
    total: int
    pending_count: int


@router.get("/approvals", response_model=ApprovalListResponse)
async def list_approvals(
    status: ApprovalStatus | None = None,
    limit: int = Query(default=20, le=100),
    offset: int = Query(default=0, ge=0),
    current_user = Depends(get_current_user),
    approval_service: ApprovalService = Depends(get_approval_service),
):
    """List approval requests for workspace.

    Filters:
    - status: Filter by status (pending, approved, rejected, expired)

    Returns paginated list of approval requests.
    Requires workspace admin permission.
    """
    await verify_workspace_admin(current_user, current_user.workspace_id)

    requests, total = await approval_service.list_requests(
        workspace_id=current_user.workspace_id,
        status=status,
        limit=limit,
        offset=offset,
    )

    pending_count = await approval_service.count_pending(current_user.workspace_id)

    return ApprovalListResponse(
        requests=[
            ApprovalRequestResponse(
                id=str(r.id),
                agent_name=r.agent_name,
                action_type=r.action_type,
                status=ApprovalStatus(r.status),
                created_at=r.created_at,
                expires_at=r.expires_at,
                requested_by=r.user.name if r.user else "Unknown",
                context_preview=_get_context_preview(r.payload),
            )
            for r in requests
        ],
        total=total,
        pending_count=pending_count,
    )


def _get_context_preview(payload: dict) -> str:
    """Generate brief context from payload."""
    if "title" in payload:
        return payload["title"][:100]
    if "issues" in payload:
        return f"{len(payload['issues'])} issues to create"
    return "Action pending approval"
```

```python
class ApprovalResolution(BaseModel):
    """Request to resolve an approval."""
    approved: bool
    note: str | None = None


class ApprovalDetailResponse(BaseModel):
    """Full approval request details."""
    id: str
    agent_name: str
    action_type: str
    status: ApprovalStatus
    payload: dict
    context: dict | None
    created_at: datetime
    expires_at: datetime
    resolved_at: datetime | None
    resolved_by: str | None
    resolution_note: str | None


@router.get("/approvals/{approval_id}", response_model=ApprovalDetailResponse)
async def get_approval(
    approval_id: UUID,
    current_user = Depends(get_current_user),
    approval_service: ApprovalService = Depends(get_approval_service),
):
    """Get approval request details including payload."""
    request = await approval_service.get_request(approval_id)

    if not request or request.workspace_id != current_user.workspace_id:
        raise HTTPException(404, "Approval request not found")

    return ApprovalDetailResponse(
        id=str(request.id),
        agent_name=request.agent_name,
        action_type=request.action_type,
        status=ApprovalStatus(request.status),
        payload=request.payload,
        context=request.context,
        created_at=request.created_at,
        expires_at=request.expires_at,
        resolved_at=request.resolved_at,
        resolved_by=request.resolver.name if request.resolver else None,
        resolution_note=request.resolution_note,
    )


@router.post("/approvals/{approval_id}/resolve")
async def resolve_approval(
    approval_id: UUID,
    body: ApprovalResolution,
    current_user = Depends(get_current_user),
    approval_service: ApprovalService = Depends(get_approval_service),
    orchestrator = Depends(get_sdk_orchestrator),
):
    """Resolve an approval request.

    If approved, executes the pending action.
    If rejected, discards the action.
    """
    await verify_workspace_admin(current_user, current_user.workspace_id)

    request = await approval_service.get_request(approval_id)

    if not request or request.workspace_id != current_user.workspace_id:
        raise HTTPException(404, "Approval request not found")

    if request.status != "pending":
        raise HTTPException(400, f"Request already {request.status}")

    # Resolve the request
    await approval_service.resolve(
        approval_id=approval_id,
        resolved_by=current_user.id,
        approved=body.approved,
        note=body.note,
    )

    result = {"approved": body.approved, "action_result": None}

    # If approved, execute the action
    if body.approved:
        try:
            action_result = await _execute_approved_action(
                orchestrator,
                request.agent_name,
                request.action_type,
                request.payload,
                current_user,
            )
            result["action_result"] = action_result
        except Exception as e:
            result["action_error"] = str(e)

    return result


async def _execute_approved_action(
    orchestrator,
    agent_name: str,
    action_type: str,
    payload: dict,
    current_user,
) -> dict:
    """Execute the approved action."""
    if action_type == "create_issues":
        # Create issues from extraction
        created_ids = []
        for issue_data in payload.get("issues", []):
            issue = await issue_service.create(
                project_id=current_user.current_project_id,
                **issue_data,
            )
            created_ids.append(str(issue.id))
        return {"created_issues": created_ids}

    # Add other action types as needed
    return {"executed": True}
```

```python
"""Approval expiration job.

Runs periodically to mark expired pending approvals as expired.
"""
from datetime import datetime, timezone

from sqlalchemy import update
from sqlalchemy.ext.asyncio import AsyncSession

from pilot_space.infrastructure.database.models import AIApprovalRequest


async def expire_pending_approvals(db_session: AsyncSession) -> int:
    """Mark expired pending approvals as expired.

    Returns number of expired approvals.
    """
    now = datetime.now(timezone.utc)

    result = await db_session.execute(
        update(AIApprovalRequest)
        .where(
            AIApprovalRequest.status == "pending",
            AIApprovalRequest.expires_at < now,
        )
        .values(status="expired")
    )

    await db_session.commit()

    return result.rowcount


# Scheduled with APScheduler or similar
async def run_expiration_job():
    """Run the expiration job."""
    from pilot_space.infrastructure.database.connection import get_session

    async with get_session() as session:
        count = await expire_pending_approvals(session)
        if count > 0:
            logger.info(f"Expired {count} approval requests")
```
