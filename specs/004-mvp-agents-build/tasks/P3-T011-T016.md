```python
from supabase import AsyncClient
from pilot_space.core.config import settings
import structlog

logger = structlog.get_logger()

class SecureKeyStorage:
    """Secure API key storage using Supabase Vault (DD-002).

    All keys are encrypted at rest using Supabase Vault.
    Keys are never logged or exposed in error messages.
    """

    def __init__(self, supabase_client: AsyncClient):
        self.client = supabase_client

    async def store_api_key(
        self,
        workspace_id: str,
        provider: str,
        api_key: str,
    ) -> None:
        """Store encrypted API key for workspace."""
        # Encrypt using Supabase Vault
        encrypted = await self._encrypt(api_key)

        # Upsert to avoid duplicates
        await self.client.table("workspace_api_keys").upsert({
            "workspace_id": workspace_id,
            "provider": provider,
            "encrypted_key": encrypted,
            "updated_at": "now()",
        }, on_conflict="workspace_id,provider").execute()

        logger.info(
            "api_key_stored",
            workspace_id=workspace_id,
            provider=provider,
            key_preview=f"{api_key[:4]}...{api_key[-4:]}",
        )

    async def get_api_key(
        self,
        workspace_id: str,
        provider: str,
    ) -> str | None:
        """Retrieve and decrypt API key."""
        result = await self.client.table("workspace_api_keys").select(
            "encrypted_key"
        ).eq("workspace_id", workspace_id).eq("provider", provider).single().execute()

        if not result.data:
            return None

        return await self._decrypt(result.data["encrypted_key"])

    async def validate_api_key(
        self,
        provider: str,
        api_key: str,
    ) -> bool:
        """Validate API key by making test call to provider."""
        try:
            if provider == "anthropic":
                from anthropic import AsyncAnthropic
                client = AsyncAnthropic(api_key=api_key)
                await client.messages.create(
                    model="claude-3-5-haiku-20241022",
                    max_tokens=1,
                    messages=[{"role": "user", "content": "test"}],
                )
            elif provider == "openai":
                from openai import AsyncOpenAI
                client = AsyncOpenAI(api_key=api_key)
                await client.models.list()
            elif provider == "google":
                import google.generativeai as genai
                genai.configure(api_key=api_key)
                model = genai.GenerativeModel("gemini-2.0-flash")
                await model.generate_content_async("test")
            return True
        except Exception as e:
            logger.warning("api_key_validation_failed", provider=provider, error=str(e))
            return False

    async def _encrypt(self, value: str) -> str:
        """Encrypt value using Supabase Vault."""
        result = await self.client.rpc(
            "vault.create_secret",
            {"secret": value, "name": f"api_key_{id(value)}"}
        ).execute()
        return result.data

    async def _decrypt(self, secret_id: str) -> str:
        """Decrypt value from Supabase Vault."""
        result = await self.client.rpc(
            "vault.decrypt_secret",
            {"id": secret_id}
        ).execute()
        return result.data
```

```python
from dataclasses import dataclass
from datetime import datetime, timedelta, UTC
from enum import Enum
from uuid import uuid4
from sqlalchemy.ext.asyncio import AsyncSession

class ApprovalStatus(Enum):
    PENDING = "pending"
    APPROVED = "approved"
    REJECTED = "rejected"
    EXPIRED = "expired"

@dataclass
class AIApprovalRequest:
    id: str
    user_id: str
    workspace_id: str
    action_type: str
    description: str
    payload: dict
    confidence: float
    status: ApprovalStatus
    expires_at: datetime
    created_at: datetime
    resolved_at: datetime | None = None
    resolved_by: str | None = None

class ApprovalService:
    """Human-in-the-loop approval for AI actions (DD-003)."""

    DEFAULT_EXPIRATION_HOURS = 24

    def __init__(
        self,
        db: AsyncSession,
        notification_service: "NotificationService",
    ):
        self.db = db
        self.notifications = notification_service

    async def check_approval_required(
        self,
        action_type: str,
        project_settings: dict | None = None,
    ) -> bool:
        """Check if action requires approval."""
        if action_type in self.ALWAYS_REQUIRE_APPROVAL:
            return True

        if action_type in self.DEFAULT_REQUIRE_APPROVAL:
            # Check project override
            if project_settings:
                disabled = project_settings.get("disabled_approvals", [])
                return action_type not in disabled
            return True

        return False

    async def create_approval_request(
        self,
        user_id: str,
        workspace_id: str,
        action_type: str,
        description: str,
        payload: dict,
        confidence: float = 0.8,
        expiration_hours: int | None = None,
    ) -> AIApprovalRequest:
        """Create new approval request."""
        request = AIApprovalRequest(
            id=str(uuid4()),
            user_id=user_id,
            workspace_id=workspace_id,
            action_type=action_type,
            description=description,
            payload=payload,
            confidence=confidence,
            status=ApprovalStatus.PENDING,
            expires_at=datetime.now(UTC) + timedelta(
                hours=expiration_hours or self.DEFAULT_EXPIRATION_HOURS
            ),
            created_at=datetime.now(UTC),
        )

        # Persist
        await self._store(request)

        # Notify
        await self.notifications.send_approval_request(
            user_id=user_id,
            workspace_id=workspace_id,
            request=request,
        )

        return request

    async def resolve(
        self,
        request_id: str,
        approved: bool,
        resolved_by: str,
    ) -> AIApprovalRequest:
        """Resolve approval request."""
        request = await self.get_by_id(request_id)
        if not request:
            raise ValueError(f"Approval request not found: {request_id}")

        if request.status != ApprovalStatus.PENDING:
            raise ValueError(f"Request already resolved: {request.status}")

        if datetime.now(UTC) > request.expires_at:
            request.status = ApprovalStatus.EXPIRED
            await self._update(request)
            raise ValueError("Request has expired")

        request.status = ApprovalStatus.APPROVED if approved else ApprovalStatus.REJECTED
        request.resolved_at = datetime.now(UTC)
        request.resolved_by = resolved_by

        await self._update(request)
        return request
```

```python
from dataclasses import dataclass, field, asdict
from datetime import datetime, timedelta, UTC
from typing import Any
from uuid import uuid4
import json
from redis.asyncio import Redis

@dataclass
class AISession:
    id: str
    user_id: str
    agent_type: str
    context: dict[str, Any] = field(default_factory=dict)
    messages: list[dict] = field(default_factory=list)
    created_at: datetime = field(default_factory=lambda: datetime.now(UTC))
    last_activity: datetime = field(default_factory=lambda: datetime.now(UTC))
    total_cost_usd: float = 0.0

    def is_expired(self, timeout_minutes: int = 30) -> bool:
        return datetime.now(UTC) - self.last_activity > timedelta(minutes=timeout_minutes)

class SessionManager:
    """Manage AI conversation sessions in Redis."""

    def __init__(self, redis_client: Redis):
        self.redis = redis_client
        self.session_ttl_seconds = 1800  # 30 minutes

    async def create_session(
        self,
        user_id: str,
        agent_type: str,
        initial_context: dict | None = None,
    ) -> AISession:
        """Create new AI session."""
        session = AISession(
            id=str(uuid4()),
            user_id=user_id,
            agent_type=agent_type,
            context=initial_context or {},
        )
        await self._store(session)
        return session

    async def get_session(self, session_id: str) -> AISession | None:
        """Retrieve session by ID."""
        data = await self.redis.get(f"ai_session:{session_id}")
        if not data:
            return None
        return self._deserialize(data)

    async def update_session(
        self,
        session_id: str,
        message: dict | None = None,
        context_update: dict | None = None,
        cost_delta: float = 0.0,
    ) -> AISession | None:
        """Update session with new data."""
        session = await self.get_session(session_id)
        if not session:
            return None

        if message:
            session.messages.append(message)
        if context_update:
            session.context.update(context_update)

        session.total_cost_usd += cost_delta
        session.last_activity = datetime.now(UTC)

        await self._store(session)
        return session

    async def end_session(self, session_id: str) -> None:
        """End and cleanup session."""
        await self.redis.delete(f"ai_session:{session_id}")

    async def _store(self, session: AISession) -> None:
        """Store session in Redis."""
        await self.redis.setex(
            f"ai_session:{session.id}",
            self.session_ttl_seconds,
            self._serialize(session),
        )

    def _serialize(self, session: AISession) -> str:
        """Serialize session to JSON."""
        data = asdict(session)
        data["created_at"] = session.created_at.isoformat()
        data["last_activity"] = session.last_activity.isoformat()
        return json.dumps(data)

    def _deserialize(self, data: bytes) -> AISession:
        """Deserialize session from JSON."""
        obj = json.loads(data)
        obj["created_at"] = datetime.fromisoformat(obj["created_at"])
        obj["last_activity"] = datetime.fromisoformat(obj["last_activity"])
        return AISession(**obj)
```

```python
from dataclasses import dataclass
from datetime import datetime, timedelta, UTC
from typing import TypeVar, Callable, AsyncIterator
import asyncio
import random
import structlog

logger = structlog.get_logger()
T = TypeVar("T")

@dataclass
class CircuitBreakerConfig:
    failure_threshold: int = 5
    success_threshold: int = 1
    timeout_seconds: float = 60.0
    half_open_max_calls: int = 1

class CircuitBreakerState:
    """Circuit breaker state machine."""

    def __init__(self, config: CircuitBreakerConfig = CircuitBreakerConfig()):
        self.config = config
        self.failures = 0
        self.successes = 0
        self.last_failure: datetime | None = None
        self.state = "closed"

    def record_failure(self) -> None:
        self.failures += 1
        self.last_failure = datetime.now(UTC)
        if self.failures >= self.config.failure_threshold:
            self.state = "open"
            logger.warning("circuit_breaker_opened", failures=self.failures)

    def record_success(self) -> None:
        if self.state == "half-open":
            self.successes += 1
            if self.successes >= self.config.success_threshold:
                self._reset()
        else:
            self.failures = 0

    def can_attempt(self) -> bool:
        if self.state == "closed":
            return True
        if self.state == "open":
            if self.last_failure and (
                datetime.now(UTC) - self.last_failure
                > timedelta(seconds=self.config.timeout_seconds)
            ):
                self.state = "half-open"
                self.successes = 0
                return True
            return False
        return True  # half-open

    def _reset(self) -> None:
        self.state = "closed"
        self.failures = 0
        self.successes = 0

class CircuitBreaker:
    """Per-provider circuit breaker registry."""

    _instances: dict[str, CircuitBreakerState] = {}

    @classmethod
    def get_or_create(cls, name: str) -> CircuitBreakerState:
        if name not in cls._instances:
            cls._instances[name] = CircuitBreakerState()
        return cls._instances[name]

class ResilientExecutor:
    """Execute AI operations with retry, timeout, and circuit breaker."""

    def __init__(self):
        pass

    async def execute(
        self,
        provider: str,
        operation: Callable[[], AsyncIterator[T]],
        timeout_sec: int = 300,
        max_retries: int = 3,
    ) -> AsyncIterator[T]:
        """Execute with resilience patterns."""
        circuit_breaker = CircuitBreaker.get_or_create(provider)

        if not circuit_breaker.can_attempt():
            raise CircuitBreakerOpenError(f"Circuit breaker open for: {provider}")

        last_error = None

        for attempt in range(max_retries):
            try:
                async with asyncio.timeout(timeout_sec):
                    async for item in operation():
                        yield item

                circuit_breaker.record_success()
                return

            except asyncio.TimeoutError as e:
                last_error = e
                circuit_breaker.record_failure()
                logger.warning("timeout_retry", provider=provider, attempt=attempt + 1)

                if attempt < max_retries - 1:
                    delay = (2 ** attempt) + random.uniform(0, 1)
                    await asyncio.sleep(delay)

            except Exception as e:
                last_error = e
                circuit_breaker.record_failure()
                logger.error("operation_error", provider=provider, error=str(e))

                if attempt < max_retries - 1:
                    delay = (2 ** attempt) + random.uniform(0, 1)
                    await asyncio.sleep(delay)

        raise last_error or TimeoutError(f"All retries exhausted for {provider}")


class CircuitBreakerOpenError(Exception):
    """Raised when circuit breaker is open."""
    pass
```
