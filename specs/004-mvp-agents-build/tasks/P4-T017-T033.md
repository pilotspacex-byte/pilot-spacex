```python
"""MCP Server Factory for Claude Agent SDK.

Creates configured MCP servers with Pilot Space tools for agent use.

Reference: docs/architect/claude-agent-sdk-architecture.md:2200-2400
"""
from collections.abc import Callable
from typing import Any

from claude_agent_sdk import create_sdk_mcp_server, tool
from sqlalchemy.ext.asyncio import AsyncSession

# Tool registry
_TOOLS: dict[str, Callable] = {}
_TOOL_CATEGORIES: dict[str, list[str]] = {
    "database": [],
    "github": [],
    "search": [],
}


def register_tool(category: str):
    """Decorator to register an MCP tool.

    Usage:
        @register_tool("database")
        @tool
        async def get_issue_context(...):
            ...
    """
    def decorator(func: Callable) -> Callable:
        _TOOLS[func.__name__] = func
        _TOOL_CATEGORIES[category].append(func.__name__)
        return func
    return decorator


class MCPServerFactory:
    """Factory for creating MCP servers with Pilot Space tools.

    Each agent requests a server with specific tools enabled.
    The factory provides read-only data access to the database.

    Usage:
        factory = MCPServerFactory(db_session)
        server = factory.create(tools=["get_issue_context", "get_note_content"])
    """

    def __init__(self, db_session: AsyncSession) -> None:
        """Initialize factory with database session.

        Args:
            db_session: SQLAlchemy async session for data access
        """
        self._db_session = db_session

    def create(
        self,
        tools: list[str] | None = None,
        categories: list[str] | None = None,
    ) -> Any:
        """Create MCP server with selected tools.

        Args:
            tools: Specific tool names to include
            categories: Tool categories to include (database, github, search)

        Returns:
            Configured MCP server instance
        """
        # Determine which tools to include
        selected_tools: list[Callable] = []

        if tools:
            for name in tools:
                if name in _TOOLS:
                    selected_tools.append(_TOOLS[name])

        if categories:
            for category in categories:
                if category in _TOOL_CATEGORIES:
                    for name in _TOOL_CATEGORIES[category]:
                        if _TOOLS[name] not in selected_tools:
                            selected_tools.append(_TOOLS[name])

        # If nothing specified, include all tools
        if not tools and not categories:
            selected_tools = list(_TOOLS.values())

        # Create MCP server with tools
        return create_sdk_mcp_server(
            tools=selected_tools,
            context={"db_session": self._db_session},
        )

    @classmethod
    def get_all_tools(cls) -> list[str]:
        """Get list of all registered tool names."""
        return list(_TOOLS.keys())

    @classmethod
    def get_tools_by_category(cls, category: str) -> list[str]:
        """Get tool names for a category."""
        return _TOOL_CATEGORIES.get(category, [])
```

```python
"""Database MCP tools for Pilot Space data access.

These tools provide read-only access to issues, notes, projects.
Write tools require approval flow (DD-003).
"""
from datetime import datetime
from typing import Any
from uuid import UUID

from claude_agent_sdk import tool
from sqlalchemy import select
from sqlalchemy.orm import selectinload

from pilot_space.ai.tools.mcp_server import register_tool
from pilot_space.domain.models.issue import Issue
from pilot_space.domain.models.note import Note


@register_tool("database")
@tool
async def get_issue_context(
    issue_id: str,
    include_notes: bool = True,
    include_related: bool = True,
    include_activity: bool = True,
    context: dict | None = None,
) -> dict[str, Any]:
    """Get comprehensive context for an issue.

    Retrieves issue details along with related notes, linked issues,
    and recent activity. Used by AIContextAgent for issue understanding.

    Args:
        issue_id: UUID of the issue
        include_notes: Whether to include linked notes (default True)
        include_related: Whether to include related issues (default True)
        include_activity: Whether to include activity log (default True)
        context: Tool context with db_session

    Returns:
        Dict with issue details and related data:
        {
            "issue": {...},
            "notes": [...],
            "related_issues": {...},
            "activity": [...]
        }
    """
    db_session = context["db_session"]
    uuid_id = UUID(issue_id)

    # Build query with eager loading to avoid N+1
    query = select(Issue).where(Issue.id == uuid_id)

    if include_notes:
        query = query.options(selectinload(Issue.linked_notes))
    if include_related:
        query = query.options(
            selectinload(Issue.parent),
            selectinload(Issue.children),
            selectinload(Issue.linked_issues),
        )

    result = await db_session.execute(query)
    issue = result.scalar_one_or_none()

    if not issue:
        return {"error": f"Issue {issue_id} not found"}

    response = {
        "issue": {
            "id": str(issue.id),
            "identifier": issue.identifier,
            "title": issue.title,
            "description": issue.description,
            "state": issue.state.value if issue.state else None,
            "priority": issue.priority,
            "labels": [label.name for label in issue.labels] if issue.labels else [],
            "assignee": issue.assignee.name if issue.assignee else None,
            "created_at": issue.created_at.isoformat() if issue.created_at else None,
            "updated_at": issue.updated_at.isoformat() if issue.updated_at else None,
        }
    }

    if include_notes and issue.linked_notes:
        response["notes"] = [
            {
                "id": str(note.id),
                "title": note.title,
                "content_preview": note.content[:500] if note.content else "",
            }
            for note in issue.linked_notes[:5]  # Limit to 5 notes
        ]

    if include_related:
        response["related_issues"] = {
            "parent": _issue_summary(issue.parent) if issue.parent else None,
            "children": [_issue_summary(c) for c in issue.children[:10]],
            "linked": [_issue_summary(l) for l in issue.linked_issues[:10]],
        }

    if include_activity:
        # Fetch recent activity separately (last 20 items)
        response["activity"] = await _get_issue_activity(db_session, uuid_id)

    return response

        "identifier": issue.identifier,
        "title": issue.title,
        "state": issue.state.value if issue.state else None,
    }
```

```python
@register_tool("database")
@tool
async def find_similar_issues(
    query_text: str,
    project_id: str | None = None,
    limit: int = 5,
    context: dict | None = None,
) -> dict[str, Any]:
    """Find similar issues using embedding similarity.

    Uses vector search to find semantically similar issues.
    Useful for duplicate detection and related issue discovery.

    Args:
        query_text: Text to find similar issues for
        project_id: Optional project ID to scope search
        limit: Maximum results (default 5)
        context: Tool context with db_session

    Returns:
        List of similar issues with similarity scores
    """
    db_session = context["db_session"]

    # Get embedding for query text
    # (Implementation depends on embedding service)
    embedding = await _get_embedding(query_text)

    # Vector similarity search
    query = (
        select(Issue)
        .where(Issue.embedding.isnot(None))
        .order_by(Issue.embedding.cosine_distance(embedding))
        .limit(limit)
    )

    if project_id:
        query = query.where(Issue.project_id == UUID(project_id))

    result = await db_session.execute(query)
    issues = result.scalars().all()

    return {
        "similar_issues": [
            {
                "id": str(issue.id),
                "identifier": issue.identifier,
                "title": issue.title,
                "state": issue.state.value if issue.state else None,
                "similarity_score": 1 - issue.embedding.cosine_distance(embedding),
            }
            for issue in issues
        ]
    }
```

```python
    Retrieves cycle details including velocity and progress.

    """

    # Calculate metrics
    total_issues = len(cycle.issues)
    completed = sum(1 for i in cycle.issues if i.state and i.state.value == "done")

    return {
        "cycle": {
            "id": str(cycle.id),
            "name": cycle.name,
            "start_date": cycle.start_date.isoformat() if cycle.start_date else None,
            "end_date": cycle.end_date.isoformat() if cycle.end_date else None,
            "status": cycle.status.value if cycle.status else None,
        },
        "metrics": {
            "total_issues": total_issues,
            "completed_issues": completed,
            "progress_percent": (completed / total_issues * 100) if total_issues > 0 else 0,
        }
    }
```

```python
@register_tool("search")

async def semantic_search(
    query: str,
    workspace_id: str,
    content_types: list[str] | None = None,
    limit: int = 10,
    context: dict | None = None,
) -> dict[str, Any]:
    """Semantic search across workspace content.

    Uses embeddings to find relevant content across issues,
    notes, pages, and comments.

    Args:
        query: Search query text
        workspace_id: UUID of workspace to search
        content_types: Filter by type (issue, note, page, comment)
        limit: Maximum results (default 10)
        context: Tool context with db_session

    Returns:
        Search results with relevance scores
    """
    db_session = context["db_session"]

    # Get query embedding
    embedding = await _get_embedding(query)

    results = []

    # Search issues
    if not content_types or "issue" in content_types:
        issue_results = await _search_issues(db_session, workspace_id, embedding, limit)
        results.extend(issue_results)

    # Search notes
    if not content_types or "note" in content_types:
        note_results = await _search_notes(db_session, workspace_id, embedding, limit)
        results.extend(note_results)

    # Sort by relevance and limit
    results.sort(key=lambda x: x["score"], reverse=True)
    return {"results": results[:limit]}
```

```python
@register_tool("github")
@tool
async def post_pr_comment(
    pr_number: int,
    repo_id: str,
    body: str,
    path: str | None = None,
    line: int | None = None,
    context: dict | None = None,
) -> dict[str, Any]:
    """Post a comment on a pull request.

    AUTO_EXECUTE action (DD-003) with notification.

    Args:
        pr_number: PR number
        repo_id: UUID of the repository integration
        body: Comment body (markdown)
        path: Optional file path for line comment
        line: Optional line number for line comment
        context: Tool context with db_session

    Returns:
        Posted comment details
    """
    db_session = context["db_session"]

    repo = await _get_repo_integration(db_session, UUID(repo_id))
    if not repo:
        return {"error": f"Repository {repo_id} not found"}

    github_client = await _get_github_client(repo.access_token)

    if path and line:
        # Line comment
        comment = await github_client.create_review_comment(
            repo.owner, repo.name, pr_number, body, path, line
        )
    else:
        # General PR comment
        comment = await github_client.create_issue_comment(
            repo.owner, repo.name, pr_number, body
        )

    return {
        "comment": {
            "id": comment.id,
            "url": comment.html_url,
            "body": body,
        }
    }
```
