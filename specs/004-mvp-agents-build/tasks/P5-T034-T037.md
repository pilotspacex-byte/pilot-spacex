"""SDK Base Agent for Claude Agent SDK.

All agents extend this class to inherit telemetry, cost tracking,
and MCP tool integration.

Reference: docs/architect/claude-agent-sdk-architecture.md:2027-2123
"""
from abc import ABC, abstractmethod
from collections.abc import AsyncIterator
from dataclasses import dataclass
from typing import Generic, TypeVar
from uuid import UUID

from claude_agent_sdk import ClaudeAgentOptions, ClaudeSDKClient

from pilot_space.ai.infrastructure.cost_tracker import CostTracker
from pilot_space.ai.infrastructure.resilience import ResilientExecutor
from pilot_space.ai.providers.provider_selector import ProviderSelector
from pilot_space.ai.telemetry import AIOperation, track_ai_operation
from pilot_space.ai.tools.mcp_server import MCPServerFactory

InputT = TypeVar("InputT")
OutputT = TypeVar("OutputT")


@dataclass
class AgentContext:
    """Execution context for an agent."""

    workspace_id: UUID
    user_id: UUID
    operation_id: UUID | None = None


class SDKBaseAgent(ABC, Generic[InputT, OutputT]):
    """Abstract base class for Claude Agent SDK agents.

    Provides:
    - MCP tool access via mcp_server
    - Provider/model selection via provider_selector
    - Cost tracking via cost_tracker
    - Resilience via resilient_executor
    - Telemetry via _track_operation()

    Subclasses must implement:
    - AGENT_NAME: str class attribute
    - DEFAULT_MODEL: str class attribute
    - execute(): Main entry point
    """

    AGENT_NAME: str = "base"  # Override in subclass
    DEFAULT_MODEL: str = "claude-sonnet-4-20250514"  # Override in subclass

    def __init__(
        self,
        mcp_server: MCPServerFactory,
        provider_selector: ProviderSelector,
        cost_tracker: CostTracker,
        resilient_executor: ResilientExecutor,
    ) -> None:
        """Initialize agent with infrastructure dependencies."""
        self._mcp_server = mcp_server
        self._provider_selector = provider_selector
        self._cost_tracker = cost_tracker
        self._resilient_executor = resilient_executor

    def _build_options(
        self,
        context: AgentContext,
        *,
        max_tokens: int | None = None,
        temperature: float = 0.7,
        tools: list[str] | None = None,
    ) -> ClaudeAgentOptions:
        """Build ClaudeAgentOptions with telemetry hooks.

        Args:
            context: Execution context with workspace/user IDs
            max_tokens: Maximum tokens to generate (None = model default)
            temperature: Sampling temperature (0.0-1.0)
            tools: List of MCP tool names to enable

        Returns:
            Configured ClaudeAgentOptions
        """
        # Select provider and model for this agent type
        provider, model = self._provider_selector.select(self.AGENT_NAME)
        model = model or self.DEFAULT_MODEL

        # Get MCP server with selected tools
        mcp = self._mcp_server.create(tools=tools) if tools else None

        return ClaudeAgentOptions(
            model=model,
            max_tokens=max_tokens,
            temperature=temperature,
            mcp_server=mcp,
            metadata={
                "workspace_id": str(context.workspace_id),
                "user_id": str(context.user_id),
                "agent_name": self.AGENT_NAME,
                "operation_id": str(context.operation_id) if context.operation_id else None,
            },
            on_token_usage=lambda usage: self._on_token_usage(context, provider, model, usage),
        )

    def _on_token_usage(
        self,
        context: AgentContext,
        provider: str,
        model: str,
        usage: dict,
    ) -> None:
        """Callback for token usage tracking.

        Called by SDK after each API call with usage stats.
        """
        self._cost_tracker.record(
            workspace_id=context.workspace_id,
            user_id=context.user_id,
            agent_name=self.AGENT_NAME,
            provider=provider,
            model=model,
            input_tokens=usage.get("input_tokens", 0),
            output_tokens=usage.get("output_tokens", 0),
            operation_id=context.operation_id,
        )

    @track_ai_operation(AIOperation.AGENT_EXECUTE)
    async def run(
        self,
        input_data: InputT,
        context: AgentContext,
    ) -> OutputT:
        """Execute agent with telemetry wrapper.

        This is the public entry point. It wraps execute() with
        telemetry and error handling.
        """
        async def _execute() -> OutputT:
            return await self.execute(input_data, context)

        return await self._resilient_executor.execute(_execute)

    @abstractmethod
    async def execute(
        self,
        input_data: InputT,
        context: AgentContext,
    ) -> OutputT:
        """Execute the agent logic.

        Subclasses implement this method with their specific logic.

        Args:
            input_data: Agent-specific input data
            context: Execution context

        Returns:
            Agent-specific output
        """
        ...


class StreamingSDKBaseAgent(SDKBaseAgent[InputT, OutputT], ABC):
    """Base class for streaming agents (ghost text, AI context).

    Adds streaming-specific methods and configuration.
    """

    @abstractmethod
    async def stream(
        self,
        input_data: InputT,
        context: AgentContext,
    ) -> AsyncIterator[str]:
        """Stream output tokens as they're generated.

        Args:
            input_data: Agent-specific input data
            context: Execution context

        Yields:
            Output tokens/chunks as strings
        """
        ...

    async def execute(
        self,
        input_data: InputT,
        context: AgentContext,
    ) -> OutputT:
        """Execute by collecting stream into final output.

        Default implementation collects all streamed tokens.
        Override if output format differs from concatenated stream.
        """
        chunks: list[str] = []
        async for chunk in self.stream(input_data, context):
            chunks.append(chunk)
        return "".join(chunks)  # type: ignore
```

"""SDK Orchestrator for Claude Agent SDK.

Central coordinator for all AI agents, integrating infrastructure
services for approval, cost tracking, and session management.

Reference: docs/architect/claude-agent-sdk-architecture.md
"""
from collections.abc import AsyncIterator
from dataclasses import dataclass
from enum import Enum
from typing import Any
from uuid import UUID

from pilot_space.ai.agents.sdk_base import AgentContext, SDKBaseAgent, StreamingSDKBaseAgent
from pilot_space.ai.infrastructure.approval import ApprovalService, ActionClassification
from pilot_space.ai.infrastructure.cost_tracker import CostTracker
from pilot_space.ai.infrastructure.key_storage import SecureKeyStorage
from pilot_space.ai.infrastructure.resilience import ResilientExecutor
from pilot_space.ai.providers.provider_selector import ProviderSelector
from pilot_space.ai.session.session_manager import SessionManager
from pilot_space.ai.tools.mcp_server import MCPServerFactory


class AgentName(str, Enum):
    """Registered agent names."""

    GHOST_TEXT = "ghost_text"
    MARGIN_ANNOTATION = "margin_annotation"
    ISSUE_EXTRACTOR = "issue_extractor"
    AI_CONTEXT = "ai_context"
    CONVERSATION = "conversation"
    PR_REVIEW = "pr_review"
    DOC_GENERATOR = "doc_generator"
    TASK_DECOMPOSER = "task_decomposer"
    DIAGRAM_GENERATOR = "diagram_generator"
    ISSUE_ENHANCER = "issue_enhancer"
    ASSIGNEE_RECOMMENDER = "assignee_recommender"
    DUPLICATE_DETECTOR = "duplicate_detector"


@dataclass
class ExecutionResult:
    """Result from agent execution."""

    success: bool
    output: Any
    cost_usd: float
    requires_approval: bool = False
    approval_id: UUID | None = None


class SDKOrchestrator:
    """Main orchestrator for Claude Agent SDK agents.

    Responsibilities:
    - Agent registration and routing
    - Approval flow integration (DD-003)
    - Session management for multi-turn agents
    - Cost tracking aggregation
    - SSE streaming coordination

    Usage:
        orchestrator = SDKOrchestrator(...)
        result = await orchestrator.execute("pr_review", input_data, context)
    """

    def __init__(
        self,
        key_storage: SecureKeyStorage,
        approval_service: ApprovalService,
        cost_tracker: CostTracker,
        session_manager: SessionManager,
        provider_selector: ProviderSelector,
        resilient_executor: ResilientExecutor,
        mcp_server_factory: MCPServerFactory,
    ) -> None:
        """Initialize orchestrator with infrastructure dependencies."""
        self._key_storage = key_storage
        self._approval_service = approval_service
        self._cost_tracker = cost_tracker
        self._session_manager = session_manager
        self._provider_selector = provider_selector
        self._resilient_executor = resilient_executor
        self._mcp_server_factory = mcp_server_factory

        # Agent registry - populated by register_agents()
        self._agents: dict[str, SDKBaseAgent] = {}

    def register_agent(self, name: str, agent: SDKBaseAgent) -> None:
        """Register an agent instance.

        Args:
            name: Agent identifier (use AgentName enum values)
            agent: Configured agent instance
        """
        self._agents[name] = agent

    def register_agents(self) -> None:
        """Register all available agents.

        Called during application startup to populate the agent registry.
        Each agent is initialized with shared infrastructure dependencies.
        """
        # Import agents here to avoid circular imports
        from pilot_space.ai.agents.ghost_text_agent import GhostTextAgent
        from pilot_space.ai.agents.margin_annotation_agent import MarginAnnotationAgent
        from pilot_space.ai.agents.issue_extractor_agent import IssueExtractorAgent
        from pilot_space.ai.agents.ai_context_agent import AIContextAgent
        from pilot_space.ai.agents.conversation_agent import ConversationAgent
        from pilot_space.ai.agents.pr_review_agent import PRReviewAgent

        # Common dependencies
        deps = {
            "mcp_server": self._mcp_server_factory,
            "provider_selector": self._provider_selector,
            "cost_tracker": self._cost_tracker,
            "resilient_executor": self._resilient_executor,
        }

        # Register agents
        self._agents[AgentName.GHOST_TEXT] = GhostTextAgent(**deps)
        self._agents[AgentName.MARGIN_ANNOTATION] = MarginAnnotationAgent(**deps)
        self._agents[AgentName.ISSUE_EXTRACTOR] = IssueExtractorAgent(**deps)
        self._agents[AgentName.AI_CONTEXT] = AIContextAgent(
            **deps,
            session_manager=self._session_manager,
        )
        self._agents[AgentName.CONVERSATION] = ConversationAgent(
            **deps,
            session_manager=self._session_manager,
        )
        self._agents[AgentName.PR_REVIEW] = PRReviewAgent(**deps)

    async def execute(
        self,
        agent_name: str,
        input_data: Any,
        context: AgentContext,
    ) -> ExecutionResult:
        """Execute an agent without approval flow.

        Use for actions that don't require human approval.

        Args:
            agent_name: Name of agent to execute
            input_data: Agent-specific input
            context: Execution context

        Returns:
            ExecutionResult with output and cost

        Raises:
            KeyError: If agent_name not registered
        """
        agent = self._agents[agent_name]

        # Check API key availability
        await self._ensure_api_key(context.workspace_id)

        output = await agent.run(input_data, context)

        # Get cost from tracker
        cost = self._cost_tracker.get_operation_cost(context.operation_id)

        return ExecutionResult(
            success=True,
            output=output,
            cost_usd=cost,
        )

    async def execute_with_approval(
        self,
        agent_name: str,
        action_type: str,
        input_data: Any,
        context: AgentContext,
        *,
        payload: dict[str, Any] | None = None,
    ) -> ExecutionResult:
        """Execute an agent with approval flow (DD-003).

        For actions classified as ALWAYS_REQUIRE or DEFAULT_REQUIRE,
        creates an approval request and returns without executing.

        Args:
            agent_name: Name of agent to execute
            action_type: Type of action for classification
            input_data: Agent-specific input
            context: Execution context
            payload: Optional approval payload (defaults to input_data)

        Returns:
            ExecutionResult - if requires_approval=True, check approval_id
        """
        # Classify action
        classification = self._approval_service.classify(action_type)

        if classification == ActionClassification.ALWAYS_REQUIRE_APPROVAL:
            # Create approval request, don't execute
            approval_id = await self._approval_service.create_request(
                workspace_id=context.workspace_id,
                user_id=context.user_id,
                agent_name=agent_name,
                action_type=action_type,
                payload=payload or {"input": input_data},
            )
            return ExecutionResult(
                success=True,
                output=None,
                cost_usd=0,
                requires_approval=True,
                approval_id=approval_id,
            )

        if classification == ActionClassification.DEFAULT_REQUIRE_APPROVAL:
            # Check workspace setting
            requires_approval = await self._approval_service.workspace_requires_approval(
                context.workspace_id, action_type
            )
            if requires_approval:
                approval_id = await self._approval_service.create_request(
                    workspace_id=context.workspace_id,
                    user_id=context.user_id,
                    agent_name=agent_name,
                    action_type=action_type,
                    payload=payload or {"input": input_data},
                )
                return ExecutionResult(
                    success=True,
                    output=None,
                    cost_usd=0,
                    requires_approval=True,
                    approval_id=approval_id,
                )

        # AUTO_EXECUTE: proceed with execution
        return await self.execute(agent_name, input_data, context)

    async def stream(
        self,
        agent_name: str,
        input_data: Any,
        context: AgentContext,
    ) -> AsyncIterator[str]:
        """Stream output from a streaming agent.

        Args:
            agent_name: Name of streaming agent
            input_data: Agent-specific input
            context: Execution context

        Yields:
            Output tokens/chunks as strings

        Raises:
            TypeError: If agent doesn't support streaming
        """
        agent = self._agents[agent_name]

        if not isinstance(agent, StreamingSDKBaseAgent):
            raise TypeError(f"Agent {agent_name} does not support streaming")

        await self._ensure_api_key(context.workspace_id)

        async for chunk in agent.stream(input_data, context):
            yield chunk

    async def _ensure_api_key(self, workspace_id: UUID) -> None:
        """Verify workspace has valid API key.

        Raises:
            ValueError: If no valid API key configured
        """
        key = await self._key_storage.get_key(workspace_id, "anthropic")
        if not key:
            raise ValueError("Workspace has no Anthropic API key configured")
```

"""SSE streaming utilities for AI endpoints.

Reference: DD-066 SSE for AI streaming
"""
import json
from collections.abc import AsyncIterator
from typing import Any

from fastapi import Request
from fastapi.responses import StreamingResponse


def create_sse_event(event: str, data: dict[str, Any]) -> str:
    """Format an SSE event.

    Args:
        event: Event type (e.g., "token", "done", "error")
        data: Event payload (will be JSON-encoded)

    Returns:
        Formatted SSE event string
    """
    json_data = json.dumps(data, ensure_ascii=False)
    return f"event: {event}\ndata: {json_data}\n\n"


async def sse_stream_generator(
    stream: AsyncIterator[str],
    request: Request,
) -> AsyncIterator[str]:
    """Generate SSE events from an async iterator.

    Args:
        stream: Async iterator yielding string chunks
        request: FastAPI request for disconnect detection

    Yields:
        Formatted SSE events
    """
    try:
        async for chunk in stream:
            # Check if client disconnected
            if await request.is_disconnected():
                break

            # Emit token event
            yield create_sse_event("token", {"content": chunk})

        # Emit done event
        yield create_sse_event("done", {"status": "complete"})

    except Exception as e:
        # Emit error event
        yield create_sse_event("error", {
            "message": str(e),
            "type": type(e).__name__,
        })


def create_sse_stream(
    stream: AsyncIterator[str],
    request: Request,
) -> StreamingResponse:
    """Create SSE streaming response.

    Usage:
        @router.post("/ai/notes/{note_id}/ghost-text")
        async def ghost_text(note_id: UUID, request: Request):
            stream = orchestrator.stream("ghost_text", input_data, context)
