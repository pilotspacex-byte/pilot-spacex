// T205: frontend/src/stores/cost-dashboard-store.ts
export class CostDashboardStore {
  isLoading = false;
  error: string | null = null;
  summary: CostSummary | null = null;
  dateRange: { start: string; end: string } | null = null;

  constructor(private rootStore: AIStore) {
    makeAutoObservable(this);
  }

  async loadSummary(): Promise<void> {
    runInAction(() => { this.isLoading = true; });

    try {
      const response = await aiApi.getCostSummary(
        this.dateRange?.start,
        this.dateRange?.end
      );
      runInAction(() => {
        this.summary = response.data;
        this.isLoading = false;
      });
    } catch (err) {
      runInAction(() => {
        this.error = err instanceof Error ? err.message : 'Failed to load';
        this.isLoading = false;
      });
    }
  }

  setDateRange(range: { start: string; end: string }): void {
    this.dateRange = range;
    this.loadSummary();
  }
}

// T214: frontend/src/features/ai/components/conversation-input.tsx
export function ConversationInput({ onSend, disabled }: ConversationInputProps) {
  const [input, setInput] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (input.trim() && !disabled) {
      onSend(input);
      setInput('');
    }
  };

  return (
    <form onSubmit={handleSubmit} className="p-4 border-t">
      <div className="flex gap-2">
        <Input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Type a message..."
          disabled={disabled}
        />
        <Button type="submit" disabled={disabled || !input.trim()}>
          <Send className="h-4 w-4" />
        </Button>
      </div>
    </form>
  );
}

// T216: frontend/src/stores/conversation-store.ts
export class ConversationStore {
  messages: Array<{ role: 'user' | 'assistant'; content: string }> = [];
  isStreaming = false;
  error: string | null = null;
  sessionId: string | null = null;

  private client: SSEClient | null = null;

  constructor(private rootStore: AIStore) {
    makeAutoObservable(this);
  }

  async sendMessage(content: string): Promise<void> {
    this.messages.push({ role: 'user', content });

    runInAction(() => {
      this.isStreaming = true;
      this.error = null;
    });

    this.client = new SSEClient({
      url: aiApi.getConversationUrl(),
      onMessage: (event) => {
        runInAction(() => {
          if (event.type === 'token') {
            const lastMessage = this.messages[this.messages.length - 1];
            if (lastMessage?.role === 'assistant') {
              lastMessage.content += (event.data as { content: string }).content;
            } else {
              this.messages.push({ role: 'assistant', content: (event.data as { content: string }).content });
            }
          }
          if (event.type === 'complete') {
            this.sessionId = (event.data as { session_id: string }).session_id;
          }
        });
      },
      onComplete: () => {
        runInAction(() => { this.isStreaming = false; });
      },
      onError: (err) => {
        runInAction(() => {
          this.isStreaming = false;
          this.error = err.message;
        });
      },
    });

    await this.client.connect();
  }

  abort(): void {
    this.client?.abort();
    this.isStreaming = false;
  }

  clear(): void {
    this.messages = [];
    this.sessionId = null;
  }
}
