```typescript
// frontend/tests/e2e/ai/ghost-text.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Ghost Text', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/notes/test-note');
    await page.waitForSelector('[data-testid="note-editor"]');
  });

  // T223: Ghost text suggestion appears
  test('should show ghost text suggestion after typing pause', async ({ page }) => {
    const editor = page.locator('[data-testid="note-editor"]');

    await editor.click();
    await editor.type('The authentication system should ');

    // Wait for debounce + API response
    await page.waitForTimeout(2500);

    // Ghost text should appear
    const ghostText = page.locator('.ghost-text-suggestion');
    await expect(ghostText).toBeVisible();
  });

  // T224: Tab accepts ghost text
  test('should accept ghost text on Tab press', async ({ page }) => {
    const editor = page.locator('[data-testid="note-editor"]');

    await editor.click();
    await editor.type('The authentication system should ');
    await page.waitForTimeout(2500);

    // Get ghost text content
    const ghostText = await page.locator('.ghost-text-suggestion').textContent();

    // Press Tab to accept
    await page.keyboard.press('Tab');

    // Ghost text should be inserted into editor
    const editorContent = await editor.textContent();
    expect(editorContent).toContain(ghostText);

    // Ghost text decoration should be gone
    await expect(page.locator('.ghost-text-suggestion')).not.toBeVisible();
  });

  // T225: Esc dismisses ghost text
  test('should dismiss ghost text on Escape press', async ({ page }) => {
    const editor = page.locator('[data-testid="note-editor"]');

    await editor.click();
    await editor.type('The authentication system should ');
    await page.waitForTimeout(2500);

    await expect(page.locator('.ghost-text-suggestion')).toBeVisible();

    // Press Escape to dismiss
    await page.keyboard.press('Escape');

    await expect(page.locator('.ghost-text-suggestion')).not.toBeVisible();
  });
});
```

```typescript
import { test, expect } from '@playwright/test';

test.describe('AI Context', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/issues/test-issue');
    await page.waitForSelector('[data-testid="issue-detail"]');
  });

  // T226: AI context panel opens
  test('should open AI context panel on button click', async ({ page }) => {
    await page.click('[data-testid="ai-context-button"]');

    await expect(page.locator('[data-testid="ai-context-panel"]')).toBeVisible();
    await expect(page.locator('text=Generate AI Context')).toBeVisible();
  });

  // T227: Streaming phases display
  test('should show streaming phases during generation', async ({ page }) => {
    await page.click('[data-testid="ai-context-button"]');
    await page.click('text=Generate AI Context');

    // Should show 5 phases
    const phases = page.locator('[data-testid="ai-context-phase"]');
    await expect(phases).toHaveCount(5);

    // First phase should be in progress
    await expect(phases.first()).toHaveAttribute('data-status', 'in_progress');

    // Wait for completion
    await page.waitForSelector('[data-testid="ai-context-result"]', { timeout: 60000 });

    // All phases should be complete
    const completedPhases = page.locator('[data-testid="ai-context-phase"][data-status="complete"]');
    await expect(completedPhases).toHaveCount(5);
  });

  // T228: Claude Code prompt copy
  test('should copy Claude Code prompt to clipboard', async ({ page, context }) => {
    await context.grantPermissions(['clipboard-read', 'clipboard-write']);

    await page.click('[data-testid="ai-context-button"]');
    await page.click('text=Generate AI Context');
    await page.waitForSelector('[data-testid="claude-code-prompt"]', { timeout: 60000 });

    await page.click('[data-testid="copy-prompt-button"]');

    // Check clipboard content
    const clipboardText = await page.evaluate(() => navigator.clipboard.readText());
    expect(clipboardText).toContain('Claude Code');

    // Should show "Copied" feedback
    await expect(page.locator('text=Copied')).toBeVisible();
  });
});
```
