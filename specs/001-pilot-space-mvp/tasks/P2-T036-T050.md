```python
"""Generic repository base class."""
from typing import Generic, TypeVar, Sequence
from uuid import UUID

from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy.orm import selectinload

from pilot_space.infrastructure.database.base import BaseModel

T = TypeVar("T", bound=BaseModel)


class BaseRepository(Generic[T]):
    """Base repository with CRUD operations."""

    model: type[T]

    def __init__(self, session: AsyncSession) -> None:
        self.session = session

    async def get(self, id: UUID, include_deleted: bool = False) -> T | None:
        """Get entity by ID."""
        query = select(self.model).where(self.model.id == id)
        if not include_deleted:
            query = query.where(self.model.is_deleted == False)
        result = await self.session.execute(query)
        return result.scalar_one_or_none()

    async def get_all(
        self,
        *,
        include_deleted: bool = False,
        options: list | None = None,
    ) -> Sequence[T]:
        """Get all entities."""
        query = select(self.model)
        if not include_deleted:
            query = query.where(self.model.is_deleted == False)
        if options:
            query = query.options(*options)
        result = await self.session.execute(query)
        return result.scalars().all()

    async def create(self, entity: T) -> T:
        """Create new entity."""
        self.session.add(entity)
        await self.session.flush()
        await self.session.refresh(entity)
        return entity

    async def update(self, entity: T) -> T:
        """Update existing entity."""
        await self.session.flush()
        await self.session.refresh(entity)
        return entity

    async def delete(self, entity: T) -> None:
        """Soft delete entity."""
        entity.is_deleted = True
        entity.deleted_at = func.now()
        await self.session.flush()

    async def hard_delete(self, entity: T) -> None:
        """Permanently delete entity."""
        await self.session.delete(entity)
        await self.session.flush()

    async def restore(self, entity: T) -> T:
        """Restore soft-deleted entity (must pass permission check first)."""
        entity.is_deleted = False
        entity.deleted_at = None
        await self.session.flush()
        await self.session.refresh(entity)
        return entity

    async def get_deleted(
        self,
        workspace_id: UUID | None = None,
    ) -> Sequence[T]:
        """Get soft-deleted entities for trash view."""
        query = select(self.model).where(self.model.is_deleted == True)
        if workspace_id and hasattr(self.model, 'workspace_id'):
            query = query.where(self.model.workspace_id == workspace_id)
        result = await self.session.execute(query)
        return result.scalars().all()

    async def get_page(
        self,
        cursor: UUID | None,
        limit: int = 20,
        filters: dict | None = None,
    ) -> tuple[Sequence[T], UUID | None]:
        """Get paginated results with cursor."""
        query = select(self.model).where(self.model.is_deleted == False)

        if cursor:
            query = query.where(self.model.id > cursor)

        if filters:
            for key, value in filters.items():
                query = query.where(getattr(self.model, key) == value)

        query = query.order_by(self.model.id).limit(limit + 1)
        result = await self.session.execute(query)
        items = list(result.scalars().all())

        next_cursor = items[-1].id if len(items) > limit else None
        return items[:limit], next_cursor
```

```python
from enum import Enum
from sqlalchemy import ForeignKey, String, Boolean, Enum as SQLEnum
from sqlalchemy.dialects.postgresql import JSONB
from sqlalchemy.orm import Mapped, mapped_column
from cryptography.fernet import Fernet

from .base import BaseModel, WorkspaceScopedMixin

class AIProvider(str, Enum):
    ANTHROPIC = "anthropic"
    OPENAI = "openai"
    GEMINI = "gemini"

class AIConfiguration(BaseModel, WorkspaceScopedMixin):
    __tablename__ = "ai_configurations"

    provider: Mapped[AIProvider] = mapped_column(SQLEnum(AIProvider), nullable=False)
    api_key_encrypted: Mapped[str] = mapped_column(String(512), nullable=False)
    model_id: Mapped[str | None] = mapped_column(String(100), nullable=True)
    is_active: Mapped[bool] = mapped_column(Boolean, default=True)
    settings: Mapped[dict] = mapped_column(JSONB, default=dict)

    __table_args__ = (
        UniqueConstraint("workspace_id", "provider", name="uq_ai_config_workspace_provider"),
    )

    def decrypt_api_key(self, fernet: Fernet) -> str:
        return fernet.decrypt(self.api_key_encrypted.encode()).decode()
```

### T048a: Create rate limiting middleware

**Complexity**: 🟡 7/20 | **Priority**: P1 | **Story**: INFRA
**Blocked by**: T065 (Redis)

#### Objective
Create rate limiting middleware implementing per-workspace tracking per NFR-019.

#### Acceptance Criteria

- [ ] AC2: Standard rate limit: 1000 requests/minute per workspace
- [ ] AC3: AI endpoint rate limit: 100 requests/minute per workspace

- [ ] AC5: Returns `429 Too Many Requests` with `Retry-After` header

- [ ] AC8: Structured logging for rate limit events

#### Target Files

#### Code Structure

```python
from datetime import datetime, timedelta
from fastapi import Request, HTTPException
from starlette.middleware.base import BaseHTTPMiddleware
import redis.asyncio as redis

class RateLimitMiddleware(BaseHTTPMiddleware):
    def __init__(
        self,
        app,
        redis_client: redis.Redis,
        standard_limit: int = 1000,
        ai_limit: int = 100,
        window_seconds: int = 60,
    ):
        super().__init__(app)
        self.redis = redis_client
        self.standard_limit = standard_limit
        self.ai_limit = ai_limit
        self.window = window_seconds

    async def dispatch(self, request: Request, call_next):
        # Skip health checks
        if request.url.path in ["/health", "/ready"]:
            return await call_next(request)

        workspace_id = request.headers.get("X-Workspace-ID") or "anonymous"
        endpoint_type = "ai" if "/ai/" in request.url.path else "standard"
        limit = self.ai_limit if endpoint_type == "ai" else self.standard_limit

        key = f"ratelimit:{workspace_id}:{endpoint_type}"
        current = await self.redis.incr(key)
        if current == 1:
            await self.redis.expire(key, self.window)

        if current > limit:
            ttl = await self.redis.ttl(key)
            raise HTTPException(
                status_code=429,
                detail="Rate limit exceeded",
                headers={"Retry-After": str(ttl)}
            )

        return await call_next(request)
```

*Note: NFR-019 rate limiting - 1000 req/min standard, 100 req/min AI endpoints.*

```python
"""Dependency injection container."""
from dependency_injector import containers, providers

from pilot_space.config import Settings, get_settings
from pilot_space.infrastructure.database.engine import async_session_factory
from pilot_space.infrastructure.database.repositories.user_repository import UserRepository
from pilot_space.infrastructure.database.repositories.workspace_repository import WorkspaceRepository
from pilot_space.infrastructure.database.repositories.project_repository import ProjectRepository
from pilot_space.infrastructure.auth.supabase_auth import SupabaseAuthClient


class Container(containers.DeclarativeContainer):
    """Application container."""

    wiring_config = containers.WiringConfiguration(
        packages=["pilot_space.api.v1.routers"],
    )

    # Configuration
    config = providers.Singleton(get_settings)

    # Database session
    db_session = providers.Resource(
        async_session_factory,
    )

    # Auth client
    auth_client = providers.Singleton(
        SupabaseAuthClient,
        settings=config,
    )

    # Repositories
    user_repository = providers.Factory(
        UserRepository,
        session=db_session,
    )

    workspace_repository = providers.Factory(
        WorkspaceRepository,
        session=db_session,
    )

    project_repository = providers.Factory(
        ProjectRepository,
        session=db_session,
    )
```
