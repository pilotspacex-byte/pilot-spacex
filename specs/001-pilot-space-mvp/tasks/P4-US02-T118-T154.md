### T129a [US2]: Create ApprovalFlowService

**Complexity**: 🟡 7/20 | **Priority**: P1 | **Story**: US2
**Blocked by**: T129

#### Objective
Create ApprovalFlowService for human-in-the-loop AI action approvals per DD-003.

#### Acceptance Criteria
- [ ] AC1: `backend/src/pilot_space/application/services/approval_flow_service.py` exists
- [ ] AC2: `PendingApproval` entity stores pending AI actions
- [ ] AC3: `request_approval(action_type, payload, user_id)` queues for approval
- [ ] AC4: `approve(approval_id)` executes the queued action
- [ ] AC5: `reject(approval_id, reason)` cancels with feedback
- [ ] AC6: Auto-approve non-destructive actions per DD-003

#### Guidelines
```python
# 1. Approval action types
class ActionType(str, Enum):
    CREATE_ISSUE = "create_issue"        # Auto-approve
    UPDATE_ISSUE = "update_issue"        # Auto-approve
    DELETE_ISSUE = "delete_issue"        # Requires approval
    BULK_UPDATE = "bulk_update"          # Requires approval
    MERGE_DUPLICATE = "merge_duplicate"  # Requires approval
    SEND_NOTIFICATION = "send_notification"  # Auto-approve

# 2. Service structure
class ApprovalFlowService:
    CRITICAL_ACTIONS = {ActionType.DELETE_ISSUE, ActionType.BULK_UPDATE, ActionType.MERGE_DUPLICATE}

    async def request_approval(
        self, action_type: ActionType, payload: dict, user_id: UUID
    ) -> UUID | None:
        """Returns approval_id if requires approval, None if auto-approved."""
        if action_type not in self.CRITICAL_ACTIONS:
            await self._execute_action(action_type, payload)
            return None
        return await self._create_pending_approval(action_type, payload, user_id)

    async def approve(self, approval_id: UUID, approver_id: UUID) -> Any:
        approval = await self.repository.get(approval_id)
        result = await self._execute_action(approval.action_type, approval.payload)
        await self._mark_approved(approval_id, approver_id)
        return result
```

# 1. Decorator implementation
def log_activity(action: ActivityAction, entity_type: str):
    def decorator(func: Callable[P, Awaitable[T]]) -> Callable[P, Awaitable[T]]:
        @functools.wraps(func)
        async def wrapper(*args: P.args, **kwargs: P.kwargs) -> T:
            start = time.monotonic()
            result = await func(*args, **kwargs)
            duration = time.monotonic() - start

            # Fire-and-forget activity logging
            asyncio.create_task(
                record_activity(
                    action=action,
                    entity_type=entity_type,
                    entity_id=_extract_entity_id(result),
                    actor_id=_get_current_user_id(),
                    metadata={"duration_ms": int(duration * 1000)},
                )
            )
            return result
        return wrapper
    return decorator

# 2. Usage
class IssueService:
    @log_activity(ActivityAction.CREATE, "issue")
    async def create(self, payload: CreateIssuePayload) -> Issue:
        ...

@router.post("/{issue_id}/restore", response_model=IssueResponse)
async def restore_issue(
    issue_id: UUID,
    service: Annotated[RestoreIssueService, Depends(get_restore_issue_service)],
    current_user: Annotated[User, Depends(get_current_user)],
) -> IssueResponse:

# Service
class RestoreIssueService:

        return await self.repository.save(issue)

async def list_trash(
    project_id: UUID | None = None,
    pagination: Annotated[PaginationParams, Depends()],
    repository: Annotated[IssueRepository, Depends(get_issue_repository)],
) -> PaginatedResponse[IssueResponse]:

# Repository method
async def list_deleted(self, project_id: UUID | None, pagination: PaginationParams):
    query = select(Issue).where(Issue.deleted_at.isnot(None))
    if project_id:
        query = query.where(Issue.project_id == project_id)
    query = query.order_by(Issue.deleted_at.desc())
    return await self.paginate(query, pagination)
