```typescript
import { makeAutoObservable } from 'mobx';
import { AuthStore } from './AuthStore';
import { WorkspaceStore } from './WorkspaceStore';
import { UIStore } from './UIStore';

export class RootStore {
  authStore: AuthStore;
  workspaceStore: WorkspaceStore;
  uiStore: UIStore;

  constructor() {
    makeAutoObservable(this);
    this.authStore = new AuthStore(this);
    this.workspaceStore = new WorkspaceStore(this);
    this.uiStore = new UIStore(this);
  }

  reset(): void {
    this.authStore.reset();
    this.workspaceStore.reset();
    this.uiStore.reset();
  }

  hydrate(data: HydrationData): void {
    if (data.user) this.authStore.setUser(data.user);
    if (data.workspace) this.workspaceStore.setCurrentWorkspace(data.workspace);
  }
}
```

**Complexity**: 🟡 6/20 | **Priority**: P1 | **Story**: INFRA

- [ ] AC6: Structured logging for retry attempts
- [ ] AC7: Metrics: `edge_function_timeout_total`, `edge_function_retry_total`
- [ ] AC8: Integration with QueueClient for automatic retry handling

#### Target Files
- `backend/src/pilot_space/infrastructure/queue/error_handlers.py`

#### Code Structure

```python
import asyncio
from datetime import datetime, timedelta
from typing import Callable, TypeVar
import structlog

logger = structlog.get_logger(__name__)

T = TypeVar("T")

class EdgeFunctionTimeoutHandler:
    def __init__(
        self,
        max_retries: int = 3,
        base_delay: float = 1.0,
        timeout: float = 30.0,
    ):
        self.max_retries = max_retries
        self.base_delay = base_delay
        self.timeout = timeout

    async def execute_with_retry(
        self,
        func: Callable[..., T],
        *args,
        **kwargs
    ) -> T:
        last_error = None
        for attempt in range(self.max_retries + 1):
            try:
                return await asyncio.wait_for(
                    func(*args, **kwargs),
                    timeout=self.timeout
                )
            except asyncio.TimeoutError as e:
                last_error = e
                delay = self.base_delay * (2 ** attempt)
                logger.warning(
                    "edge_function_timeout",
                    attempt=attempt + 1,
                    max_retries=self.max_retries,
                    delay=delay
                )
                if attempt < self.max_retries:
                    await asyncio.sleep(delay)

        logger.error("edge_function_failed", error=str(last_error))
        raise last_error
```

*Note: Handles Edge Function timeouts per spec.md edge case requirements.*
