#### Guidelines
```python
# 1. Job registration
from pilot_space.infrastructure.queue import queue_client

@queue_client.job(
    "pr_review",
    visibility_timeout=600,  # 10 minutes
    max_retries=3,
    dead_letter_queue="pr_review_dlq",
)
async def handle_pr_review(payload: PRReviewJobPayload) -> None:
    handler = PRReviewJobHandler()
    await handler.execute(payload)

# 2. Payload schema
class PRReviewJobPayload(BaseModel):
    integration_id: UUID
    pr_number: int
    workspace_id: UUID
    triggered_by: UUID
    review_scope: list[str] = ["architecture", "code", "security", "performance"]

# 3. Enqueue from service
async def trigger_review(self, payload: TriggerPRReviewPayload) -> UUID:
    job_id = await queue_client.enqueue(
        "pr_review",
        PRReviewJobPayload(
            integration_id=payload.integration_id,
            pr_number=payload.pr_number,
            workspace_id=payload.workspace_id,
            triggered_by=payload.user_id,
        ),
    )
    await self._create_review_record(job_id, payload)
    return job_id
```
