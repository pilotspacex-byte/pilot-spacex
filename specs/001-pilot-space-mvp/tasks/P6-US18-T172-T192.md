### Webhook Security

```python
# HMAC-SHA256 signature validation
import hmac
import hashlib

def verify_github_signature(payload: bytes, signature: str, secret: str) -> bool:
    expected = "sha256=" + hmac.new(
        secret.encode(), payload, hashlib.sha256
    ).hexdigest()
    return hmac.compare_digest(expected, signature)
```

```python
# 1. Job registration with Supabase Queues
from pilot_space.infrastructure.queue import queue_client

@queue_client.job("scan_commits")
async def scan_commits_job(payload: ScanCommitsPayload) -> None:
    scanner = CommitScannerService()
    await scanner.execute(payload)

# 2. Scanner service
class CommitScannerService:
    ISSUE_PATTERNS = [
        r"(?:fixes|closes|resolves)\s+#(\d+)",
        r"(?:fixes|closes|resolves)\s+([A-Z]+-\d+)",
    ]

    async def execute(self, payload: ScanCommitsPayload) -> None:
        integration = await self.repo.get(payload.integration_id)
        commits = await self.github.list_commits(
            integration.config["owner"],
            integration.config["repo"],
            since=integration.last_scan_at,
        )

        for commit in commits:
            refs = self._extract_issue_refs(commit.message)
            for ref in refs:
                issue = await self._resolve_issue(ref, integration.workspace_id)
                if issue:
                    await self._create_link(issue.id, commit)
                    if self._is_closing_ref(commit.message, ref):
                        await self.auto_transition.execute(issue.id, "done")

        await self._update_scan_timestamp(integration.id)

# 3. Schedule registration
queue_client.schedule(
    "scan_commits",
    cron="*/15 * * * *",
    payload={"scope": "all_integrations"},
)
```

#### Guidelines
```python
@router.get("/{issue_id}/branch-name", response_model=BranchNameResponse)
async def get_branch_name(
    issue_id: UUID,
    repository: Annotated[IssueRepository, Depends(get_issue_repository)],
) -> BranchNameResponse:
    issue = await repository.get(issue_id)
    branch_type = _determine_branch_type(issue.labels)
    slug = _slugify(issue.title, max_length=50)
    branch_name = f"{branch_type}/{issue.identifier}-{slug}"
    return BranchNameResponse(
        branch_name=branch_name,
        git_command=f"git checkout -b {branch_name}",
    )

def _determine_branch_type(labels: list[str]) -> str:
    if any(l in ["bug", "bugfix"] for l in labels):
        return "bugfix"
    if any(l in ["urgent", "hotfix"] for l in labels):
        return "hotfix"
    if any(l in ["chore", "maintenance"] for l in labels):
        return "chore"
    return "feature"

def _slugify(text: str, max_length: int = 50) -> str:
    slug = re.sub(r"[^a-zA-Z0-9]+", "-", text.lower())
    return slug[:max_length].strip("-")
```

function BranchSuggestion({ issueId }: Props) {
  const { data, isLoading } = useBranchName(issueId);
  const [copied, setCopied] = useState<'name' | 'command' | null>(null);

  const copyToClipboard = async (text: string, type: 'name' | 'command') => {
    await navigator.clipboard.writeText(text);
    setCopied(type);
    setTimeout(() => setCopied(null), 2000);
  };

  if (isLoading) return <Skeleton className="h-16 w-full" />;

  return (
    <div className="space-y-2">
      <div className="flex items-center gap-2">
        <code className="flex-1 bg-muted p-2 rounded">{data.branchName}</code>
        <Button
          variant="ghost"
          size="icon"
          onClick={() => copyToClipboard(data.branchName, 'name')}
        >
          {copied === 'name' ? <Check /> : <Copy />}
        </Button>
      </div>
      <div className="flex items-center gap-2">
        <code className="flex-1 bg-muted p-2 rounded text-sm">{data.gitCommand}</code>
        <Button
          variant="ghost"
          size="icon"
          onClick={() => copyToClipboard(data.gitCommand, 'command')}
        >
          {copied === 'command' ? <Check /> : <Copy />}
        </Button>
      </div>
    </div>
  );
}
