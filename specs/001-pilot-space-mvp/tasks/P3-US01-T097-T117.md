  notes: Map<string, Note> = new Map();
  annotations: Map<string, NoteAnnotation[]> = new Map();
  isLoading = false;
  isSaving = false;
  lastSavedAt: Date | null = null;

  constructor(private rootStore: RootStore) {
    makeAutoObservable(this, {}, { autoBind: true });
  }

  get hasUnsavedChanges(): boolean {
    // Compare current content with last saved
  }

  get pendingAnnotations(): NoteAnnotation[] {
    const noteId = this.currentNote?.id;
    if (!noteId) return [];
    return (this.annotations.get(noteId) ?? [])
      .filter(a => a.status === 'pending');
  }

  async loadNote(noteId: string): Promise<void> {
    this.isLoading = true;
    try {
      const note = await noteService.getNote(noteId);
      runInAction(() => {
        this.currentNote = note;
        this.notes.set(noteId, note);
      });
    } finally {
      runInAction(() => { this.isLoading = false; });
    }
  }

}

class NoteService:

        if note.is_pinned:

```typescript
export interface GhostTextRequest {
  text: string;
  cursorPosition: number;
  context: string;
}

export interface GhostTextEvent {
  type: 'text' | 'done' | 'error';
  data: string;
}

export class GhostTextService {
  private eventSource: EventSource | null = null;
  private abortController: AbortController | null = null;

  async requestCompletion(
    request: GhostTextRequest,
    onChunk: (text: string) => void,
    onComplete: () => void,
    onError: (error: Error) => void
  ): Promise<void> {
    // Cancel any existing request
    this.cancel();

    this.abortController = new AbortController();

    const url = new URL('/api/v1/ai/ghost-text', window.location.origin);

    this.eventSource = new EventSource(url.toString());

    this.eventSource.onmessage = (event) => {
      const data: GhostTextEvent = JSON.parse(event.data);
      if (data.type === 'text') {
        onChunk(data.data);
      } else if (data.type === 'done') {
        onComplete();
        this.cleanup();
      }
    };

    this.eventSource.onerror = () => {
      onError(new Error('SSE connection failed'));
      this.cleanup();
    };
  }

  cancel(): void {
    this.abortController?.abort();
    this.cleanup();
  }

  private cleanup(): void {
    this.eventSource?.close();
    this.eventSource = null;
    this.abortController = null;
  }
}
```
