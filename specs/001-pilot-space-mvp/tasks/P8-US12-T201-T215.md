"""AI Context Agent for issue context aggregation."""
from dataclasses import dataclass, field

from anthropic import Anthropic

from pilot_space.ai.agents.base import BaseAgent, AgentContext, TaskType


@dataclass
class RelatedItem:
    """Reference to related content."""
    id: str
    type: str  # issue, note, page
    title: str
    relevance_score: float
    excerpt: str


@dataclass
class CodeReference:
    """Reference to code."""
    file_path: str
    line_range: tuple[int, int]
    description: str
    relevance: str


@dataclass
class TaskItem:
    """Task in checklist."""
    description: str
    completed: bool
    dependencies: list[str]
    estimated_effort: str  # S, M, L


@dataclass
class AIContextInput:
    """Input for context generation."""
    issue_id: str
    issue_title: str
    issue_description: str
    workspace_id: str
    conversation_history: list[dict] = field(default_factory=list)
    refinement_query: str | None = None


@dataclass
class AIContextOutput:
    """Generated AI context."""
    summary: str
    related_issues: list[RelatedItem]

    related_pages: list[RelatedItem]
