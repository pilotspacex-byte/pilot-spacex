Suggested Error Handler

async function handleOAuthError(error: OAuthError) {
  const message = getHumanReadableError(error.code);

  // Show user-friendly toast with retry
  toast.error(message, {
    action: {
      label: 'Try Again',
      onClick: () => retryOAuth(error.provider)
    }
  });

  // Log for debugging
  logger.error('OAuth failed', { error, provider });
}
