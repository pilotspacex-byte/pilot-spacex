    ```python
    """Model override resolution for per-session model selection (AIPR-04)."""
    from __future__ import annotations
    from dataclasses import dataclass
    from uuid import UUID
    from pydantic import BaseModel, Field
    from pilot_space.api.v1.schemas.base import BaseSchema
    from pilot_space.infrastructure.logging import get_logger

    logger = get_logger(__name__)


    class ModelOverride(BaseSchema):
        """User-selected model for a chat session."""
        provider: str = Field(..., description="Provider name, e.g. 'anthropic', 'kimi', 'custom'")
        model: str = Field(..., description="Model ID, e.g. 'claude-sonnet-4'")
        config_id: str = Field(..., description="AIConfiguration.id — used to look up api_key + base_url")


    @dataclass
    class ResolvedModelConfig:
        """Resolved provider credentials for a user-selected model."""
        api_key: str
        model: str
        provider: str
        base_url: str | None = None


    async def resolve_model_override(
        model_override: ModelOverride,
        workspace_id: UUID,
        db,  # AsyncSession — avoid import cycle
    ) -> ResolvedModelConfig | None:
        """Resolve a ModelOverride to decrypted credentials.

        Returns None if config not found (agent falls back to workspace default).
        Never raises — failure is logged and treated as fallback.
        """
        from uuid import UUID as _UUID
        from pilot_space.infrastructure.database.repositories.ai_configuration_repository import AIConfigurationRepository
        from pilot_space.infrastructure.encryption import decrypt_api_key, EncryptionError

        try:
            config_id = _UUID(model_override.config_id)
        except ValueError:
            logger.warning("model_override_invalid_config_id", config_id=model_override.config_id)
            return None

        try:
            repo = AIConfigurationRepository(session=db)
            config = await repo.get_by_workspace_and_id(workspace_id, config_id)
            if not config or not config.is_active:
                logger.warning("model_override_config_not_found", config_id=str(config_id))
                return None

            api_key = decrypt_api_key(config.api_key_encrypted)
            return ResolvedModelConfig(
                api_key=api_key,
                model=model_override.model,
                provider=model_override.provider,
                base_url=getattr(config, "base_url", None),
            )
        except (EncryptionError, Exception) as e:
            logger.warning("model_override_resolution_failed", error=str(e))
            return None


    __all__ = ["ModelOverride", "ResolvedModelConfig", "resolve_model_override"]
    ```
