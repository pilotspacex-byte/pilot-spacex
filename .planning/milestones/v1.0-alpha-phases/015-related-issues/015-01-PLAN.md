```python
"""IssueSuggestionDismissalRepository — CRUD for per-user suggestion dismissals."""
from uuid import UUID
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from pilot_space.infrastructure.database.models.issue_suggestion_dismissal import IssueSuggestionDismissal

class IssueSuggestionDismissalRepository:
    def __init__(self, session: AsyncSession) -> None:
        self._session = session

    async def get_dismissed_target_ids(self, user_id: UUID, source_issue_id: UUID) -> set[UUID]:
        """Return set of target_issue_ids dismissed by this user for the given source issue."""
        result = await self._session.execute(
            select(IssueSuggestionDismissal.target_issue_id)
            .where(
                IssueSuggestionDismissal.user_id == user_id,
                IssueSuggestionDismissal.source_issue_id == source_issue_id,
            )
        )
        return {row[0] for row in result.all()}

    async def create_dismissal(
        self, workspace_id: UUID, user_id: UUID, source_issue_id: UUID, target_issue_id: UUID
    ) -> IssueSuggestionDismissal:
        """Persist a dismissal. UNIQUE constraint is the idempotency guard."""
        dismissal = IssueSuggestionDismissal(
            workspace_id=workspace_id,
            user_id=user_id,
            source_issue_id=source_issue_id,
            target_issue_id=target_issue_id,
        )
        self._session.add(dismissal)
        await self._session.flush()
        return dismissal
```
