    Add to `frontend/src/lib/tree-utils.ts`:
    ```typescript
    /**
     * Flattens a PageTreeNode[] tree back into a flat array.
     * Needed because useProjectPageTree returns post-select PageTreeNode[] (tree structure),
     * but getAncestors requires a flat array to traverse parentId chains.
     */
    export function flattenTree(nodes: PageTreeNode[]): Array<{ id: string; title: string; parentId: string | null }> {
      const result: Array<{ id: string; title: string; parentId: string | null }> = [];
      function walk(nodeList: PageTreeNode[]) {
        for (const node of nodeList) {
          result.push({ id: node.id, title: node.title, parentId: node.parentId });
          if (node.children.length > 0) {
            walk(node.children);
          }
        }
      }
      walk(nodes);
      return result;
    }
    ```

       ```typescript
       // Fetch project tree data — reuses the same TanStack Query cache as sidebar
       // (same query key = no duplicate fetch). Returns PageTreeNode[] (post-select).
       const { data: treeData } = useProjectPageTree(
         workspaceId,
         note?.projectId ?? '',
         !!note?.projectId
       );

       // Derive breadcrumb ancestors by flattening tree then walking parentId chain
       const ancestors = useMemo(() => {
         if (!treeData || !note?.id) return [];
         const flatNotes = flattenTree(treeData);
         return getAncestors(note.id, flatNotes);
       }, [treeData, note?.id]);

       // Get project name for breadcrumb root
       const projectName = useMemo(() => {
         if (!note?.projectId) return undefined;
         const project = workspaceStore.currentWorkspace?.projects?.find(
           (p: { id: string }) => p.id === note.projectId
         );
         return project?.name;
       }, [note?.projectId, workspaceStore.currentWorkspace]);
       ```

       IMPORTANT: This approach uses `useProjectPageTree` which shares the same query key
       as the sidebar. TanStack Query deduplicates — no extra network request. The hook
       returns `PageTreeNode[]` (post-select), so we flatten with `flattenTree` before
       passing to `getAncestors`. This avoids the silent failure bug where
       `queryClient.getQueryData(...)?.items` would be undefined because `select`
       transforms the cached data.

    3. Add content sanitization function (outside the component, pure function):
       ```typescript
       /**
        * Strips data-property-block divs from TipTap content to prevent
        * unknown node errors when non-issue pages are opened in NoteCanvas.
        * See RESEARCH.md Pattern 5 and STATE.md concern about editor coupling.
        */
       function sanitizeNoteContent(content: JSONContent | undefined): JSONContent | undefined {
         if (!content?.content) return content;
         return {
           ...content,
           content: content.content.filter(
             (node: Record<string, unknown>) => {
               const attrs = node.attrs as Record<string, unknown> | undefined;
               return !(node.type === 'propertyBlock' || attrs?.['data-property-block']);
             }
           ),
         };
       }
       ```
