From frontend/src/stores/ai/AISettingsStore.ts (MobX store pattern):
```typescript
import { makeAutoObservable, runInAction } from 'mobx';

export class AISettingsStore {
  items: Item[] = [];
  isLoading = false;
  error: string | null = null;

  constructor() { makeAutoObservable(this); }

  async loadItems(workspaceId: string): Promise<void> {
    runInAction(() => { this.isLoading = true; this.error = null; });
    try {
      const data = await apiClient.get<ListResponse>('/endpoint', { params: { workspace_id: workspaceId } });
      runInAction(() => { this.items = data.items; });
    } catch (e) {
      runInAction(() => { this.error = String(e); });
    } finally {
      runInAction(() => { this.isLoading = false; });
    }
  }
}
```

Backend API contracts (from Plan 03):
```
POST   /api/v1/workspaces/{workspace_id}/mcp-servers
       body: { display_name, url, auth_type, auth_token?, oauth_client_id?, oauth_auth_url?, oauth_token_url?, oauth_scopes? }
       → 201 WorkspaceMcpServerResponse

GET    /api/v1/workspaces/{workspace_id}/mcp-servers
       → 200 { items: WorkspaceMcpServerResponse[], total: int }

GET    /api/v1/workspaces/{workspace_id}/mcp-servers/{server_id}/status
       → 200 { server_id, status: "connected"|"failed"|"unknown", checked_at }

DELETE /api/v1/workspaces/{workspace_id}/mcp-servers/{server_id}
       → 204

GET    /api/v1/workspaces/{workspace_id}/mcp-servers/{server_id}/oauth-url
       → 200 { auth_url: string }
```
