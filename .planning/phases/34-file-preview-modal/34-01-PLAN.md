    ```typescript
    export type RendererType = 'image' | 'markdown' | 'text' | 'json' | 'code' | 'csv' | 'download';

    // Extension to lowlight language name mapping
    // Mirrors SUPPORTED_LANGUAGES in CodeBlockExtension.ts
    const EXT_TO_LANG: Record<string, string> = {
      py: 'python', js: 'javascript', ts: 'typescript',
      tsx: 'typescript', jsx: 'javascript', rb: 'ruby',
      go: 'go', rs: 'rust', java: 'java', cs: 'csharp',
      php: 'php', swift: 'swift', kt: 'kotlin',
      html: 'html', css: 'css', scss: 'scss',
      json: 'json', yaml: 'yaml', yml: 'yaml',
      md: 'markdown', sh: 'bash', bash: 'bash',
      sql: 'sql', graphql: 'graphql', gql: 'graphql',
      c: 'c', cpp: 'cpp', h: 'c', hpp: 'cpp',
      dockerfile: 'dockerfile',
    };

    const CODE_EXTENSIONS = new Set([
      'js', 'ts', 'jsx', 'tsx', 'py', 'rb', 'go', 'rs', 'java',
      'c', 'cpp', 'h', 'hpp', 'cs', 'php', 'swift', 'kt',
      'sh', 'bash', 'yaml', 'yml', 'toml', 'sql', 'graphql', 'gql',
      'dockerfile', 'html', 'css', 'scss',
    ]);

    function getExtension(filename: string): string {
      const parts = filename.split('.');
      if (parts.length < 2) return '';
      return parts[parts.length - 1].toLowerCase();
    }

    export function getLanguageForFile(filename: string): string {
      const ext = getExtension(filename);
      return EXT_TO_LANG[ext] ?? 'plaintext';
    }

    function resolveCodeOrText(filename: string): RendererType {
      const ext = getExtension(filename);
      return CODE_EXTENSIONS.has(ext) ? 'code' : 'text';
    }

    export function resolveRenderer(mimeType: string, filename: string): RendererType {
      const lowerMime = mimeType.toLowerCase();
      const ext = getExtension(filename);

      // Image types
      if (lowerMime.startsWith('image/')) return 'image';

      // CSV — check both MIME and extension
      if (lowerMime === 'text/csv' || ext === 'csv') return 'csv';

      // Markdown
      if (lowerMime === 'text/markdown' || ext === 'md') return 'markdown';

      // JSON
      if (lowerMime === 'application/json' || ext === 'json') return 'json';

      // HTML — CRITICAL: route to code renderer (escaped source), never live HTML
      if (lowerMime === 'text/html' || ext === 'html' || ext === 'htm') return 'code';

      // text/* types — resolve to code or plain text by extension
      if (lowerMime.startsWith('text/')) return resolveCodeOrText(filename);

      // Everything else — download fallback
      return 'download';
    }
    ```

    Step 1 — Create `frontend/src/features/artifacts/hooks/useFileContent.ts`:

    ```typescript
    'use client';

    /**
     * useFileContent — Fetches file content from a Supabase Storage signed URL.
     *
     * IMPORTANT: Uses window.fetch(), NOT apiClient. Signed URLs are external Supabase
     * Storage URLs that include the auth token in the URL itself. apiClient injects an
     * Authorization: Bearer header that Supabase Storage rejects with 403.
     *
     * staleTime: 55 minutes max. Supabase signed URLs expire at 60 minutes.
     * retry: false. A 403 means the URL has expired — retrying won't help.
     */

    import { useQuery } from '@tanstack/react-query';
    import type { RendererType } from '../utils/mime-type-router';

    export const artifactKeys = {
      all: ['artifacts'] as const,
      signedUrl: (artifactId: string) => ['artifacts', 'signed-url', artifactId] as const,
      fileContent: (signedUrl: string) => ['file-content', signedUrl] as const,
    };

    export interface UseFileContentResult {
      content: string | undefined;
      isLoading: boolean;
      isError: boolean;
      isExpired: boolean;
    }

    /**
     * Fetches text content from a signed URL when the modal is open.
     * Skips fetch for image renderer — images use the URL directly in <img src>.
     * Skips fetch when modal is closed (enabled: false).
     */
    export function useFileContent(
      signedUrl: string,
      rendererType: RendererType,
      open: boolean
    ): UseFileContentResult {
      const shouldFetch = open && !!signedUrl && rendererType !== 'image' && rendererType !== 'download';

      const { data, error, isLoading, isError } = useQuery({
        queryKey: artifactKeys.fileContent(signedUrl),
        queryFn: async () => {
          // Use window.fetch directly — DO NOT use apiClient
          const res = await fetch(signedUrl);
          if (res.status === 403) throw new Error('URL_EXPIRED');
          if (!res.ok) throw new Error(`Fetch failed: ${res.status}`);
          return res.text();
        },
        enabled: shouldFetch,
        staleTime: 1000 * 55 * 60, // 55 minutes — URLs expire at 60 min
        gcTime: 1000 * 60 * 60,    // 1 hour
        retry: false,               // Do NOT retry 403 — show "Link expired" UI immediately
      });

      const isExpired = isError && (error as Error)?.message === 'URL_EXPIRED';

      return {
        content: data,
        isLoading: shouldFetch ? isLoading : false,
        isError,
        isExpired,
      };
    }
    ```
