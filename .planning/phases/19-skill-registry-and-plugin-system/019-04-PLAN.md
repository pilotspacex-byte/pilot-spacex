**plugins.ts** — thin API client following the apiClient pattern from services/api/:
```typescript
import { apiClient } from './client'
import type { InstalledPlugin, AvailablePlugin } from '../../stores/ai/PluginsStore'

export const pluginsApi = {
  getInstalled: (workspaceId: string) =>
    apiClient.get<InstalledPlugin[]>(`/workspaces/${workspaceId}/plugins`),
  browse: (workspaceId: string, repoUrl: string) =>
    apiClient.get<AvailablePlugin[]>(`/workspaces/${workspaceId}/plugins/browse`, { params: { repo_url: repoUrl } }),
  install: (workspaceId: string, payload: { repo_url: string; skill_name: string }) =>
    apiClient.post<InstalledPlugin>(`/workspaces/${workspaceId}/plugins`, payload),
  uninstall: (workspaceId: string, pluginId: string) =>
    apiClient.delete(`/workspaces/${workspaceId}/plugins/${pluginId}`),
  checkUpdates: (workspaceId: string) =>
    apiClient.get<{ plugins: InstalledPlugin[] }>(`/workspaces/${workspaceId}/plugins/check-updates`),
  saveGitHubPat: (workspaceId: string, pat: string) =>
    apiClient.post(`/workspaces/${workspaceId}/plugins/github-credential`, { pat }),
  getGitHubCredential: (workspaceId: string) =>
    apiClient.get<{ has_pat: boolean }>(`/workspaces/${workspaceId}/plugins/github-credential`),
}
```
