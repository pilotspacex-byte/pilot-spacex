Step 1 — Create `frontend/src/features/approvals/hooks/use-approvals.ts`:
```typescript
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { api } from "@/lib/api";  // use existing api client

export function useApprovals(workspaceSlug: string, status = "PENDING") {
  return useQuery({
    queryKey: ["approvals", workspaceSlug, status],
    queryFn: () => api.get(`/workspaces/${workspaceSlug}/approvals?status=${status}`).then(r => r.data),
  });
}

export function usePendingApprovalCount(workspaceSlug: string) {
  const { data } = useApprovals(workspaceSlug, "PENDING");
  return (data?.items?.length ?? 0) as number;
}

export function useResolveApproval(workspaceSlug: string) {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ id, action, reason }: { id: string; action: "approve" | "reject"; reason?: string }) =>
      api.post(`/workspaces/${workspaceSlug}/approvals/${id}/resolve`, { action, reason }),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["approvals", workspaceSlug] }),
  });
}
```

```typescript
function useAIPolicy(workspaceSlug: string) {
  return useQuery({
    queryKey: ["ai-policy", workspaceSlug],
    queryFn: () => api.get(`/workspaces/${workspaceSlug}/settings/ai-policy`).then(r => r.data),
  });
}

function useSetAIPolicy(workspaceSlug: string) {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ role, action_type, requires_approval }: PolicyRowInput) =>
      api.put(`/workspaces/${workspaceSlug}/settings/ai-policy/${role}/${action_type}`, { requires_approval }),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["ai-policy", workspaceSlug] }),
    onError: () => toast.error("Failed to save policy"),
  });
}
```
