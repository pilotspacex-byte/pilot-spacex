Step 1 — Create `frontend/src/components/workspace/ai-not-configured-banner.tsx`:
```tsx
"use client";
import { useState, useEffect } from "react";
import { useQuery } from "@tanstack/react-query";
import { X, Settings } from "lucide-react";
import Link from "next/link";
import { api } from "@/lib/api";

const DISMISS_KEY = "ai_banner_dismissed";

interface Props {
  workspaceSlug: string;
  isOwner: boolean;
}

export function AiNotConfiguredBanner({ workspaceSlug, isOwner }: Props) {
  const [dismissed, setDismissed] = useState(() =>
    typeof window !== "undefined" && sessionStorage.getItem(DISMISS_KEY) === "true"
  );

  const { data } = useQuery({
    queryKey: ["ai-status", workspaceSlug],
    queryFn: () => api.get(`/workspaces/${workspaceSlug}/settings/ai-status`).then(r => r.data),
    enabled: isOwner,  // only owners see the banner; don't fetch for non-owners
    staleTime: 60 * 1000,
  });

  if (!isOwner || dismissed || data?.byok_configured !== false) return null;

  return (
    <div className="bg-amber-50 border-b border-amber-200 px-4 py-2 flex items-center justify-between text-sm">
      <span className="text-amber-800">
        AI features are disabled — no API key configured.{" "}
        <Link href={`/${workspaceSlug}/settings/api-keys`} className="underline font-medium">
          Configure a key in Settings
        </Link>
      </span>
      <button
        onClick={() => {
          sessionStorage.setItem(DISMISS_KEY, "true");
          setDismissed(true);
        }}
        className="ml-4 text-amber-600 hover:text-amber-800"
        aria-label="Dismiss"
      >
        <X className="h-4 w-4" />
      </button>
    </div>
  );
}
```
