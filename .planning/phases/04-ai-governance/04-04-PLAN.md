```python
"""AI Governance API — policy CRUD, BYOK status, artifact rollback."""
import uuid
from fastapi import APIRouter, Depends, HTTPException, Query
from pydantic import BaseModel
from pilot_space.api.v1.routers.ai_approvals import verify_workspace_admin
from pilot_space.infrastructure.database.models.audit_log import ActorType, AuditLog
from pilot_space.infrastructure.database.models.workspace_ai_policy import WorkspaceAIPolicy
from pilot_space.infrastructure.database.repositories.workspace_ai_policy_repository import WorkspaceAIPolicyRepository
from pilot_space.infrastructure.database.repositories.audit_log_repository import AuditLogRepository
from pilot_space.dependencies.auth import SessionDep, CurrentUser

router = APIRouter(tags=["ai-governance"])

# ROLLBACK_DISPATCH: maps resource_type string to the service class that can restore it.
# Scoped to issue + note for v1; extend in v2.
_ROLLBACK_RESOURCE_TYPES = {"issue", "note"}


class PolicyRowIn(BaseModel):
    requires_approval: bool


class AIStatusResponse(BaseModel):
    byok_configured: bool
    providers: list[str]


@router.get("/workspaces/{workspace_slug}/settings/ai-policy")
async def get_ai_policy(
    workspace_slug: str,
    current_user: CurrentUser,
    session: SessionDep,
) -> list[dict]:
    """Return all policy rows for the workspace. Absence of a row = hardcoded default."""
    workspace = await verify_workspace_admin(workspace_slug, current_user, session)
    repo = WorkspaceAIPolicyRepository(session)
    rows = await repo.list_for_workspace(workspace.id)
    return [{"role": r.role, "action_type": r.action_type, "requires_approval": r.requires_approval} for r in rows]


@router.put("/workspaces/{workspace_slug}/settings/ai-policy/{role}/{action_type}")
async def set_ai_policy(
    workspace_slug: str,
    role: str,
    action_type: str,
    body: PolicyRowIn,
    current_user: CurrentUser,
    session: SessionDep,
) -> dict:
    """Upsert a policy row. Owner role is not configurable."""
    if role.upper() == "OWNER":
        raise HTTPException(status_code=400, detail="Owner role policy is not configurable.")
    workspace = await verify_workspace_admin(workspace_slug, current_user, session)
    # Verify current_user is OWNER (not just admin) for write
    # (RLS policy enforces this at DB level; add application-level check for clarity)
    repo = WorkspaceAIPolicyRepository(session)
    policy = await repo.upsert(workspace.id, role.upper(), action_type, body.requires_approval)
    return {"role": policy.role, "action_type": policy.action_type, "requires_approval": policy.requires_approval}


@router.delete("/workspaces/{workspace_slug}/settings/ai-policy/{role}/{action_type}", status_code=204)
async def delete_ai_policy(
    workspace_slug: str, role: str, action_type: str,
    current_user: CurrentUser, session: SessionDep,
) -> None:
    """Delete a policy row (revert to hardcoded default)."""
    workspace = await verify_workspace_admin(workspace_slug, current_user, session)
    repo = WorkspaceAIPolicyRepository(session)
    await repo.delete(workspace.id, role.upper(), action_type)


@router.get("/workspaces/{workspace_slug}/settings/ai-status", response_model=AIStatusResponse)
async def get_ai_status(
    workspace_slug: str,
    current_user: CurrentUser,
    session: SessionDep,
) -> AIStatusResponse:
    """Return BYOK configuration status for the workspace."""
    workspace = await verify_workspace_admin(workspace_slug, current_user, session)
    # Import WorkspaceAPIKey repository (same pattern as workspace_ai_settings.py)
    from pilot_space.infrastructure.database.repositories.workspace_api_key_repository import WorkspaceAPIKeyRepository
    key_repo = WorkspaceAPIKeyRepository(session)
    keys = await key_repo.list_for_workspace(workspace.id)
    configured_providers = [k.provider for k in keys if k.is_active]
    return AIStatusResponse(
        byok_configured=len(configured_providers) > 0,
        providers=configured_providers,
    )


@router.post("/workspaces/{workspace_slug}/audit/{entry_id}/rollback")
async def rollback_ai_artifact(
    workspace_slug: str,
    entry_id: uuid.UUID,
    current_user: CurrentUser,
    session: SessionDep,
) -> dict:
    """Roll back an AI-created or AI-modified artifact to its pre-AI state."""
    workspace = await verify_workspace_admin(workspace_slug, current_user, session)
    audit_repo = AuditLogRepository(session)
    entry = await audit_repo.get_by_id(entry_id)
    if entry is None:
        raise HTTPException(status_code=404, detail="Audit entry not found.")
    if not _is_rollback_eligible(entry):
        raise HTTPException(
            status_code=400,
            detail="Entry is not rollback-eligible. Rollback applies only to AI create/update actions.",
        )
    before_state = entry.payload.get("before") or {}
    await _dispatch_rollback(entry.resource_type, entry.resource_id, before_state, session)
    # Write rollback audit entry
    current_state = entry.payload.get("after") or {}
    await audit_repo.write(
        action="ai.rollback",
        actor_type=ActorType.USER,
        actor_id=current_user.id,
        resource_type=entry.resource_type,
        resource_id=entry.resource_id,
        payload={"before": current_state, "after": before_state},
    )
    return {"status": "rolled_back", "entry_id": str(entry_id)}


def _is_rollback_eligible(entry: AuditLog) -> bool:
    return (
        entry.actor_type == ActorType.AI
        and entry.resource_type in _ROLLBACK_RESOURCE_TYPES
        and (entry.action.endswith(".create") or entry.action.endswith(".update"))
    )


async def _dispatch_rollback(
    resource_type: str,
    resource_id: uuid.UUID,
    before_state: dict,
    session,
) -> None:
    """Dispatch rollback to the appropriate service based on resource_type."""
    if resource_type == "issue":
        from pilot_space.application.services.issue_service import IssueService
        service = IssueService(session=session)
        await service.update(resource_id, before_state)
    elif resource_type == "note":
        from pilot_space.application.services.note_service import NoteService
        service = NoteService(session=session)
        await service.update(resource_id, before_state)
    else:
        raise HTTPException(status_code=422, detail=f"Rollback not supported for resource_type '{resource_type}'.")
```
