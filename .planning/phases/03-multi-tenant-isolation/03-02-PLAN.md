**Step 2: Encryption helpers** — Create `backend/src/pilot_space/infrastructure/workspace_encryption.py`:

```python
"""Workspace-level envelope encryption helpers.

Workspace key is stored encrypted with the system master Fernet key.
Content fields (notes.body, issues.description, audit_log.ai_input/ai_output)
are encrypted with the workspace key when configured.

Encryption is opt-in: get_workspace_content_key() returns None when no key is
configured, and callers must check for None before encrypting/decrypting.
"""
import base64
from cryptography.fernet import Fernet, InvalidToken
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from pilot_space.infrastructure.encryption import get_encryption_service
from pilot_space.infrastructure.database.models.workspace_encryption_key import WorkspaceEncryptionKey

def validate_workspace_key(raw_key: str) -> None:
    """Validate that raw_key is a valid 32-byte URL-safe base64 Fernet key.

    Raises ValueError with actionable message if invalid.
    """
    try:
        Fernet(raw_key.encode())
    except Exception:
        raise ValueError(
            "Key must be a 32-byte URL-safe base64 string. "
            "Use the Generate Key button to create a valid key."
        )

def store_workspace_key(raw_key: str) -> str:
    """Encrypt workspace key with master key for storage."""
    validate_workspace_key(raw_key)
    return get_encryption_service().encrypt(raw_key)

def retrieve_workspace_key(encrypted_key: str) -> str:
    """Decrypt workspace key from storage."""
    return get_encryption_service().decrypt(encrypted_key)

def encrypt_content(content: str, workspace_key: str) -> str:
    """Encrypt a content field with the workspace key."""
    f = Fernet(workspace_key.encode())
    return f.encrypt(content.encode()).decode()

def decrypt_content(ciphertext: str, workspace_key: str) -> str:
    """Decrypt a content field with the workspace key."""
    f = Fernet(workspace_key.encode())
    return f.decrypt(ciphertext.encode()).decode()

async def get_workspace_content_key(
    session: AsyncSession, workspace_id: str
) -> str | None:
    """Fetch the raw workspace encryption key for content operations.

    Returns None if no key is configured (plaintext mode).
    Raises InvalidToken if master key cannot decrypt the stored key.
    """
    result = await session.execute(
        select(WorkspaceEncryptionKey).where(
            WorkspaceEncryptionKey.workspace_id == workspace_id
        )
    )
    key_record = result.scalar_one_or_none()
    if key_record is None:
        return None
    return retrieve_workspace_key(key_record.encrypted_workspace_key)
```

**Step 1: Repository** — Create `backend/src/pilot_space/infrastructure/database/repositories/workspace_encryption_repository.py`:

```python
"""Repository for workspace encryption key records."""
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from pilot_space.infrastructure.database.models.workspace_encryption_key import WorkspaceEncryptionKey
from pilot_space.infrastructure.workspace_encryption import (
    store_workspace_key, retrieve_workspace_key, validate_workspace_key
)

class WorkspaceEncryptionRepository:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def get_key_record(self, workspace_id: str) -> WorkspaceEncryptionKey | None:
        result = await self.session.execute(
            select(WorkspaceEncryptionKey).where(
                WorkspaceEncryptionKey.workspace_id == workspace_id
            )
        )
        return result.scalar_one_or_none()

    async def upsert_key(self, workspace_id: str, raw_key: str) -> WorkspaceEncryptionKey:
        validate_workspace_key(raw_key)
        existing = await self.get_key_record(workspace_id)
        if existing:
            existing.encrypted_workspace_key = store_workspace_key(raw_key)
            existing.key_hint = raw_key[-8:]
            existing.key_version += 1
            await self.session.flush()
            return existing
        record = WorkspaceEncryptionKey(
            workspace_id=workspace_id,
            encrypted_workspace_key=store_workspace_key(raw_key),
            key_hint=raw_key[-8:],
            key_version=1,
        )
        self.session.add(record)
        await self.session.flush()
        return record
```

Endpoints (all under prefix `/workspaces/{workspace_slug}/encryption`, tagged `["encryption"]`):

- `GET /` — returns `{enabled: bool, key_hint: str|None, key_version: int|None, last_rotated: datetime|None}`. Requires ADMIN or OWNER role. Does NOT return `encrypted_workspace_key`.

- `PUT /key` — accepts `{key: str}` body. Validates key format (return 422 if invalid with message "Key must be a 32-byte URL-safe base64 string"). Requires OWNER role (settings:manage permission). Stores encrypted key, returns `{key_version: int, key_hint: str}`.

- `POST /verify` — accepts `{key: str}` body. Fetches current encrypted key, decrypts with provided key to verify they match. Returns `{verified: true, key_version: 1}` or 422 "Key does not match current workspace key". No data returned — verification only.

- `POST /generate-key` — returns `{key: str}` where key is `Fernet.generate_key().decode()`. No auth required beyond workspace membership (key is just a generated value, not stored).
