"""Super-admin bearer token dependency.

Completely separate from Supabase JWT auth. Used exclusively by /api/v1/admin/* routes.
The token is a deployment-time env var (PILOT_SPACE_SUPER_ADMIN_TOKEN), not a DB record.

Usage:
    @router.get("/admin/workspaces")
    async def list_workspaces(_: Annotated[None, Depends(get_super_admin)]):
        ...
"""
from typing import Annotated
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer
from pilot_space.config import get_settings

_bearer = HTTPBearer(auto_error=False)  # auto_error=False so we can return 401 not 403

async def get_super_admin(
    credentials: Annotated[HTTPAuthorizationCredentials | None, Depends(_bearer)],
) -> None:
    """Validate super-admin bearer token. Returns None on success, raises 401 on failure."""
    settings = get_settings()
    expected = settings.pilot_space_super_admin_token

    if credentials is None or expected is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Super-admin access requires a valid PILOT_SPACE_SUPER_ADMIN_TOKEN",
            headers={"WWW-Authenticate": "Bearer"},
        )
    if credentials.credentials != expected.get_secret_value():
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid super-admin token",
            headers={"WWW-Authenticate": "Bearer"},
        )
```

"""
Super-admin operator dashboard routes.

These routes bypass workspace JWT auth entirely. Access is controlled by
PILOT_SPACE_SUPER_ADMIN_TOKEN bearer token (see dependencies/admin.py).

Uses service_role DB connection — RLS is bypassed for cross-workspace queries.
All responses are read-only; no workspace data is mutated via admin routes.
"""
import logging
from typing import Annotated
from fastapi import APIRouter, Depends, Query
from sqlalchemy import select, func, text
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker
from pilot_space.config import get_settings
from pilot_space.dependencies.admin import get_super_admin
from pilot_space.infrastructure.database.models.workspace import Workspace
from pilot_space.infrastructure.database.models.workspace_member import WorkspaceMember
from pilot_space.infrastructure.database.models.audit_log import AuditLog

logger = logging.getLogger(__name__)

router = APIRouter(tags=["super-admin"])

# Lazy-init service_role session factory (same pattern as SCIM)
_admin_session_factory: async_sessionmaker | None = None

def _get_admin_session_factory() -> async_sessionmaker:
    global _admin_session_factory
    if _admin_session_factory is None:
        settings = get_settings()
        engine = create_async_engine(
            str(settings.database_service_role_url),
            pool_pre_ping=True,
        )
        _admin_session_factory = async_sessionmaker(engine, expire_on_commit=False)
    return _admin_session_factory
```
