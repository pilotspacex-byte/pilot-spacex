```python
"""ArtifactRepository — data access for the artifacts table."""

from __future__ import annotations

from datetime import datetime
from typing import TYPE_CHECKING
from uuid import UUID

from sqlalchemy import and_, delete, select, update

from pilot_space.infrastructure.database.models.artifact import Artifact

if TYPE_CHECKING:
    from sqlalchemy.ext.asyncio import AsyncSession


class ArtifactRepository:
    def __init__(self, session: AsyncSession) -> None:
        self.session = session

    async def create(self, artifact: Artifact) -> Artifact:
        self.session.add(artifact)
        await self.session.flush()
        await self.session.refresh(artifact)
        return artifact

    async def get_by_id(self, artifact_id: UUID) -> Artifact | None:
        result = await self.session.execute(
            select(Artifact).where(Artifact.id == artifact_id)
        )
        return result.scalar_one_or_none()

    async def list_by_project(
        self, workspace_id: UUID, project_id: UUID
    ) -> list[Artifact]:
        result = await self.session.execute(
            select(Artifact)
            .where(
                and_(
                    Artifact.workspace_id == workspace_id,
                    Artifact.project_id == project_id,
                    Artifact.status == "ready",
                )
            )
            .order_by(Artifact.created_at.desc())
        )
        return list(result.scalars().all())

    async def update_status(self, artifact_id: UUID, status: str) -> None:
        await self.session.execute(
            update(Artifact)
            .where(Artifact.id == artifact_id)
            .values(status=status)
        )

    async def delete(self, artifact_id: UUID) -> bool:
        result = await self.session.execute(
            delete(Artifact)
            .where(Artifact.id == artifact_id)
            .returning(Artifact.id)
        )
        return result.scalar_one_or_none() is not None

    async def delete_stale_pending(self, older_than: datetime) -> list[Artifact]:
        """Fetch and delete pending_upload records older than `older_than`."""
        result = await self.session.execute(
            select(Artifact).where(
                and_(
                    Artifact.status == "pending_upload",
                    Artifact.created_at < older_than,
                )
            )
        )
        stale = list(result.scalars().all())
        if stale:
            ids = [a.id for a in stale]
            await self.session.execute(
                delete(Artifact).where(Artifact.id.in_(ids))
            )
        return stale


__all__ = ["ArtifactRepository"]
```

"""ArtifactUploadService — DB-first file upload for note artifacts."""

from __future__ import annotations

import pathlib
from typing import TYPE_CHECKING
from uuid import UUID, uuid4

from pilot_space.infrastructure.database.models.artifact import Artifact
from pilot_space.infrastructure.logging import get_logger

if TYPE_CHECKING:
    from sqlalchemy.ext.asyncio import AsyncSession
    from pilot_space.api.v1.schemas.artifacts import ArtifactResponse
    from pilot_space.infrastructure.database.repositories.artifact_repository import ArtifactRepository
    from pilot_space.infrastructure.storage.client import SupabaseStorageClient

logger = get_logger(__name__)

_BUCKET = "note-artifacts"
_MAX_BYTES = 10 * 1024 * 1024  # 10MB — flat limit, no per-type differentiation

_ALLOWED_EXTENSIONS: frozenset[str] = frozenset({
    ".md", ".csv", ".json", ".txt", ".xlsx", ".xls",
    ".py", ".js", ".ts", ".jsx", ".tsx", ".html", ".css", ".scss",
    ".go", ".rs", ".java", ".c", ".cpp", ".h", ".yaml", ".yml",
    ".toml", ".xml", ".sql", ".sh", ".rb", ".php", ".swift", ".kt",
    ".png", ".jpg", ".jpeg", ".gif", ".webp", ".svg", ".bmp", ".ico",
})

# Extension → known MIME type prefixes for cross-check (not exhaustive; advisory only)
_EXT_TO_MIME_PREFIX: dict[str, str] = {
    ".png": "image/", ".jpg": "image/", ".jpeg": "image/",
    ".gif": "image/", ".webp": "image/", ".svg": "image/",
    ".bmp": "image/", ".ico": "image/",
    ".html": "text/", ".css": "text/", ".csv": "text/",
    ".md": "text/", ".txt": "text/", ".sh": "text/",
}


def _validate_file_type(filename: str, content_type: str) -> None:
    ext = pathlib.Path(filename).suffix.lower()
    if ext not in _ALLOWED_EXTENSIONS:
        raise ValueError("UNSUPPORTED_FILE_TYPE")
    # Optional MIME cross-check: if we have a known prefix for this ext, enforce it
    expected_prefix = _EXT_TO_MIME_PREFIX.get(ext)
    if expected_prefix and not content_type.startswith(expected_prefix):
        raise ValueError("MIME_MISMATCH")


def _validate_file_size(file_data: bytes) -> None:
    if len(file_data) == 0:
        raise ValueError("EMPTY_FILE")
    if len(file_data) > _MAX_BYTES:
        raise ValueError("FILE_TOO_LARGE")


class ArtifactUploadService:
    def __init__(
        self,
        session: AsyncSession,
        storage_client: SupabaseStorageClient,
        artifact_repo: ArtifactRepository,
    ) -> None:
        self._session = session
        self._storage = storage_client
        self._repo = artifact_repo

    async def upload(
        self,
        file_data: bytes,
        filename: str,
        content_type: str,
        workspace_id: UUID,
        project_id: UUID,
        user_id: UUID,
    ) -> "ArtifactResponse":
        # Validate first — no I/O until these pass
        _validate_file_type(filename, content_type)
        _validate_file_size(file_data)

        artifact_id = uuid4()
        # Key format: {workspace_id}/{project_id}/{artifact_id}/{filename}
        # IMPORTANT: no bucket prefix in key — bucket passed separately to storage client
        storage_key = f"{workspace_id}/{project_id}/{artifact_id}/{filename}"

        # Step 1 (DB-first): create record as pending_upload
        artifact = Artifact(
            id=artifact_id,
            workspace_id=workspace_id,
            project_id=project_id,
            user_id=user_id,
            filename=filename,
            mime_type=content_type,
            size_bytes=len(file_data),
            storage_key=storage_key,
            status="pending_upload",
        )
        await self._repo.create(artifact)
        logger.info("artifact_db_record_created", artifact_id=str(artifact_id))

        # Step 2: upload to storage (may fail — DB record stays pending_upload for cleanup)
        await self._storage.upload_object(
            bucket=_BUCKET,
            key=storage_key,
            data=file_data,
            content_type=content_type,
        )
        logger.info("artifact_uploaded_to_storage", artifact_id=str(artifact_id),
                    size_bytes=len(file_data))

        # Step 3: mark ready only after successful storage upload
        await self._repo.update_status(artifact_id, "ready")

        from pilot_space.api.v1.schemas.artifacts import ArtifactResponse
        return ArtifactResponse(
            id=artifact_id,
            project_id=project_id,
            user_id=user_id,
            filename=filename,
            mime_type=content_type,
            size_bytes=len(file_data),
            storage_key=storage_key,
            status="ready",
            created_at=artifact.created_at,
        )

    async def delete(self, artifact_id: UUID, user_id: UUID, workspace_id: UUID) -> None:
        artifact = await self._repo.get_by_id(artifact_id)
        if artifact is None:
            raise ValueError("NOT_FOUND")
        if artifact.workspace_id != workspace_id:
            raise ValueError("NOT_FOUND")
        if artifact.user_id != user_id:
            raise PermissionError("FORBIDDEN")
        await self._storage.delete_object(bucket=_BUCKET, key=artifact.storage_key)
        await self._repo.delete(artifact_id)
        logger.info("artifact_deleted", artifact_id=str(artifact_id))


__all__ = ["ArtifactUploadService"]
```

"""artifact_cleanup — deletes stale pending_upload artifact records and their storage objects."""

from __future__ import annotations

from datetime import UTC, datetime, timedelta
from typing import TYPE_CHECKING

from pilot_space.infrastructure.database.repositories.artifact_repository import ArtifactRepository
from pilot_space.infrastructure.logging import get_logger

if TYPE_CHECKING:
    from sqlalchemy.ext.asyncio import AsyncSession
    from pilot_space.infrastructure.storage.client import SupabaseStorageClient

logger = get_logger(__name__)

_BUCKET = "note-artifacts"
_STALE_AFTER_HOURS = 24


async def run_artifact_cleanup(
    session: AsyncSession,
    storage_client: SupabaseStorageClient,
) -> int:
    """Delete pending_upload artifact records older than 24h and their storage objects.

    Returns:
        Number of artifact records deleted.
    """
    cutoff = datetime.now(UTC) - timedelta(hours=_STALE_AFTER_HOURS)
    repo = ArtifactRepository(session)
    stale = await repo.delete_stale_pending(older_than=cutoff)

    deleted_count = 0
    for artifact in stale:
        try:
            await storage_client.delete_object(bucket=_BUCKET, key=artifact.storage_key)
            deleted_count += 1
            logger.info("artifact_cleanup_deleted", artifact_id=str(artifact.id),
                        storage_key=artifact.storage_key)
        except Exception as exc:  # noqa: BLE001
            # Non-fatal — log and continue to next artifact
            logger.warning("artifact_cleanup_storage_delete_failed",
                          artifact_id=str(artifact.id), error=str(exc))
            deleted_count += 1  # DB record already deleted; count it

    logger.info("artifact_cleanup_complete", deleted_count=deleted_count)
    return deleted_count


__all__ = ["run_artifact_cleanup"]
```
