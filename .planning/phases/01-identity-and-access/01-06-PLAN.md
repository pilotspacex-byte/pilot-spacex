SCIM Bearer token auth pattern (from RESEARCH.md):
```python
async def get_scim_workspace(
    request: Request,
    workspace_slug: str,
    session: SessionDep,
) -> Workspace:
    auth_header = request.headers.get("Authorization", "")
    if not auth_header.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="Missing SCIM bearer token")
    token = auth_header[7:]
    token_hash = hashlib.sha256(token.encode()).hexdigest()
    workspace = await workspace_repo.get_by_slug(workspace_slug, session)
    settings = workspace.settings or {}
    if settings.get("scim_token_hash") != token_hash:
        raise HTTPException(status_code=401, detail="Invalid SCIM bearer token")
    return workspace
```
