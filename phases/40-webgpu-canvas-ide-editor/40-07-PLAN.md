         ```typescript
         export function useMonacoNote(options: {
           noteId: string;
           editor: monaco.editor.IStandaloneCodeEditor | null;
           monacoInstance: typeof monaco | null;
           content: string;
           collabEnabled: boolean;
           ghostTextFetcher: GhostTextFetcher;
           memberFetcher?: MemberFetcher;
         }) {
           const theme = useMonacoTheme(options.monacoInstance);
           const viewZonePortals = useMonacoViewZones(options.editor, options.content);
           useMonacoGhostText(options.monacoInstance, options.editor, options.ghostTextFetcher, options.noteId);
           useMonacoSlashCmd(options.monacoInstance, options.editor, options.memberFetcher);
           const collab = useMonacoCollab(
             options.editor,
             options.editor?.getModel() ?? null,
             options.noteId,
             options.collabEnabled,
           );

           // Apply markdown decorations
           useEffect(() => {
             if (!options.editor || !options.monacoInstance) return;
             const disposable = applyMarkdownDecorations(options.editor, options.monacoInstance);
             return () => disposable.dispose();
           }, [options.editor, options.monacoInstance]);

           return { theme, viewZonePortals, collab };
         }
         ```
