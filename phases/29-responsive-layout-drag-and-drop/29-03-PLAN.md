       - Build a full node map (not just flat items) for subtree height lookup: create `fullNodeMap` from recursive walk of `treeNodes` mapping id -> PageTreeNode
       - Add `handleDragOver` handler:
         ```
         const handleDragOver = (event: DragOverEvent) => {
           const { active, over } = event;
           if (!over || active.id === over.id) { setInvalidDropTargetId(null); return; }
           const activeFullNode = fullNodeMap.get(active.id as string);
           const overMeta = nodeMap.get(over.id as string);
           if (!activeFullNode || !overMeta) { setInvalidDropTargetId(null); return; }
           // Re-parent case: active becomes child of over's parent
           // The new depth of active = overMeta.depth (same level as over)
           // But if parentIds differ, active becomes child of over -> newActiveDepth = overMeta.depth + 1...
           // Actually: in handleDragEnd, different parentId means movePage to over's parent.
           // So active would sit at over's depth level. The deepest descendant would be at:
           // overMeta.depth + getSubtreeHeight(activeFullNode)
           // For same-parent reorder: depth doesn't change, always valid.
           const activeParentId = nodeMap.get(active.id as string)?.parentId;
           const overParentId = overMeta.parentId;
           if (activeParentId === overParentId) {
             // Same parent reorder — depth unchanged, always valid
             setInvalidDropTargetId(null);
             return;
           }
           // Re-parent: active moves to over's parent, so active.depth becomes overMeta.depth
           const newDeepestDepth = overMeta.depth + getSubtreeHeight(activeFullNode);
           if (newDeepestDepth > 2) {
             setInvalidDropTargetId(over.id as string);
           } else {
             setInvalidDropTargetId(null);
           }
         };
         ```
