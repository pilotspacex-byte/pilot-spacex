**Parse function** (useMemo, called once per data+activeSheet change):
```typescript
function parseSheet(workbook: XLSX.WorkBook, sheetName: string): ParsedSheet {
  const sheet = workbook.Sheets[sheetName];
  if (!sheet) return { headers: [], rows: [], totalRows: 0, isTruncated: false };
  const raw = XLSX.utils.sheet_to_json<string[]>(sheet, { header: 1 });
  const headers = (raw[0] ?? []).map(String);
  const dataRows = raw.slice(1, 501).map(row => row.map(cell => cell != null ? String(cell) : ''));
  return {
    headers,
    rows: dataRows,
    totalRows: raw.length - 1,
    isTruncated: raw.length - 1 > 500,
  };
}
```
