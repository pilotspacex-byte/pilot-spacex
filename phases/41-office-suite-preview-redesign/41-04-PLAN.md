```typescript
React.useEffect(() => {
  function handleKeyDown(e: KeyboardEvent) {
    // Skip if user is in an input/textarea
    if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return;
    if (e.key === 'ArrowLeft') {
      setCurrentSlide(s => Math.max(0, s - 1));
    } else if (e.key === 'ArrowRight') {
      setCurrentSlide(s => Math.min(totalSlides - 1, s + 1));
    } else if (e.key === 'f' || e.key === 'F') {
      toggleFullscreen();
    }
  }
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [totalSlides]);
```

**Thumbnail rendering approach:**
Since rendering individual slide thumbnails requires the PPTX rendering library, use a simplified approach:
1. Each thumbnail is a small placeholder with slide number
2. Active thumbnail has accent ring
3. Use IntersectionObserver for lazy visibility
4. Auto-scroll active thumbnail into view

**Implementation:**
```typescript
export function PptxThumbnailStrip({
  data,
  currentSlide,
  totalSlides,
  onSlideSelect,
}: PptxThumbnailStripProps) {
  const activeRef = React.useRef<HTMLButtonElement>(null);
  const listRef = React.useRef<HTMLDivElement>(null);

  // Auto-scroll active thumbnail into view
  React.useEffect(() => {
    activeRef.current?.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  }, [currentSlide]);

  // Keyboard navigation within thumbnail strip
  const handleKeyDown = React.useCallback((e: React.KeyboardEvent) => {
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      onSlideSelect(Math.min(totalSlides - 1, currentSlide + 1));
    } else if (e.key === 'ArrowUp') {
      e.preventDefault();
      onSlideSelect(Math.max(0, currentSlide - 1));
    }
  }, [currentSlide, totalSlides, onSlideSelect]);

  if (totalSlides === 0) return null;

  return (
    <div className="w-[156px] shrink-0 border-r border-border/60 bg-muted/20 flex flex-col">
      <div className="px-3 py-2 border-b border-border/60">
        <p className="text-[10px] font-semibold uppercase tracking-wider text-muted-foreground/70">
          Slides
        </p>
      </div>
      <ScrollArea className="flex-1">
        <div
          ref={listRef}
          role="listbox"
          aria-orientation="vertical"
          aria-label="Slide thumbnails"
          onKeyDown={handleKeyDown}
          className="p-2 space-y-2"
        >
          {Array.from({ length: totalSlides }, (_, i) => {
            const isActive = i === currentSlide;
            return (
              <button
                key={i}
                ref={isActive ? activeRef : undefined}
                role="option"
                aria-selected={isActive}
                onClick={() => onSlideSelect(i)}
                className={cn(
                  'flex items-start gap-2 w-full rounded-md p-1 transition-all',
                  'focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-primary',
                  isActive ? 'ring-2 ring-primary bg-background' : 'hover:bg-muted/50'
                )}
              >
                {/* Slide number */}
                <span className={cn(
                  'shrink-0 w-4 text-right',
                  isActive
                    ? 'text-[10px] font-semibold text-primary'
                    : 'text-[10px] text-muted-foreground'
                )}>
                  {i + 1}
                </span>
                {/* Thumbnail placeholder — 16:9 aspect ratio */}
                <div className={cn(
                  'flex-1 aspect-video rounded border bg-white flex items-center justify-center',
                  isActive ? 'border-primary/30' : 'border-border/60'
                )}>
                  <span className="text-[10px] text-muted-foreground/50">Slide {i + 1}</span>
                </div>
              </button>
            );
          })}
        </div>
      </ScrollArea>
    </div>
  );
}
```
