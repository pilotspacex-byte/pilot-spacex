    **API Client** (`artifact-annotations.ts`):
    Create `annotationsApi` object following the pattern from `artifacts.ts`:

    ```typescript
    import { apiClient } from './client';

    export interface AnnotationResponse {
      id: string;
      artifact_id: string;
      slide_index: number;
      content: string;
      user_id: string;
      created_at: string;
      updated_at: string;
    }

    export interface AnnotationListResponse {
      annotations: AnnotationResponse[];
      total: number;
    }

    export const annotationsApi = {
      list(workspaceId: string, projectId: string, artifactId: string, slideIndex?: number): Promise<AnnotationListResponse> {
        const params = slideIndex !== undefined ? `?slide_index=${slideIndex}` : '';
        return apiClient.get<AnnotationListResponse>(
          `/workspaces/${workspaceId}/projects/${projectId}/artifacts/${artifactId}/annotations${params}`
        );
      },
      create(workspaceId: string, projectId: string, artifactId: string, data: { slide_index: number; content: string }): Promise<AnnotationResponse> {
        return apiClient.post<AnnotationResponse>(
          `/workspaces/${workspaceId}/projects/${projectId}/artifacts/${artifactId}/annotations`,
          data
        );
      },
      update(workspaceId: string, projectId: string, artifactId: string, annotationId: string, data: { content: string }): Promise<AnnotationResponse> {
        return apiClient.put<AnnotationResponse>(
          `/workspaces/${workspaceId}/projects/${projectId}/artifacts/${artifactId}/annotations/${annotationId}`,
          data
        );
      },
      delete(workspaceId: string, projectId: string, artifactId: string, annotationId: string): Promise<void> {
        return apiClient.delete<void>(
          `/workspaces/${workspaceId}/projects/${projectId}/artifacts/${artifactId}/annotations/${annotationId}`
        );
      },
    };
    ```

    **TanStack Query Hook** (`usePptxAnnotations.ts`):
    Create `usePptxAnnotations(params: { workspaceId: string; projectId: string; artifactId: string; slideIndex: number })` hook.

    Query keys: `['artifact-annotations', artifactId, slideIndex]`

    Return:
    - `annotations`: AnnotationResponse[] from query data
    - `total`: number
    - `isLoading`: boolean
    - `createAnnotation`: useMutation with optimistic update (temp ID `temp-${Date.now()}`, add to list immediately)
    - `updateAnnotation`: useMutation with optimistic update (replace content in cache)
    - `deleteAnnotation`: useMutation with optimistic update (remove from cache immediately)

    Optimistic pattern per RESEARCH.md Pattern 5:
    - `onMutate`: cancel queries, snapshot previous, set optimistic data
    - `onError`: rollback to previous
    - `onSettled`: invalidate queries

    The hook must be a plain React hook (NOT observer) to avoid flushSync issues with React 19.
