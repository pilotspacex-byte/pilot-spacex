The docs detail page at `[workspaceSlug]/docs/[slug]/page.tsx` is a Server Component that uses `fs.readFileSync` to load markdown content from disk. This is incompatible with `output: 'export'` because:
- Static export cannot use `fs` or `path` modules at request time
- The doubly-dynamic route `[workspaceSlug]/docs/[slug]` cannot provide `generateStaticParams` for arbitrary workspace slugs

The fix is to create a build-time manifest that bundles all markdown content as importable strings, then convert the docs pages to client components.

**Step 1: Create docs-manifest.ts**

Create `frontend/src/features/docs/lib/docs-manifest.ts` that imports all markdown files as raw strings at build time:

```typescript
/**
 * Build-time docs content manifest.
 *
 * Imports all markdown content files as raw strings so they are available
 * in both standalone (SSR) and static export (Tauri) builds without
 * requiring fs.readFileSync at runtime.
 */

// Raw string imports — bundled at build time by webpack/turbopack
import overviewContent from '../content/overview.md';
import gettingStartedContent from '../content/getting-started.md';
import architectureContent from '../content/architecture.md';
import aiAgentContent from '../content/ai-agent.md';
import noteFirstContent from '../content/note-first.md';
import apiReferenceContent from '../content/api-reference.md';

/** Map of doc file slug to raw markdown content string */
export const docsManifest: Record<string, string> = {
  overview: overviewContent,
  'getting-started': gettingStartedContent,
  architecture: architectureContent,
  'ai-agent': aiAgentContent,
  'note-first': noteFirstContent,
  'api-reference': apiReferenceContent,
};
```

NOTE: Next.js with webpack/turbopack can import `.md` files as raw strings. If this fails, add a webpack `raw-loader` config or use `?raw` suffix. Verify by running the build. If the import fails, add to `next.config.ts`:

```typescript
  config.module.rules.push({ test: /\.md$/, type: 'asset/source' });
  return config;
},
```

**Step 2: Convert docs detail page to client component**

Replace `frontend/src/app/(workspace)/[workspaceSlug]/docs/[slug]/page.tsx` with a client-side version that uses the manifest instead of fs.readFileSync:

```typescript
'use client';

/**
 * Docs Detail Route — renders a single documentation page from markdown.
 *
 * Route: /[workspaceSlug]/docs/[slug]
 * Content is loaded from the build-time manifest (no server-side file I/O).
 * Headings are extracted client-side for table of contents.
 */

import { notFound } from 'next/navigation';
import { useParams } from 'next/navigation';
import { DocsPage, docsBySlug } from '@/features/docs';
import { extractHeadings } from '@/features/docs/lib/markdown-headings';
import { docsManifest } from '@/features/docs/lib/docs-manifest';

export default function DocsDetailPage() {
  const params = useParams<{ workspaceSlug: string; slug: string }>();
  const slug = params.slug;

  const doc = docsBySlug.get(slug);
  if (!doc) {
    notFound();
  }

  const content = docsManifest[doc.file];
  if (!content) {
    notFound();
  }

  const headings = extractHeadings(content);

  return <DocsPage slug={slug} content={content} headings={headings} />;
}
```

**Step 3: Convert docs index page to client component**

Replace `frontend/src/app/(workspace)/[workspaceSlug]/docs/page.tsx` with a client-side redirect:

```typescript
'use client';

/**
 * Docs Index Route — redirects to the default documentation page.
 *
 * Route: /[workspaceSlug]/docs
 */

import { useParams, useRouter } from 'next/navigation';
import { useEffect } from 'react';
import { defaultDocSlug } from '@/features/docs';

export default function DocsIndexPage() {
  const params = useParams<{ workspaceSlug: string }>();
  const router = useRouter();

  useEffect(() => {
    if (params.workspaceSlug) {
      router.replace(`/${params.workspaceSlug}/docs/${defaultDocSlug}`);
    }
  }, [params.workspaceSlug, router]);

  return null;
}
```
