**1. Create API client** (`artifact-annotations.ts`):
```typescript
import { apiClient } from './client';

export interface ArtifactAnnotation {
  id: string;
  artifactId: string;
  slideIndex: number;
  content: string;
  userId: string;
  workspaceId: string;
  createdAt: string;
  updatedAt: string;
}

interface AnnotationListResponse {
  annotations: ArtifactAnnotation[];
  total: number;
}

export const annotationApi = {
  list(wid: string, pid: string, aid: string, slideIndex: number): Promise<ArtifactAnnotation[]> {
    return apiClient.get<AnnotationListResponse>(
      `/workspaces/${wid}/projects/${pid}/artifacts/${aid}/annotations?slideIndex=${slideIndex}`
    ).then(res => res.annotations);
  },
  create(wid: string, pid: string, aid: string, body: { slideIndex: number; content: string }): Promise<ArtifactAnnotation> {
    return apiClient.post(`/workspaces/${wid}/projects/${pid}/artifacts/${aid}/annotations`, body);
  },
  update(wid: string, pid: string, aid: string, annotationId: string, body: { content: string }): Promise<ArtifactAnnotation> {
    return apiClient.put(`/workspaces/${wid}/projects/${pid}/artifacts/${aid}/annotations/${annotationId}`, body);
  },
  delete(wid: string, pid: string, aid: string, annotationId: string): Promise<void> {
    return apiClient.delete(`/workspaces/${wid}/projects/${pid}/artifacts/${aid}/annotations/${annotationId}`);
  },
};
```

**2. Create TanStack Query hooks** (`use-slide-annotations.ts`):
- Query key factory: `annotationKeys.all`, `annotationKeys.slide(artifactId, slideIndex)`
- `useSlideAnnotations(workspaceId, projectId, artifactId, slideIndex)` — useQuery returning `ArtifactAnnotation[]`, enabled when all params truthy, staleTime 30s
- `useCreateAnnotation(workspaceId, projectId, artifactId)` — useMutation with optimistic add to list cache. mutationFn calls `annotationApi.create`. onMutate: cancel queries for current slideIndex, snapshot, optimistically append new annotation (generate temp UUID). onError: rollback. onSettled: invalidate.
- `useUpdateAnnotation(workspaceId, projectId, artifactId)` — useMutation accepting `{ annotationId, content, slideIndex }`. Optimistic update in cache. onError: rollback + `toast.error('Update failed. Please try again.')`. onSettled: invalidate.
- `useDeleteAnnotation(workspaceId, projectId, artifactId)` — useMutation accepting `{ annotationId, slideIndex }`. Optimistic remove from cache. onError: rollback + `toast.error('Delete failed. Please try again.')`. onSettled: invalidate.
- Follow exact optimistic update pattern from `use-delete-artifact.ts`: cancelQueries, snapshot, setQueryData, return context, rollback on error
