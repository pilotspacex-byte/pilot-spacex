    3. Debounce search input at 300ms using a useEffect + setTimeout pattern:
       ```typescript
       React.useEffect(() => {
         const timer = setTimeout(() => setSearchTerm(searchInput), 300);
         return () => clearTimeout(timer);
       }, [searchInput]);
       ```
       Do NOT add a lodash/debounce dependency — the setTimeout pattern is sufficient.

    4. Compute match count with useMemo:
       ```typescript
       const matchCount = React.useMemo(() => {
         if (!searchTerm || !sheetData) return 0;
         const term = searchTerm.toLowerCase();
         let count = 0;
         for (const row of sheetData.rows) {
           for (const cell of row) {
             if (String(cell).toLowerCase().includes(term)) count++;
           }
         }
         // Also check headers
         for (const h of sheetData.headers) {
           if (String(h).toLowerCase().includes(term)) count++;
         }
         return count;
       }, [searchTerm, sheetData]);
       ```

    5. Add search toolbar above the table (below truncation banner, above ScrollArea):
       ```tsx
       <div className="flex items-center gap-2 px-3 py-2 border-b shrink-0">

           value={searchInput}
           onChange={(e) => setSearchInput(e.target.value)}
           className="h-8 max-w-xs text-sm"
         />
         {searchTerm && (
           <span className="text-xs text-muted-foreground whitespace-nowrap">
             {matchCount} {matchCount === 1 ? 'match' : 'matches'}
           </span>
         )}

    6. Highlight matching cells — create a helper to render cell content:

       function highlightCell(value: string, term: string): React.ReactNode {
         if (!term) return value;
         const idx = value.toLowerCase().indexOf(term.toLowerCase());
         if (idx === -1) return value;
         return (
           <>
             {value.slice(0, idx)}
             <mark className="bg-primary/20 rounded-sm px-0.5">{value.slice(idx, idx + term.length)}</mark>
             {value.slice(idx + term.length)}
           </>
         );
       }
       ```
